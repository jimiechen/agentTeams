async function clickRetryButton(trae: any) {
  return await trae.evaluate(`(() => {
    const result = { found: false, clicked: false, method: '' };

    // 路径1: aria-label 精确查找（最可靠）
    let btn = document.querySelector('button[aria-label="重试"]');
    if (btn) {
      btn.click();
      return { ...result, found: true, clicked: true, method: 'aria-label' };
    }

    // 路径2: 文本内容查找（兜底）
    const allBtns = document.querySelectorAll('button');
    for (const b of allBtns) {
      if (b.textContent?.trim() === '重试') {
        b.click();
        return { ...result, found: true, clicked: true, method: 'text-content' };
      }
    }

    // 路径3: 在"手动终止输出"消息旁查找（DEVCLI 特有场景）
    const turns = document.querySelectorAll('.chat-turn');
    for (let i = turns.length - 1; i >= 0; i--) {
      if (turns[i].textContent.includes('手动终止输出')) {
        btn = turns[i].querySelector('button[aria-label="重试"]');
        if (btn) {
          btn.click();
          return { ...result, found: true, clicked: true, method: 'previous-message' };
        }
        break;
      }
    }

    return result;
  })()`);
}

## **四、完整自愈状态机（针对当前日志场景）**

### **4.1 `frozen` 模式的完整处理流程**

```
mode=frozen 触发
      ↓
Step 1: 第一次状态检查（before_recovery）
      ↓
status=interrupted?
  ├── YES → 执行 clickRetryButton（三路查找）
  │           ↓
  │         等待 3 秒
  │           ↓
  │         第二次状态检查（after_retry）
  │           ├── status=in_progress → ✅ 恢复成功
  │           │     写入 heartbeat.jsonl: recovery=success
  │           │     发送飞书 info 通知
  │           │
  │           └── status=interrupted → 重试计数 +1
  │                 ├── count < 3 → 等待 30 秒后再次尝试
  │                 └── count >= 3 → 进入 interrupted 终态
  │                       发送飞书 critical 告警
  │
  └── NO（isStuck=true）→ 点击"后台运行"（terminal-hang 路径）
```

### **4.2 重试计数与防止无限循环**

```typescript
class FrozenRecoveryHandler {
  private retryCount = 0;
  private readonly MAX_RETRIES = 3;
  private readonly RETRY_INTERVAL_MS = 30_000;

  async handle(signal: HeartbeatSignal): Promise<RecoveryResult> {
    if (this.retryCount >= this.MAX_RETRIES) {
      // 重试耗尽，进入终态，不再尝试
      await this.sendCriticalAlert(signal.workspace);
      return { action: 'give_up', reason: 'max_retries_exceeded' };
    }

    const result = await clickRetryButton(this.trae);
    this.retryCount++;

    if (result.clicked) {
      await sleep(3000);
      const check = await checkTaskStatus(this.trae, signal.workspace);
      if (check.status === 'in_progress') {
        this.retryCount = 0; // 成功后重置计数
        return { action: 'recovered', method: result.method };
      }
    }

    // 本次未成功，等待下一个心跳周期
    return { action: 'pending', retryCount: this.retryCount };
  }
}
```
