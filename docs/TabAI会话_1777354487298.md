### **下面给你两份可以直接 `git apply` 的完整测试脚本，`test-probe-signals.ts` 覆盖 5 信号注入与回归自检，`test-stuck-scenarios.ts` 覆盖三类失败的真实诱导与恢复验证，两份脚本共享同一套 `TestReporter` 输出统一的 PASS/FAIL 报告到 `runs/test-report-*.jsonl`**

两个脚本都遵循你现有工程约定——走 `CDPClient` + `loadConfig`，不引入新依赖；所有注入型测试都有 `cleanup` 守护，不污染真实 Trae 会话；所有真实 prompt 测试都走飞书通道，端到端验证链路。文件放在 `mvp-runner/scripts/` 目录下，直接 `npx tsx scripts/test-probe-signals.ts` 就能跑。

class TestReporter {
  private path: string;
  private results: any[] = [];

  constructor(private kind: string) {
    mkdirSync(path.resolve('./runs'), { recursive: true });
    const ts = new Date().toISOString().replace(/[:.]/g, '-');
    this.path = path.resolve(`./runs/test-${kind}-${ts}.jsonl`);
    writeFileSync(this.path, '');
    log('report file: %s', this.path);
  }

  record(entry: any) {
    this.results.push(entry);
    appendFileSync(this.path, JSON.stringify(entry) + '
');
  }

  summary() {
    const total = this.results.length;
    const passed = this.results.filter(r => r.status === 'PASS').length;
    const failed = total - passed;
    const rate = total > 0 ? ((passed / total) * 100).toFixed(1) : '0.0';

    console.log('
' + '='.repeat(60));
    console.log(`Test Summary: `);
    console.log(`  Total:   `);
    console.log(`  Passed:  ✅`);
    console.log(`  Failed:  ❌`);
    console.log(`  Rate:    %`);
    console.log(`  Report:  `);
    console.log('='.repeat(60));

    if (failed > 0) {
      console.log('
Failed cases:');
      this.results
        .filter(r => r.status !== 'PASS')
        .forEach(r => console.log(`  ❌ : `));
    }

    return { total, passed, failed, rate: parseFloat(rate) };
  }

  get passed() {
    return this.results.every(r => r.status === 'PASS');
  }
}

async function main() {
  const cfg = loadConfig();
  const cdp = new CDPClient(cfg.cdp.host, cfg.cdp.port);

  console.log('
🔌 Connecting to Trae CDP...');
  await cdp.connect();
  console.log('✅ Connected to %s:%d', cfg.cdp.host, cfg.cdp.port);

  const reporter = new TestReporter('probe-signals');

  for (const c of CASES) {
    console.log(`
▶ []`);
    console.log(`   `);
    const startedAt = Date.now();

    try {
      // 1. 注入
      if (c.inject.trim() !== 'true') {
        await cdp.evaluate(c.inject);
      }
      await sleep(c.settleMs ?? 500);

      // 2. 断言注入后
      const snapAfterInject = await captureSnapshot(cdp);
      const injectResult = c.assertAfterInject(snapAfterInject);

      if (!injectResult.ok) {
        console.log(`  ❌ inject assertion FAILED: `);
        console.log(`     snapshot: %o`, snapAfterInject);
        reporter.record({
          name: c.name,
          phase: 'after-inject',
          status: 'FAIL',
          reason: injectResult.reason,
          snapshot: snapAfterInject,
          durationMs: Date.now() - startedAt,
        });
        // 即便断言失败，也要执行 cleanup 防止污染
        if (c.cleanup.trim() !== 'true') await cdp.evaluate(c.cleanup);
        continue;
      }

      // 3. 清理
      if (c.cleanup.trim() !== 'true') {
        await cdp.evaluate(c.cleanup);
      }
      await sleep(c.settleMs ?? 500);

      // 4. 断言清理后
      const snapAfterCleanup = await captureSnapshot(cdp);
      const cleanupResult = c.assertAfterCleanup(snapAfterCleanup);

      if (!cleanupResult.ok) {
        console.log(`  ⚠️  cleanup assertion FAILED: `);
        reporter.record({
          name: c.name,
          phase: 'after-cleanup',
          status: 'FAIL',
          reason: cleanupResult.reason,
          snapshot: snapAfterCleanup,
          durationMs: Date.now() - startedAt,
        });
        continue;
      }

      console.log(`  ✅ PASS (inject + cleanup both verified)`);
      reporter.record({
        name: c.name,
        status: 'PASS',
        snapshotInject: snapAfterInject,
        snapshotCleanup: snapAfterCleanup,
        durationMs: Date.now() - startedAt,
      });

    } catch (err) {
      console.log(`  💥 ERROR: `);
      reporter.record({
        name: c.name,
        status: 'ERROR',
        error: (err as Error).message,
        stack: (err as Error).stack,
        durationMs: Date.now() - startedAt,
      });
      // 异常时也尝试清理
      try { if (c.cleanup.trim() !== 'true') await cdp.evaluate(c.cleanup); } catch {}
    }
  }

  const summary = reporter.summary();
  await cdp.dispose();

  process.exit(summary.failed === 0 ? 0 : 1);
}

function sleep(ms: number): Promise<void> {
  return new Promise(r => setTimeout(r, ms));
}

main().catch(err => {

  process.exit(2);
});

/** 持续采集快照，直到满足 predicate 或超时 */
async function waitForSignal(
  cdp: CDPClient,
  predicate: (snap: any) => boolean,
  timeoutMs: number,
  pollIntervalMs = 1000,
): Promise<{ matched: boolean; snap: any; waitedMs: number }> {
  const startedAt = Date.now();
  while (Date.now() - startedAt < timeoutMs) {
    const snap = await captureSnapshot(cdp);
    if (predicate(snap)) {
      return { matched: true, snap, waitedMs: Date.now() - startedAt };
    }
    await sleep(pollIntervalMs);
  }
  return { matched: false, snap: null, waitedMs: timeoutMs };
}
