// mvp-runner/src/heartbeat/recovery-executor.ts

// ─── 核心修复：task-interrupted 的三路重试查找 ───────────────────────────────

private async clickRetryButton(): Promise<{ clicked: boolean; method: string }> {
  return await this.trae.evaluate(`(() => {
    // 路径 1: aria-label 精确查找（最可靠，优先）
    let btn = document.querySelector('button[aria-label="重试"]');
    if (btn) {
      btn.click();
      return { clicked: true, method: 'aria-label' };
    }

    // 路径 2: 文本内容全量扫描（兜底）
    const allBtns = Array.from(document.querySelectorAll('button'));
    const textBtn = allBtns.find(b => b.textContent?.trim() === '重试');
    if (textBtn) {
      textBtn.click();
      return { clicked: true, method: 'text-content' };
    }

    // 路径 3: 在"手动终止输出"消息旁查找（DEVCLI 手动中断特有场景）
    const turns = document.querySelectorAll('.chat-turn');
    for (let i = turns.length - 1; i >= 0; i--) {
      if (turns[i].textContent?.includes('手动终止输出')) {
        const nearBtn = turns[i].querySelector('button[aria-label="重试"]');
        if (nearBtn) {
          nearBtn.click();
          return { clicked: true, method: 'previous-message' };
        }
        break; // 只找最近一条，找不到就不继续
      }
    }

    return { clicked: false, method: 'not-found' };
  })()`);
}

// ─── task-interrupted 完整处理分支 ──────────────────────────────────────────

private retryCount = 0;
private readonly MAX_RETRIES = 3;

async handleTaskInterrupted(): Promise<RecoveryResult> {
  // 超过最大重试次数，进入终态
  if (this.retryCount >= this.MAX_RETRIES) {
    this.logger.error(`[RecoveryExecutor] task-interrupted: 已达最大重试次数(${this.MAX_RETRIES})，进入终态`);
    await this.sendAlert('critical', 'DEVCLI 任务中断，自动重试耗尽，需人工介入');
    return { success: false, action: 'give_up', reason: 'max_retries_exceeded' };
  }

  this.retryCount++;
  this.logger.info(`[RecoveryExecutor] task-interrupted: 第 ${this.retryCount} 次尝试 clickRetryButton`);

  // 执行三路查找
  const result = await this.clickRetryButton();

  if (!result.clicked) {
    // 三路都找不到按钮，尝试兜底：直接向输入框发送消息
    this.logger.warn(`[RecoveryExecutor] 三路查找均未找到重试按钮，尝试发送消息兜底`);
    await this.sendContinueMessage();
    return { success: true, action: 'sent_message', method: 'fallback-input' };
  }

  this.logger.info(`[RecoveryExecutor] 重试按钮点击成功，方式: ${result.method}，等待 3s 验证状态`);
  await this.sleep(3000);

  // 验证恢复结果
  const check = await this.detector.detect();
  if (check.task.status === 'in_progress') {
    this.retryCount = 0; // 成功，重置计数器
    this.logger.info(`[RecoveryExecutor] ✅ task-interrupted 恢复成功，状态已回到 in_progress`);
    return { success: true, action: 'recovered', method: result.method };
  }

  // 本次未成功，等下一个心跳周期再试
  this.logger.warn(`[RecoveryExecutor] 重试后状态仍为 ${check.task.status}，等待下次心跳重试`);
  return { success: false, action: 'pending', retryCount: this.retryCount };
}

// ─── 兜底：向输入框发送"继续"消息 ─────────────────────────────────────────

private async sendContinueMessage(): Promise<void> {
  await this.trae.evaluate(`(() => {
    const input = document.querySelector('.chat-input-v2 textarea');
    if (!input) return;
    const nativeInputValueSetter = Object.getOwnPropertyDescriptor(
      window.HTMLTextAreaElement.prototype, 'value'
    ).set;
    nativeInputValueSetter.call(input, '请继续上一个任务');
    input.dispatchEvent(new Event('input', { bubbles: true }));
    const sendBtn = document.querySelector('.chat-input-v2-send-button');
    sendBtn?.click();
  })()`);
}
