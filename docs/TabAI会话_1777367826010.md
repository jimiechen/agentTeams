## **Step 2：飞书 CLI 调度脚本（替换外部 LLM 调用）**

原来三个 Skill 里的 `callLLM()` 全部替换为**通过 CDP 向 WikiBot 发飞书消息**，复用现有的 `runner-multi.ts` 机制。

```typescript
// scripts/wiki-scheduler.ts（完整重写版）

import { readFileSync, writeFileSync, readdirSync, existsSync, mkdirSync } from 'node:fs';
import path from 'node:path';
import { CDPClient } from '../src/cdp/client.js';
import { loadConfig } from '../src/config.js';
import { LarkClient } from '../src/lark/client.js';

// ──────────────────────────────────────────────
// 配置
// ──────────────────────────────────────────────

const WORKSPACES = ['PMCLI', 'DEVCLI'] as const;
type WorkspaceName = typeof WORKSPACES[number];

const BASE_PATH = path.resolve('./workspaces');

// ──────────────────────────────────────────────
// 工具函数
// ──────────────────────────────────────────────

function getToday(): string {
  return new Date().toISOString().split('T')[0];
}

function getRecentDates(days: number): string[] {
  return Array.from({ length: days }, (_, i) => {
    const d = new Date();
    d.setDate(d.getDate() - i);
    return d.toISOString().split('T')[0];
  });
}

function readRunsForDate(workspace: WorkspaceName, date: string): string {
  const runsDir = path.join(BASE_PATH, workspace, 'runs');
  if (!existsSync(runsDir)) return '';

  const files = readdirSync(runsDir)
    .filter(f => f.startsWith(date) && f.endsWith('.md'))
    .sort();

  if (files.length === 0) return '';

  return files.map(f => {
    const content = readFileSync(path.join(runsDir, f), 'utf-8');
    return `### 记录: \n`;
  }).join('\n---\n');
}

function readLayer1ForDays(workspace: WorkspaceName, days: number): string {
  const dailyDir = path.join(BASE_PATH, workspace, 'wiki', 'daily');
  if (!existsSync(dailyDir)) return '';

  return getRecentDates(days)
    .map(date => {
      const filePath = path.join(dailyDir, `.md`);
      if (!existsSync(filePath)) return null;
      return `### \n`;
    })
    .filter(Boolean)
    .join('\n---\n');
}

function readLayer2(workspace: WorkspaceName): string {
  const corePath = path.join(BASE_PATH, workspace, 'wiki', 'core', 'knowledge.md');
  return existsSync(corePath) ? readFileSync(corePath, 'utf-8') : '';
}

// ──────────────────────────────────────────────
// Prompt 构建（WikiBot 专用）
// ──────────────────────────────────────────────

// ──────────────────────────────────────────────
// WikiBot 任务发送（通过飞书 CLI）
// ──────────────────────────────────────────────

async function sendWikiBotTask(
  lark: LarkClient,
  prompt: string,
  taskType: 'distill' | 'merge',
  workspace: WorkspaceName
): Promise<{ success: boolean; error?: string }> {
  const taskId = `wiki-`;
  console.log(`[wiki-scheduler] 发送任务到 WikiBot: `);

  try {
    // 通过飞书发送任务消息给 WikiBot
    // WikiBot 监听飞书消息，收到后在 Trae IDE 中执行
    await lark.sendMessage({
      receiveId: process.env.WIKIBOT_LARK_USER_ID ?? '',
      content: prompt,
      msgType: 'text',
    });

    // 等待 WikiBot 完成（轮询输出文件是否生成）
    const outputPath = taskType === 'distill'
      ? path.join(BASE_PATH, workspace, 'wiki', 'daily', `.md`)
      : path.join(BASE_PATH, workspace, 'wiki', 'core', 'knowledge.md');

    const maxWaitMs = 3 * 60 * 1000; // 最多等 3 分钟
    const pollIntervalMs = 5000;      // 每 5 秒检查一次
    const startTime = Date.now();
    const expectedMtime = Date.now(); // 期望文件在此时间之后被更新

    while (Date.now() - startTime < maxWaitMs) {
      await new Promise(r => setTimeout(r, pollIntervalMs));

      if (existsSync(outputPath)) {
        const { mtimeMs } = require('node:fs').statSync(outputPath);
        if (mtimeMs >= expectedMtime) {
          console.log(`[wiki-scheduler] ✅ WikiBot 完成任务: `);
          return { success: true };
        }
      }
    }

    return { success: false, error: 'timeout: WikiBot 3分钟内未完成任务' };

  } catch (err) {
    return { success: false, error: (err as Error).message };
  }
}

// ──────────────────────────────────────────────
// 主流程
// ──────────────────────────────────────────────

async function runDailyDistill(lark: LarkClient): Promise<void> {
  const today = getToday();
  console.log(`\n=== 每日蒸馏 [${today}] ===`);

  for (const workspace of WORKSPACES) {
    // 检查今天是否已蒸馏
    const outputPath = path.join(BASE_PATH, workspace, 'wiki', 'daily', `.md`);
    if (existsSync(outputPath)) {
      console.log(`[wiki-scheduler]  今日已蒸馏，跳过`);
      continue;
    }

    // 读取今天的原始记录
    const rawContent = readRunsForDate(workspace, today);
    if (!rawContent) {
      console.log(`[wiki-scheduler]  今日无执行记录，跳过`);
      continue;
    }

    // 确保输出目录存在
    mkdirSync(path.join(BASE_PATH, workspace, 'wiki', 'daily'), { recursive: true });
    mkdirSync(path.join(BASE_PATH, workspace, 'wiki', 'core', 'backups'), { recursive: true });

    // 构建 prompt 并发送给 WikiBot
    const prompt = buildDistillPrompt(workspace, today, rawContent);
    const result = await sendWikiBotTask(lark, prompt, 'distill', workspace);

    if (!result.success) {
      console.error(`[wiki-scheduler] ❌  蒸馏失败: `);
      // 写入错误日志
      const errorLog = path.join(BASE_PATH, workspace, 'wiki', 'distill-errors.log');
      writeFileSync(errorLog, `[] distill failed: \n`, { flag: 'a' });
    }
  }
}

async function runWeeklyMerge(lark: LarkClient): Promise<void> {
  console.log(`\n=== 每周合并 [] ===`);

  for (const workspace of WORKSPACES) {
    const weekContent = readLayer1ForDays(workspace, 7);
    if (!weekContent) {
      console.log(`[wiki-scheduler]  本周无 Layer 1 记忆，跳过`);
      continue;
    }

    const existingCore = readLayer2(workspace);
    const prompt = buildMergePrompt(workspace, weekContent, existingCore);
    const result = await sendWikiBotTask(lark, prompt, 'merge', workspace);

    if (!result.success) {
      console.error(`[wiki-scheduler] ❌  合并失败: `);
      const errorLog = path.join(BASE_PATH, workspace, 'wiki', 'distill-errors.log');
      writeFileSync(errorLog, `[] merge failed: \n`, { flag: 'a' });
    }
  }
}

async function main(): Promise<void> {
  const mode = process.argv[2] ?? 'distill';
  const cfg  = loadConfig();

  const lark = new LarkClient(cfg.lark);
  await lark.connect();

  try {
    if (mode === 'distill' || mode === 'all') {
      await runDailyDistill(lark);
    }

    const isSunday = new Date().getDay() === 0;
    if (mode === 'merge' || (mode === 'all' && isSunday)) {
      await runWeeklyMerge(lark);
    }
  } finally {
    await lark.disconnect();
  }

  console.log('\n=== wiki-scheduler 完成 ===');
}

main().catch(err => {
  console.error('[wiki-scheduler] 致命错误:', err);
  process.exit(1);
});
```

## **Step 3：WikiBot 在 Trae IDE 中的工作区配置**

WikiBot 需要在 Trae IDE 里作为一个独立的任务存在，监听飞书消息并执行蒸馏任务。在现有的 `config/` 下新增一个 WikiBot 配置文件：

```yaml
# config/wikibot.yaml

bot:
  name: WikiBot
  workspace: WikiBot
  lark:
    userId: "${WIKIBOT_LARK_USER_ID}"   # WikiBot 专属飞书账号
    botToken: "${LARK_BOT_TOKEN}"

  # WikiBot 只响应来自调度脚本的系统消息，不响应普通用户
  messageFilter:
    allowedSenders:
      - "${SCHEDULER_LARK_USER_ID}"     # 调度脚本的飞书 ID
    requiredPrefix: "@WikiBot"           # 消息必须以 @WikiBot 开头

  # 执行超时配置
  timeouts:
    distillTask: 180000   # 蒸馏任务 3 分钟
    mergeTask: 120000     # 合并任务 2 分钟

  # 执行完成后回复飞书确认消息
  replyOnComplete: true
  replyTemplate: "✅ WikiBot 完成 {taskType} 任务\n工作区: {workspace}\n输出: {outputPath}\n字数: {chars}"
```

## **与原方案的差异对比**

| 维度 | 原方案（外部 LLM） | 新方案（WikiBot） |
|------|-----------------|----------------|
| **LLM 来源** | OpenAI API / 外部服务 | Trae IDE 内置模型 |
| **调用方式** | HTTPS 直连 | 飞书消息 → WikiBot → Trae |
| **鉴权** | API Key | 飞书 userId 白名单 |
| **成本** | 按 token 计费 | 复用已有 Trae 授权 |
| **离线风险** | 外部 API 不可用 | Trae IDE 需保持运行 |
| **调试方式** | 看 API 响应 | 看飞书回复消息 |
| **Skill 代码** | 需要 `callLLM()` | 只需读写文件 + 发飞书消息 |

唯一需要注意的是：**Trae IDE 必须在凌晨 4 点时保持运行**，WikiBot 才能收到飞书消息并执行蒸馏。如果 IDE 关闭，调度脚本发出的消息会在飞书里排队，下次 IDE 启动时 WikiBot 会补处理——这个行为和现有 PMCLI/DEVCLI 的离线消息处理机制完全一致，不需要额外处理。
