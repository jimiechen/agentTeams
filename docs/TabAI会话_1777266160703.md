### **`src/cdp/client.ts`**

```typescript
import CDP from 'chrome-remote-interface';
import debug from 'debug';

const log = debug('mvp:cdp');

export class CDPClient {
  private client: CDP.Client | null = null;
  private heartbeatTimer: NodeJS.Timeout | null = null;
  private reconnectAttempts = 0;
  private readonly maxReconnect = 5;
  private disposed = false;

  constructor(private host: string, private port: number) {}

  async connect(): Promise<void> {
    const targets = await CDP.List({ host: this.host, port: this.port });
    log('targets fetched: %d', targets.length);
    const main =
      targets.find(t => t.type === 'page' && /trae/i.test(t.title ?? '')) ??
      targets.find(t => t.type === 'page');
    if (!main) throw new Error('No suitable Trae target found');
    log('main target: id=%s title=%s', main.id, main.title);

    this.client = await CDP({ target: main.webSocketDebuggerUrl });
    await Promise.all([
      this.client.Runtime.enable(),
      this.client.DOM.enable(),
      this.client.Page.enable(),
    ]);

    this.client.on('disconnect', () => {
      log('cdp disconnected');
      this.stopHeartbeat();
      if (!this.disposed) this.scheduleReconnect();
    });

    this.startHeartbeat();
    this.reconnectAttempts = 0;
    log('connected & domains enabled');
  }

  get raw(): CDP.Client {
    if (!this.client) throw new Error('CDP not connected');
    return this.client;
  }

  /** 在主 Target 中执行任意 JS，返回值（returnByValue）。 */
  async evaluate<T = unknown>(expression: string): Promise<T> {
    const r = await this.raw.Runtime.evaluate({
      expression,
      returnByValue: true,
      awaitPromise: true,
    });
    if (r.exceptionDetails) {
      throw new Error(`evaluate exception: ${r.exceptionDetails.text}`);
    }
    return r.result.value as T;
  }

  /** 发送原生鼠标点击事件（mousePressed + mouseReleased）到指定坐标。 */
  async click(x: number, y: number): Promise<void> {
    const Input = this.raw.Input;
    await Input.dispatchMouseEvent({ type: 'mousePressed', x, y, button: 'left', clickCount: 1 });
    await sleep(60);
    await Input.dispatchMouseEvent({ type: 'mouseReleased', x, y, button: 'left', clickCount: 1 });
  }

  private startHeartbeat() {
    this.heartbeatTimer = setInterval(async () => {
      try {
        await this.raw.Runtime.evaluate({ expression: '1' });
      } catch (e) {
        log('heartbeat failed: %s', (e as Error).message);
      }
    }, 30_000);
  }

  private stopHeartbeat() {
    if (this.heartbeatTimer) clearInterval(this.heartbeatTimer);
    this.heartbeatTimer = null;
  }

  private async scheduleReconnect() {
    if (this.reconnectAttempts >= this.maxReconnect) {
      log('reconnect exhausted');
      return;
    }
    const delay = Math.pow(2, this.reconnectAttempts) * 1000;
    this.reconnectAttempts += 1;
    log('reconnect attempt %d in %dms', this.reconnectAttempts, delay);
    await sleep(delay);
    try {
      await this.connect();
    } catch (e) {
      log('reconnect failed: %s', (e as Error).message);
      this.scheduleReconnect();
    }
  }

  async dispose(): Promise<void> {
    this.disposed = true;
    this.stopHeartbeat();
    if (this.client) await this.client.close();
  }
}

export const sleep = (ms: number) => new Promise<void>(r => setTimeout(r, ms));
```

### **`src/selectors/resolver.ts`**

```typescript
import { readFileSync, readdirSync } from 'node:fs';
import path from 'node:path';
import debug from 'debug';
import type { CDPClient } from '../cdp/client.js';

const log = debug('mvp:selectors');

interface SelectorEntry { primary: string; fallback: string[]; }
interface SelectorFile { trae_version: string; [k: string]: unknown; }

export class SelectorResolver {
  private data: SelectorFile;
  private cache = new Map<string, string>();

  constructor(private cdp: CDPClient, configDir = './config') {
    const files = readdirSync(configDir).filter(f => /^selectors\.v.*\.json$/.test(f)).sort();
    if (files.length === 0) throw new Error('no selectors.v*.json found');
    const latest = files[files.length - 1];
    this.data = JSON.parse(readFileSync(path.join(configDir, latest), 'utf-8'));
    log('loaded %s (trae_version=%s)', latest, this.data.trae_version);
  }

  /** path 形如 'chat.input' / 'task_list.item'。返回首个能命中节点的 selector。 */
  async resolve(p: string): Promise<string> {
    const hit = this.cache.get(p);
    if (hit) return hit;

    const entry = this.lookup(p);
    const candidates = [entry.primary, ...entry.fallback];
    for (let i = 0; i < candidates.length; i += 1) {
      const sel = candidates[i];
      const found = await this.cdp.evaluate<boolean>(
        `!!document.querySelector(${JSON.stringify(sel)})`
      );
      if (found) {
        if (i > 0) console.warn(`[selectors] primary failed for ${p}, using fallback[${i - 1}]: ${sel}`);
        this.cache.set(p, sel);
        return sel;
      }
    }
    throw new SelectorResolutionError(p, candidates);
  }

  private lookup(p: string): SelectorEntry {
    const parts = p.split('.');
    let node: any = this.data;
    for (const k of parts) node = node?.[k];
    if (!node?.primary) throw new Error(`selector path not found: ${p}`);
    return node as SelectorEntry;
  }
}

export class SelectorResolutionError extends Error {
  constructor(p: string, tried: string[]) {
    super(`Selector failed at ${p}, tried: ${tried.join(' | ')}`);
    this.name = 'SelectorResolutionError';
  }
}
```

### **`src/actions/switch-task.ts`**

```typescript
import debug from 'debug';
import { CDPClient, sleep } from '../cdp/client.js';
import type { SelectorResolver } from '../selectors/resolver.js';

const log = debug('mvp:action:switch');

export async function switchTask(
  cdp: CDPClient,
  resolver: SelectorResolver,
  slotIndex: number,
): Promise<void> {
  const itemSel = await resolver.resolve('task_list.item');

  const beforeHash = await cdp.evaluate<string>(
    `String(document.body.innerText.length)`
  );

  const coord = await cdp.evaluate<{ x: number; y: number } | null>(`
    (() => {
      const items = document.querySelectorAll(${JSON.stringify(itemSel)});
      const item = items[${slotIndex}];
      if (!item) return null;
      const r = item.getBoundingClientRect();
      return { x: r.left + r.width / 2, y: r.top + r.height / 2 };
    })()
  `);
  if (!coord) throw new Error(`task slot ${slotIndex} not found (only ${itemSel} matched)`);

  log('clicking slot %d at (%d, %d)', slotIndex, coord.x, coord.y);
  await cdp.click(coord.x, coord.y);
  await sleep(1500);

  const afterHash = await cdp.evaluate<string>(
    `String(document.body.innerText.length)`
  );
  if (beforeHash === afterHash) log('warn: page text length unchanged after switch');
}
```

```typescript
import debug from 'debug';
import { CDPClient, sleep } from '../cdp/client.js';
import type { SelectorResolver } from '../selectors/resolver.js';

const log = debug('mvp:action:fill');

export async function fillPrompt(
  cdp: CDPClient,
  resolver: SelectorResolver,
  prompt: string,
): Promise<void> {
  const inputSel = await resolver.resolve('chat.input');

  // 聚焦
  const coord = await cdp.evaluate<{ x: number; y: number }>(`
    (() => {
      const el = document.querySelector(${JSON.stringify(inputSel)});
      el.focus();
      const r = el.getBoundingClientRect();
      return { x: r.left + 20, y: r.top + 20 };
    })()
  `);
  await cdp.click(coord.x, coord.y);
  await sleep(300);

  // 注入文本（Day 0 验证组合）
  const escaped = JSON.stringify(prompt);
  await cdp.evaluate(`
    (() => {
      const el = document.querySelector(${JSON.stringify(inputSel)});
      el.innerText = ${escaped};
      el.dispatchEvent(new InputEvent('input', { bubbles: true, data: ${escaped} }));
    })()
  `);
  await sleep(500);

  // 回读校验
  const actual = await cdp.evaluate<string>(`
    document.querySelector(${JSON.stringify(inputSel)}).innerText
  `);
  if (actual.trim() !== prompt.trim()) {
    log('warn: actual="%s" expected="%s"', actual, prompt);
    throw new Error('fillPrompt verification failed');
  }
}
```

### **`src/actions/wait-response.ts`**

```typescript
import debug from 'debug';
import { CDPClient, sleep } from '../cdp/client.js';
import type { SelectorResolver } from '../selectors/resolver.js';

const log = debug('mvp:action:wait');

interface Opts {
  timeoutMs?: number;
  stableMs?: number;
  pollMs?: number;
}

export async function waitResponse(
  cdp: CDPClient,
  resolver: SelectorResolver,
  opts: Opts = {},
): Promise<string> {
  const timeoutMs = opts.timeoutMs ?? 90_000;
  const stableMs = opts.stableMs ?? 1500;
  const pollMs = opts.pollMs ?? 500;
  const turnSel = await resolver.resolve('chat.chat_turn');

  const startCount = await cdp.evaluate<number>(
    `document.querySelectorAll(${JSON.stringify(turnSel)}).length`
  );
  log('start chat_turn count=%d', startCount);

  const t0 = Date.now();
  // 阶段一：等新消息出现
  while (Date.now() - t0 < timeoutMs) {
    const c = await cdp.evaluate<number>(
      `document.querySelectorAll(${JSON.stringify(turnSel)}).length`
    );
    if (c > startCount) break;
    await sleep(pollMs);
  }

  // 阶段二：等文本稳定
  let lastLen = -1;
  let stableSince = Date.now();
  while (Date.now() - t0 < timeoutMs) {
    const len = await cdp.evaluate<number>(`
      (() => {
        const all = document.querySelectorAll(${JSON.stringify(turnSel)});
        const last = all[all.length - 1];
        return last ? last.textContent.length : 0;
      })()
    `);
    if (len !== lastLen) {
      lastLen = len;
      stableSince = Date.now();
    } else if (Date.now() - stableSince >= stableMs) {
      break;
    }
    await sleep(pollMs);
  }

  if (Date.now() - t0 >= timeoutMs) {
    throw new Error(`waitResponse timeout after ${timeoutMs}ms`);
  }

  const text = await cdp.evaluate<string>(`
    (() => {
      const all = document.querySelectorAll(${JSON.stringify(turnSel)});
      const last = all[all.length - 1];
      return last ? last.textContent : '';
    })()
  `);
  log('response captured, length=%d, took=%dms', text.length, Date.now() - t0);
  return text;
}
```

### **`src/lark/parser.ts`**

```typescript
export interface ParsedCommand {
  slot: number;          // 任务槽位序号，默认从配置取
  prompt: string;        // 真实送给 AI 的内容
  raw: string;           // 原始文本（去掉 @PMCLI 后）
}

/**
 * 解析格式：
 *   @PMCLI #2 修复登录接口的 bug    → { slot: 2, prompt: '修复登录接口的 bug' }
 *   @PMCLI 写一个快排                → { slot: defaultSlot, prompt: '写一个快排' }
 */
export function parseCommand(text: string, defaultSlot: number): ParsedCommand | null {
  // 去除 @PMCLI 与多余空白；飞书的 mention 在 text 中通常是 "@_user_1"，但 message 文本里也会留 keyword
  const cleaned = text
    .replace(/@_user_\d+/g, '')
    .replace(/@PMCLI/gi, '')
    .replace(/\s+/g, ' ')
    .trim();
  if (!cleaned) return null;

  const m = /^#(\d+)\s+(.+)$/s.exec(cleaned);
  if (m) return { slot: Number(m[1]), prompt: m[2].trim(), raw: cleaned };
  return { slot: defaultSlot, prompt: cleaned, raw: cleaned };
}
```
