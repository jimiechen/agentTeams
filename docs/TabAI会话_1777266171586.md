## 一、`src/lark/client.ts`（完整版）

```typescript
import lark from '@larksuiteoapi/node-sdk';
import debug from 'debug';

const log = debug('mvp:lark');

export interface LarkInbound {
  chatId: string;
  messageId: string;
  senderId: string;
  text: string;               // 去掉 at 标签后的纯文本
  rawContent: string;         // 原始 content JSON 字符串
  mentionedKeywords: string[];// 命中的关键字列表（通常只有 PMCLI 一个）
}

export type LarkHandler = (msg: LarkInbound) => Promise<void>;

export class LarkBot {
  private client: lark.Client;
  private wsClient: lark.WSClient;
  private chatId: string;
  private mentionKeyword: string;

  constructor(opts: {
    appId: string;
    appSecret: string;
    chatId: string;
    mentionKeyword: string;
  }) {
    this.chatId = opts.chatId;
    this.mentionKeyword = opts.mentionKeyword;
    this.client = new lark.Client({
      appId: opts.appId,
      appSecret: opts.appSecret,
      disableTokenCache: false,
    });
    this.wsClient = new lark.WSClient({
      appId: opts.appId,
      appSecret: opts.appSecret,
    });
  }

  /** 启动长连接，收到 @PMCLI 的群消息时回调 handler。 */
  async start(handler: LarkHandler): Promise<void> {
    const eventDispatcher = new lark.EventDispatcher({}).register({
      'im.message.receive_v1': async (data: any) => {
        try {
          const msg = data.message;
          // 只处理目标群
          if (msg.chat_id !== this.chatId) {
            log('skip: chat_id mismatch (%s)', msg.chat_id);
            return;
          }
          // 只处理群聊 text 消息
          if (msg.chat_type !== 'group' || msg.message_type !== 'text') {
            log('skip: not a group text message');
            return;
          }

          // 解析 content
          const content = safeJsonParse(msg.content, {}) as { text?: string };
          const rawText = content.text ?? '';

          // 检查是否命中 @PMCLI：mentions 数组或文本里明文包含关键词
          const mentions: Array<{ key: string; name: string; id: any }> = msg.mentions ?? [];
          const hitByMention = mentions.some(
            m => (m.name ?? '').toUpperCase() === this.mentionKeyword.toUpperCase(),
          );
          const hitByText = new RegExp(`@?\\b${escapeRegExp(this.mentionKeyword)}\\b`, 'i').test(rawText);
          if (!hitByMention && !hitByText) {
            log('skip: no %s mention', this.mentionKeyword);
            return;
          }

          // 把 @_user_xxx 占位符替换为关键词，便于 parser 统一处理
          const cleaned = rawText.replace(/@_user_\d+/g, `@${this.mentionKeyword}`);

          const inbound: LarkInbound = {
            chatId: msg.chat_id,
            messageId: msg.message_id,
            senderId: data.sender?.sender_id?.open_id ?? 'unknown',
            text: cleaned,
            rawContent: msg.content,
            mentionedKeywords: [this.mentionKeyword],
          };
          log('inbound message_id=%s text="%s"', inbound.messageId, inbound.text);

          // 异步处理，防止长任务阻塞 WS 回调
          handler(inbound).catch(err => {
            log('handler error: %s', (err as Error).message);
          });

          return { code: 0 };
        } catch (err) {
          log('dispatch error: %s', (err as Error).message);
          return { code: 0 };
        }
      },
    });

    this.wsClient.start({ eventDispatcher });
    log('lark ws client started, listening chat_id=%s keyword=@%s', this.chatId, this.mentionKeyword);
  }

  /** 发送纯文本消息到目标群。 */
  async sendText(text: string): Promise<void> {
    await this.client.im.message.create({
      params: { receive_id_type: 'chat_id' },
      data: {
        receive_id: this.chatId,
        msg_type: 'text',
        content: JSON.stringify({ text }),
      },
    });
  }

  /** 回复指定消息（引用回复）。 */
  async reply(messageId: string, text: string): Promise<void> {
    try {
      await this.client.im.message.reply({
        path: { message_id: messageId },
        data: {
          msg_type: 'text',
          content: JSON.stringify({ text }),
        },
      });
    } catch (err) {
      log('reply failed, fallback to sendText: %s', (err as Error).message);
      await this.sendText(text);
    }
  }

  /** 回复富文本卡片（使用 post 消息类型）。 */
  async replyPost(messageId: string, title: string, body: string): Promise<void> {
    const content = {
      zh_cn: {
        title,
        content: [[{ tag: 'text', text: body }]],
      },
    };
    try {
      await this.client.im.message.reply({
        path: { message_id: messageId },
        data: {
          msg_type: 'post',
          content: JSON.stringify(content),
        },
      });
    } catch (err) {
      log('replyPost failed: %s', (err as Error).message);
    }
  }
}

function safeJsonParse<T>(s: string, fallback: T): T {
  try { return JSON.parse(s) as T; } catch { return fallback; }
}

function escapeRegExp(s: string): string {
  return s.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}
```

## 二、`src/runner.ts`

这是整条链路的**业务编排**：收到飞书消息 → 解析 → 加 ChatMutex 锁 → 四步 CDP → 回贴。

```typescript
import debug from 'debug';
import { mkdirSync, writeFileSync } from 'node:fs';
import path from 'node:path';
import type { CDPClient } from './cdp/client.js';
import type { SelectorResolver } from './selectors/resolver.js';
import { switchTask } from './actions/switch-task.js';
import { fillPrompt } from './actions/fill-prompt.js';
import { submit } from './actions/submit.js';
import { waitResponse } from './actions/wait-response.js';
import { chatMutex } from './mutex.js';
import { parseCommand } from './lark/parser.js';
import type { LarkBot, LarkInbound } from './lark/client.js';
import type { AppConfig } from './config.js';

const log = debug('mvp:runner');

export class Runner {
  private runsDir: string;

  constructor(
    private cfg: AppConfig,
    private cdp: CDPClient,
    private resolver: SelectorResolver,
    private lark: LarkBot,
  ) {
    this.runsDir = path.resolve('./runs');
    mkdirSync(this.runsDir, { recursive: true });
  }

  /** 飞书消息入口：注册为 LarkBot 的 handler。 */
  handle = async (msg: LarkInbound): Promise<void> => {
    // 白名单过滤
    const allowed = this.cfg.pmbot.allowed_users;
    if (allowed.length > 0 && !allowed.includes(msg.senderId)) {
      log('reject sender=%s (not in allowlist)', msg.senderId);
      await this.lark.reply(msg.messageId, `⛔ 权限不足，未加入白名单`);
      return;
    }

    // 解析指令
    const parsed = parseCommand(msg.text, this.cfg.pmbot.default_slot);
    if (!parsed || !parsed.prompt) {
      await this.lark.reply(msg.messageId,
        `⚠️ 指令为空。用法：\n` +
        `  @${this.cfg.lark.mentionKeyword} <prompt>\n` +
        `  @${this.cfg.lark.mentionKeyword} #<slot编号> <prompt>`);
      return;
    }

    // 立即 ACK
    await this.lark.reply(msg.messageId,
      `✅ 已收到指令 (slot=${parsed.slot})，排队中…`);

    // 加 ChatMutex 锁，确保同一时刻只有一条指令操作 Trae
    const release = await chatMutex.acquire();
    const startedAt = Date.now();
    const runId = new Date(startedAt).toISOString().replace(/[:.]/g, '-');
    try {
      log('processing run=%s slot=%d prompt="%s"', runId, parsed.slot, parsed.prompt.slice(0, 40));

      // Step 1: 切任务
      await this.lark.sendText(`🔀 切换到 slot #${parsed.slot} …`);
      await switchTask(this.cdp, this.resolver, parsed.slot);

      // Step 2: 填 Prompt
      await fillPrompt(this.cdp, this.resolver, parsed.prompt);

      // Step 3: 提交
      await submit(this.cdp, this.resolver);
      await this.lark.sendText(`📤 已提交，等待 AI 响应…`);

      // Step 4: 等响应
      const response = await waitResponse(this.cdp, this.resolver, {
        timeoutMs: this.cfg.pmbot.response_timeout_ms,
      });

      const duration = Date.now() - startedAt;

      // 落盘
      this.persistRun(runId, { parsed, response, duration, senderId: msg.senderId });

      // 回贴飞书（超长截断）
      const maxChars = this.cfg.pmbot.response_max_chars;
      const truncated = response.length > maxChars
        ? response.slice(0, maxChars) + `\n\n… (truncated, full ${response.length} chars saved to runs/${runId})`
        : response;

      await this.lark.replyPost(
        msg.messageId,
        `🤖 AI 响应 (slot ${parsed.slot}, ${Math.round(duration / 1000)}s)`,
        truncated,
      );
      log('run=%s completed in %dms', runId, duration);
    } catch (err) {
      const message = (err as Error).message;
      log('run=%s failed: %s', runId, message);
      await this.lark.reply(msg.messageId, `❌ 执行失败：${message}`);
      this.persistRun(runId, {
        parsed,
        response: `<ERROR: ${message}>`,
        duration: Date.now() - startedAt,
        senderId: msg.senderId,
      });
    } finally {
      release();
    }
  };

  private persistRun(runId: string, data: {
    parsed: { slot: number; prompt: string; raw: string };
    response: string;
    duration: number;
    senderId: string;
  }): void {
    const filename = path.join(this.runsDir, `${runId}.md`);
    const md =
      `# Run ${runId}\n\n` +
      `- **Slot**: ${data.parsed.slot}\n` +
      `- **Sender**: ${data.senderId}\n` +
      `- **Duration**: ${data.duration} ms\n\n` +
      `## Prompt\n\n> ${data.parsed.prompt.replace(/\n/g, '\n> ')}\n\n` +
      `## Response\n\n${data.response}\n`;
    writeFileSync(filename, md, 'utf-8');
    log('persisted → %s', filename);
  }
}
```

## 三、`src/index.ts`（启动入口）

把三个长连接（CDP + 飞书 WS）拉起来、接线、优雅退出。

```typescript
import debug from 'debug';
import { loadConfig } from './config.js';
import { CDPClient } from './cdp/client.js';
import { SelectorResolver } from './selectors/resolver.js';
import { LarkBot } from './lark/client.js';
import { Runner } from './runner.js';

const log = debug('mvp:boot');

async function main() {
  // 1. 加载配置
  const cfg = loadConfig();
  log('config loaded: chat_id=%s keyword=@%s cdp=%s:%d',
    cfg.lark.chatId, cfg.lark.mentionKeyword, cfg.cdp.host, cfg.cdp.port);

  // 2. 连接 CDP
  const cdp = new CDPClient(cfg.cdp.host, cfg.cdp.port);
  await cdp.connect();
  log('✅ CDP connected');

  // 3. 初始化选择器解析器
  const resolver = new SelectorResolver(cdp);

  // 4. 启动飞书 WS
  const bot = new LarkBot({
    appId: cfg.lark.appId,
    appSecret: cfg.lark.appSecret,
    chatId: cfg.lark.chatId,
    mentionKeyword: cfg.lark.mentionKeyword,
  });

  // 5. 组装 Runner 并挂载 handler
  const runner = new Runner(cfg, cdp, resolver, bot);
  await bot.start(runner.handle);
  log('✅ Lark WS listening');

  // 6. 启动自检消息
  try {
    await bot.sendText(
      `🟢 PMCLI MVP Runner 已上线\n` +
      `监听关键字: @${cfg.lark.mentionKeyword}\n` +
      `用法: @${cfg.lark.mentionKeyword} [#slot编号] <prompt>`,
    );
  } catch (err) {
    log('send online notice failed: %s', (err as Error).message);
  }

  // 7. 优雅退出
  const shutdown = async (signal: string) => {
    log('%s received, shutting down…', signal);
    try { await bot.sendText('🔴 PMCLI 下线'); } catch {}
    await cdp.dispose();
    process.exit(0);
  };
  process.on('SIGINT', () => shutdown('SIGINT'));
  process.on('SIGTERM', () => shutdown('SIGTERM'));
  process.on('uncaughtException', (err) => {
    log('uncaughtException: %s', err.stack ?? err.message);
  });
  process.on('unhandledRejection', (reason) => {
    log('unhandledRejection: %o', reason);
  });
}

main().catch(err => {
  console.error('[FATAL]', err);
  process.exit(1);
});
```

## 五、飞书侧前置配置（必须在飞书开发者后台先做）

代码能跑的前提是飞书 App 侧配置正确，否则 WS 连不上或收不到事件：

**应用权限**（飞书开发者后台 → 权限管理）：`im:message`、`im:message:send_as_bot`、`im:message.group_at_msg`、`im:message.group_at_msg:readonly`、`im:resource`。

**事件订阅**（飞书开发者后台 → 事件与回调 → 事件配置）：订阅 `im.message.receive_v1` 即"接收消息"。传输方式选 **WebSocket** 即可（对应 SDK 的 `WSClient`）。

**机器人加入群**：在目标群里点 "..." → "群机器人" → 添加机器人 → 选择你的应用。确保机器人已成功加入 `oc_9f741c1f2d5b1fc1e98a0b42c04283c5` 这个群。

**应用发布**：即使是企业自建应用，也需要在后台点"创建版本 → 发布"让权限和事件订阅生效，否则 WSClient 握手会通过但收不到 `im.message.receive_v1`。

## 六、已知风险与快速修复指引

**风险 1：WSClient 能连上但收不到消息**。99% 是权限或事件订阅没发布，不是代码问题。用 `DEBUG=@larksuiteoapi/*` 能看到更详细的 WS 握手日志。

**风险 2：`mentions` 数组里查不到 PMCLI**。飞书把 @ 后的 bot 名字放在 `mentions[i].name`，这里默认按 `name.toUpperCase() === 'PMCLI'` 匹配，但实际 bot 的显示名可能是"PMCLI 助手"之类。如果不命中，代码里 `hitByText` 会兜底用正则匹配文本，通常足够；如果连文本都不带 `PMCLI` 字样，需要改 `parser.ts` 里的识别逻辑。

**风险 3：`chat.input` 选择器在你的 Trae 版本命不中**。Day 0 报告只明确给出了 `send_button` 和 `task_item` 的类名，`chat.input` 我按 `[contenteditable='true'][class*='chat-input']` 这个启发式写的。如果启动后发送消息时抛 `fillPrompt verification failed`，用 Trae 开发者工具 Inspect Element 看一下真实 DOM，把正确的选择器塞到 `config/selectors.v2026-04-26.json` 的 `chat.input.primary`，不用改代码。

**风险 4：`waitResponse` 超时**。超时默认 90 秒、由 `pmbot.yaml` 的 `response_timeout_ms` 控制。如果经常超时，要么把它调大，要么缩短 Prompt，不要降 `stableMs`（降低会误判响应完成）。
