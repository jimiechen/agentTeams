## **Skill 3：`wiki-inject`（在线注入，每次 fillPrompt 前构建 Context）**

```typescript
// workspaces/PMCLI/.trae/skills/wiki-inject/skill.ts

import {
  readFileSync,
  existsSync,
  statSync,
} from 'node:fs';
import path from 'node:path';

// ──────────────────────────────────────────────
// 类型定义
// ──────────────────────────────────────────────

interface WikiConfig {
  layer2MaxChars: number;   // Layer 2 注入上限，默认 300
  layer1Days: number;       // 读取最近几天的 Layer 1，默认 3
  layer1MaxChars: number;   // Layer 1 总注入上限，默认 1500
  totalMaxChars: number;    // 总注入硬上限，默认 1800
  maxContextRatio: number;  // 占总 context 窗口的比例上限，默认 0.15
  contextWindowSize: number;// 模型 context 窗口大小（字符估算），默认 32000
}

interface SkillContext {
  params: {
    workspacePath: string;
    currentTask?: string;   // 当前任务描述，用于相关性过滤（可选）
    configOverride?: Partial<WikiConfig>;
  };
}

interface InjectResult {
  success: boolean;
  contextBlock: string;     // 最终注入的完整 context 字符串
  stats: {
    coreChars: number;
    dailyChars: number;
    totalChars: number;
    layer1DaysLoaded: number;
    cacheHit: boolean;
    truncated: boolean;
  };
  error?: string;
}

// ──────────────────────────────────────────────
// 内存缓存（进程级别，避免每次 fillPrompt 都读文件）
// ──────────────────────────────────────────────

interface CacheEntry {
  content: string;
  loadedAt: number;       // 加载时间戳（ms）
  fileMtime: number;      // 文件最后修改时间（ms），用于失效检测
}

const cache = new Map<string, CacheEntry>();

/** 读取文件并缓存，文件未变化时直接返回缓存内容 */
function readWithCache(filePath: string): string | null {
  if (!existsSync(filePath)) return null;

  const mtime = statSync(filePath).mtimeMs;
  const cached = cache.get(filePath);

  // 缓存命中：文件未修改
  if (cached && cached.fileMtime === mtime) {
    return cached.content;
  }

  // 缓存未命中或文件已更新：重新读取
  const content = readFileSync(filePath, 'utf-8');
  cache.set(filePath, {
    content,
    loadedAt: Date.now(),
    fileMtime: mtime,
  });
  return content;
}

/** 清除指定工作区的所有缓存（每日蒸馏完成后调用） */
export function invalidateCache(workspacePath: string): void {
  const prefix = path.resolve(workspacePath);
  for (const key of cache.keys()) {
    if (key.startsWith(prefix)) {
      cache.delete(key);
    }
  }
  console.log(`[wiki-inject] 已清除工作区缓存: `);
}

// ──────────────────────────────────────────────
// 工具函数
// ──────────────────────────────────────────────

/** 合并默认配置与用户覆盖配置 */
function resolveConfig(override?: Partial<WikiConfig>): WikiConfig {
  const defaults: WikiConfig = {
    layer2MaxChars: 300,
    layer1Days: 3,
    layer1MaxChars: 1500,
    totalMaxChars: 1800,
    maxContextRatio: 0.15,
    contextWindowSize: 32000,
  };
  if (!override) return defaults;
  return { ...defaults, ...override };
}

/** 根据 maxContextRatio 动态计算实际注入上限 */
function calcEffectiveLimit(config: WikiConfig): number {
  const ratioLimit = Math.floor(config.contextWindowSize * config.maxContextRatio);
  return Math.min(config.totalMaxChars, ratioLimit);
}

/** 获取最近 N 天的日期字符串列表（从今天往前推） */
function getRecentDates(days: number): string[] {
  const dates: string[] = [];
  for (let i = 0; i < days; i++) {
    const d = new Date();
    d.setDate(d.getDate() - i);
    dates.push(d.toISOString().split('T')[0]);
  }
  return dates; // [今天, 昨天, 前天, ...]
}

/** 截断字符串到指定长度，在句子边界截断（避免截断到句子中间） */
function smartTruncate(content: string, maxChars: number): { text: string; truncated: boolean } {
  if (content.length <= maxChars) return { text: content, truncated: false };

  // 在 maxChars 附近找最近的换行符
  const cutPoint = content.lastIndexOf('
', maxChars);
  const actualCut = cutPoint > maxChars * 0.7 ? cutPoint : maxChars;

  return {
    text: content.substring(0, actualCut) + '
[...已截断至注入上限]',
    truncated: true,
  };
}

/** 检测 Layer 2 文件是否过期（超过 8 天未更新，说明合并任务可能失败） */
function isCoreStale(corePath: string): boolean {
  if (!existsSync(corePath)) return false;
  const mtime = statSync(corePath).mtimeMs;
  const ageMs = Date.now() - mtime;
  return ageMs > 8 * 24 * 60 * 60 * 1000; // 8 天
}

// ──────────────────────────────────────────────
// Context Block 构建
// ──────────────────────────────────────────────

function buildContextBlock(
  coreContent: string,
  dailyEntries: Array<{ date: string; content: string }>,
  effectiveLimit: number,
  config: WikiConfig
): { block: string; stats: InjectResult['stats'] } {
  let totalChars = 0;
  let truncated = false;
  const parts: string[] = [];

  // ── Part 1: Layer 2 核心知识 ──────────────────
  let coreSection = '';
  if (coreContent) {
    const { text, truncated: coreTruncated } = smartTruncate(
      coreContent,
      config.layer2MaxChars
    );
    coreSection = text;
    truncated = truncated || coreTruncated;
    totalChars += coreSection.length;
  }

  // ── Part 2: Layer 1 最近 N 天 ─────────────────
  let dailySection = '';
  let dailyChars = 0;
  const layer1Budget = Math.min(
    config.layer1MaxChars,
    effectiveLimit - totalChars - 200 // 预留 200 字给 header 文字
  );

  for (const { date, content } of dailyEntries) {
    const remaining = layer1Budget - dailyChars;
    if (remaining <= 50) break; // 剩余空间不足时停止

    const entry = `
### 记忆 ()
`;
    const { text, truncated: entryTruncated } = smartTruncate(entry, remaining);
    dailySection += text + '
';
    dailyChars += text.length;
    truncated = truncated || entryTruncated;
  }

  totalChars += dailyChars;

  // ── 组装最终 Context Block ────────────────────
  const headerLine = `## 项目背景知识（自动注入 · ）`;

  if (coreSection) {
    parts.push(`
### 核心知识（长期有效）
`);
  }

  if (dailySection) {
    parts.push(`
### 近期动态（最近  天）
`);
  }

  if (parts.length === 0) {
    // 没有任何记忆可注入，返回空
    return {
      block: '',
      stats: {
        coreChars: 0,
        dailyChars: 0,
        totalChars: 0,
        layer1DaysLoaded: 0,
        cacheHit: false,
        truncated: false,
      },
    };
  }

  parts.push('---');

  const block = parts.join('
');

  return {
    block,
    stats: {
      coreChars: coreSection.length,
      dailyChars,
      totalChars: block.length,
      layer1DaysLoaded: dailyEntries.length,
      cacheHit: false, // 由外层更新
      truncated,
    },
  };
}

// ──────────────────────────────────────────────
// 主执行逻辑
// ──────────────────────────────────────────────

export async function execute(context: SkillContext): Promise<InjectResult> {
  const { workspacePath, currentTask, configOverride } = context.params;
  const config = resolveConfig(configOverride);
  const effectiveLimit = calcEffectiveLimit(config);

  const corePath  = path.resolve(workspacePath, 'wiki', 'core', 'knowledge.md');
  const dailyDir  = path.resolve(workspacePath, 'wiki', 'daily');
  const todoPath  = path.resolve(workspacePath, 'wiki', 'todo.md');

  // ── Step 1: 读取 Layer 2 核心知识（带缓存） ────

  let coreContent = '';
  let cacheHit = false;

  const cached = cache.get(corePath);
  const coreMtime = existsSync(corePath) ? statSync(corePath).mtimeMs : 0;

  if (cached && cached.fileMtime === coreMtime) {
    coreContent = cached.content;
    cacheHit = true;
  } else {
    const raw = readWithCache(corePath);
    coreContent = raw ?? '';
    cacheHit = false;
  }

  // 过期告警：Layer 2 超过 8 天未更新
  if (isCoreStale(corePath)) {
    console.warn(
      '[wiki-inject] ⚠️  Layer 2 超过 8 天未更新，wiki-merge 可能失败，请检查 distill-errors.log'
    );
  }

  // ── Step 2: 读取最近 N 天的 Layer 1（带缓存） ──

  const recentDates = getRecentDates(config.layer1Days);
  const dailyEntries: Array<{ date: string; content: string }> = [];

  for (const date of recentDates) {
    const filePath = path.join(dailyDir, `.md`);
    const content = readWithCache(filePath);
    if (content) {
      dailyEntries.push({ date, content });
    }
  }

  // ── Step 3: 读取 todo.md（附加到注入末尾） ─────

  let todoContent = '';
  if (existsSync(todoPath)) {
    const raw = readWithCache(todoPath);
    if (raw && raw.trim().length > 10) {
      // 只取最近 5 条待办，避免 todo 无限增长
      const todoLines = raw
        .split('
')
        .filter((l) => /^[-*]\s+.+/.test(l) || l.startsWith('##'))
        .slice(0, 20); // 最多 20 行
      todoContent = todoLines.join('
');
    }
  }

  // ── Step 4: 构建 Context Block ────────────────

  const { block, stats } = buildContextBlock(
    coreContent,
    dailyEntries,
    effectiveLimit,
    config
  );

  // 附加 todo（在 block 之后，不计入主注入上限）
  let finalBlock = block;
  if (todoContent && block) {
    const todoBlock = `
### 待跟进事项

`;
    // todo 额外允许 200 字
    const { text } = smartTruncate(todoBlock, 200);
    finalBlock += text;
  }

  // 如果什么都没有，返回空但不报错（首次运行时正常）
  if (!finalBlock) {
    console.log('[wiki-inject] 暂无记忆可注入（首次运行或记忆文件不存在）');
    return {
      success: true,
      contextBlock: '',
      stats: { ...stats, cacheHit },
    };
  }

  stats.cacheHit = cacheHit;

  console.log(
    `[wiki-inject] 注入完成 | ` +
    `核心: 字 | 近期: 字 | 总计: 字 | ` +
    `缓存: ` +
    ` | 截断: `
  );

  return {
    success: true,
    contextBlock: finalBlock,
    stats,
  };
}

// ──────────────────────────────────────────────
// 导出：供 fillPrompt 直接调用的便捷函数
// ──────────────────────────────────────────────

/**
 * 在 fillPrompt 前调用，将 wiki context 注入到 prompt 头部
 *
 * 用法：
 *   import { injectWikiContext } from './wiki-inject/skill.js';
 *   const enrichedPrompt = await injectWikiContext(workspacePath, rawPrompt);
 *   await fillPrompt(enrichedPrompt);
 */
export async function injectWikiContext(
  workspacePath: string,
  rawPrompt: string,
  configOverride?: Partial<WikiConfig>
): Promise<string> {
  const result = await execute({
    params: { workspacePath, currentTask: rawPrompt, configOverride },
  });

  if (!result.success || !result.contextBlock) {
    return rawPrompt; // 注入失败时原样返回，不阻断主流程
  }

  // context block 放在 prompt 前面，用分隔线隔开
  return `` +
    `
---
` +
    `## 当前任务
` +
    ``;
}
```

```typescript
// scripts/wiki-scheduler.ts
// 统一调度入口：每日蒸馏 + 每周合并 + 缓存失效
// 用法：npx tsx scripts/wiki-scheduler.ts [distill|merge|all]

import { execute as distill } from '../workspaces/PMCLI/.trae/skills/wiki-distill/skill.js';
import { execute as merge }   from '../workspaces/PMCLI/.trae/skills/wiki-merge/skill.js';
import { invalidateCache }    from '../workspaces/PMCLI/.trae/skills/wiki-inject/skill.js';
import path from 'node:path';

const WORKSPACES = [
  path.resolve('./workspaces/PMCLI'),
  path.resolve('./workspaces/DEVCLI'),
];

async function runDistill() {
  console.log('
=== [wiki-scheduler] 开始每日蒸馏 ===');
  for (const ws of WORKSPACES) {
    console.log(`
-- 工作区: `);
    const result = await distill({
      params: { workspacePath: ws },
      env: {},
    });
    console.log(
      result.success
        ? `✅ 蒸馏完成 | 压缩比: :1 | 质量: `
        : `❌ 蒸馏失败: `
    );
    // 蒸馏完成后清除该工作区的注入缓存
    if (result.success) invalidateCache(ws);
  }
}

async function runMerge() {
  console.log('
=== [wiki-scheduler] 开始每周合并 ===');
  for (const ws of WORKSPACES) {
    console.log(`
-- 工作区: `);
    const result = await merge({
      params: { workspacePath: ws },
      env: {},
    });
    console.log(
      result.success
        ? `✅ 合并完成 | 条目数:  | 字数: → `
        : `❌ 合并失败: `
    );
    if (result.success) invalidateCache(ws);
  }
}

async function main() {
  const mode = process.argv[2] ?? 'all';
  const today = new Date();
  const isSunday = today.getDay() === 0;

  if (mode === 'distill' || mode === 'all') {
    await runDistill();
  }

  if (mode === 'merge' || (mode === 'all' && isSunday)) {
    await runMerge();
  }

  console.log('
=== [wiki-scheduler] 完成 ===');
}

main().catch((err) => {
  console.error('[wiki-scheduler] 致命错误:', err);
  process.exit(1);
});
```

三个 Skill 的关键联动关系整理一下：

**缓存失效链路**是三者之间最重要的协作机制。`wiki-distill` 写完 Layer 1 后调用 `invalidateCache()`，`wiki-merge` 写完 Layer 2 后同样调用 `invalidateCache()`。这样 `wiki-inject` 的内存缓存在文件更新后第一次被调用时自动重新加载，不需要重启进程，也不需要任何外部信号。

**降级不阻断**是贯穿三个 Skill 的设计原则。`wiki-distill` 失败时写降级摘要；`wiki-merge` 失败时保留旧 Layer 2；`wiki-inject` 失败时原样返回 rawPrompt。任何一个 Skill 出问题，主流程（fillPrompt → Trae 执行任务）都不受影响，只是 AI 拿到的背景知识质量下降，不会造成任务中断。

**`injectWikiContext` 便捷函数**是 Skill 3 对外暴露的主要接口，在现有 `fillPrompt` 里只需要加一行调用：

```typescript
// 改动前
await fillPrompt(rawPrompt);

// 改动后（唯一改动）
const enrichedPrompt = await injectWikiContext(workspacePath, rawPrompt);
await fillPrompt(enrichedPrompt);
```
