export const SELECTORS = {
  // 用属性选择器代替 CSS Module 哈希
  ACTIVE_TASK: '[class*="task-item"][class*="selected"]',
  // chat-turn 本身的类是稳定的（业务类，不是 CSS Module）
  CHAT_TURN: '.chat-turn',
  // 实际承载 chat 列表的容器，由 Step 0 探测 5 的结果填入
  CHAT_LIST_CONTAINER: '<根据 Step 0 实测填入>',
  SEND_BUTTON: '.chat-input-v2-send-button',
} as const;

export const TASK_SCOPE_SCRIPT = `
(function() {
  const SELECTORS = ${JSON.stringify(SELECTORS)};
  const activeTask = document.querySelector(SELECTORS.ACTIVE_TASK);
  if (!activeTask) return null;
  // 优先在 active task 内查 chat 列表容器
  let chatRoot = activeTask.querySelector(SELECTORS.CHAT_LIST_CONTAINER);
  // 兜底：active task 自身就是 root（如果探测 6 depth = 0~2）
  if (!chatRoot) chatRoot = activeTask;
  return chatRoot;
})()
`;

// 提供给业务侧的统一获取函数（在 cdp.evaluate 内联使用）
export function buildScopedQuery(innerSelector: string): string {
  return `
    (function() {
      ${TASK_SCOPE_SCRIPT.replace('return chatRoot;', '')}
      if (!chatRoot) return null;
      return chatRoot.querySelectorAll(${JSON.stringify(innerSelector)});
    })()
  `;
}

async function getLastAIResponse(cdp: CDPClient): Promise<string> {
  const result = await cdp.evaluate<string | null>(`
    (function() {
      const SELECTORS = ${JSON.stringify(SELECTORS)};
      const activeTask = document.querySelector(SELECTORS.ACTIVE_TASK);
      if (!activeTask) return '__NO_ACTIVE_TASK__';  // 显式错误标识

      let chatRoot = activeTask.querySelector(SELECTORS.CHAT_LIST_CONTAINER) || activeTask;
      const turns = chatRoot.querySelectorAll(SELECTORS.CHAT_TURN);
      if (turns.length === 0) return '';

      for (let i = turns.length - 1; i >= 0; i--) {
        const turn = turns[i];
        if (!turn.classList.contains('user')) {
          return (turn.innerText || '').replace(/复制图片/g, '').trim();
        }
      }
      return '';
    })()
  `);

  if (result === '__NO_ACTIVE_TASK__') {
    // ⚠️ 重要：宁可失败也不返错回
    throw new TaskScopeError('Active task not found, refuse to return any chat-turn');
  }
  return result || '';
}
