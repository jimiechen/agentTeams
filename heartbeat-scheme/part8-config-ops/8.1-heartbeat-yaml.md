# 8.1 heartbeat.yaml 配置文件

## 8.1.1 配置文件概述

heartbeat.yaml 是心跳系统的核心配置文件，采用分层设计，将配置分为 global、policies、notifications 三大层级，支持灵活的策略覆盖和环境适配。

## 8.1.2 配置分层架构

```
┌─────────────────────────────────────────────────────────────┐
│                     heartbeat.yaml                          │
├─────────────────────────────────────────────────────────────┤
│  global          全局默认配置，最低优先级                    │
│  ├── daemon      Daemon 基础配置                             │
│  ├── collectors  采集器默认配置                              │
│  └── recovery    恢复执行器默认配置                          │
├─────────────────────────────────────────────────────────────┤
│  policies        策略层配置，按工作区/场景覆盖                │
│  ├── workspaces  工作区特定配置                              │
│  ├── scenarios   场景特定配置                                │
│  └── overrides   手动覆盖配置                                │
├─────────────────────────────────────────────────────────────┤
│  notifications   通知层配置                                  │
│  ├── channels    通知渠道配置                                │
│  ├── rules       通知规则                                    │
│  └── templates   通知模板                                    │
└─────────────────────────────────────────────────────────────┘
```

## 8.1.4 配置字段详解

### 8.1.4.1 global.daemon 配置

| 字段 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| `layer1Interval` | number | 5000 | Layer 1 采集间隔（毫秒） |
| `layer2Interval` | number | 15000 | Layer 2 采集间隔（毫秒） |
| `layer3Interval` | number | 60000 | Layer 3 采集间隔（毫秒） |
| `maxRecoveryRetries` | number | 3 | 最大恢复重试次数 |
| `metricsEnabled` | boolean | true | 是否启用指标收集 |
| `persistenceEnabled` | boolean | true | 是否启用持久化 |
| `persistencePath` | string | "./heartbeat.jsonl" | 持久化文件路径 |

## 8.1.5 配置加载器实现

```typescript
import yaml from 'js-yaml';
import fs from 'fs';
import path from 'path';

interface HeartbeatConfig {
  version: string;
  schema: string;
  global: GlobalConfig;
  policies: PoliciesConfig;
  notifications: NotificationsConfig;
}

class ConfigLoader {
  private configPath: string;
  private config: HeartbeatConfig | null = null;
  private watchers: ConfigChangeHandler[] = [];
  
  constructor(configPath: string = './heartbeat.yaml') {
    this.configPath = configPath;
  }
  
  load(): HeartbeatConfig {
    const fileContent = fs.readFileSync(this.configPath, 'utf-8');
    const rawConfig = yaml.load(fileContent) as HeartbeatConfig;
    
    this.config = this.resolveEnvironmentVariables(rawConfig);
    this.validateConfig(this.config);
    
    return this.config;
  }
  
  private resolveEnvironmentVariables(config: HeartbeatConfig): HeartbeatConfig {
    const resolved = JSON.stringify(config);
    const resolvedConfig = resolved.replace(
      /\$\{([^}]+)\}/g,
      (_, envVar) => process.env[envVar] || ''
    );
    
    return JSON.parse(resolvedConfig);
  }
  
  private validateConfig(config: HeartbeatConfig): void {
    const errors: string[] = [];
    
    if (config.global.daemon.layer1Interval < 1000) {
      errors.push('layer1Interval must be >= 1000ms');
    }
    
    if (config.global.daemon.layer2Interval < config.global.daemon.layer1Interval) {
      errors.push('layer2Interval must be >= layer1Interval');
    }
    
    if (config.global.daemon.maxRecoveryRetries < 0) {
      errors.push('maxRecoveryRetries must be >= 0');
    }
    
    if (errors.length > 0) {
      throw new Error(`Config validation failed:\n${errors.join('\n')}`);
    }
  }
  
  getEffectiveConfig(workspaceId: string, context: PolicyContext): EffectiveConfig {
    if (!this.config) {
      this.load();
    }
    
    let effective = { ...this.config!.global };
    
    const workspacePolicy = this.config!.policies.workspaces[workspaceId] ||
                            this.config!.policies.workspaces.default;
    if (workspacePolicy) {
      effective = this.mergeConfig(effective, workspacePolicy.config || {});
    }
    
    for (const scenario of Object.values(this.config!.policies.scenarios)) {
      if (this.matchesScenario(scenario.match, context)) {
        effective = this.mergeConfig(effective, scenario.config);
      }
    }
    
    for (const override of this.config!.policies.overrides) {
      if (override.enabled) {
        effective = this.mergeConfig(effective, override.config);
      }
    }
    
    return effective;
  }
  
  private matchesScenario(match: ScenarioMatch, context: PolicyContext): boolean {
    if (match.taskDuration) {
      const duration = context.taskDuration || 0;
      if (match.taskDuration.startsWith('>')) {
        if (duration <= parseInt(match.taskDuration.slice(1))) return false;
      } else if (match.taskDuration.startsWith('<')) {
        if (duration >= parseInt(match.taskDuration.slice(1))) return false;
      }
    }
    
    if (match.healthMode && context.healthMode !== match.healthMode) {
      return false;
    }
    
    if (match.taskType && context.taskType !== match.taskType) {
      return false;
    }
    
    return true;
  }
  
  private mergeConfig(base: any, override: any): any {
    return {
      ...base,
      ...override,
      daemon: { ...base.daemon, ...override.daemon },
      collectors: { ...base.collectors, ...override.collectors },
      recovery: { ...base.recovery, ...override.recovery }
    };
  }
  
  watch(handler: ConfigChangeHandler): void {
    this.watchers.push(handler);
    
    fs.watchFile(this.configPath, { interval: 5000 }, () => {
      try {
        const newConfig = this.load();
        this.watchers.forEach(h => h(newConfig));
      } catch (error) {
        console.error('Failed to reload config:', error);
      }
    });
  }
}

type ConfigChangeHandler = (config: HeartbeatConfig) => void;

interface PolicyContext {
  taskDuration?: number;
  healthMode?: HealthMode;
  taskType?: string;
}

type EffectiveConfig = GlobalConfig;
```

## 8.1.6 配置热更新

```typescript
class HotConfigReloader {
  private configLoader: ConfigLoader;
  private daemon: HeartbeatDaemon;
  private lastConfig: HeartbeatConfig | null = null;
  
  constructor(configLoader: ConfigLoader, daemon: HeartbeatDaemon) {
    this.configLoader = configLoader;
    this.daemon = daemon;
    
    this.configLoader.watch(this.handleConfigChange.bind(this));
  }
  
  private handleConfigChange(newConfig: HeartbeatConfig): void {
    const changes = this.diffConfig(this.lastConfig, newConfig);
    
    if (changes.requiresRestart) {
      console.log('[Config] Changes require daemon restart');
      this.notifyRestartRequired(changes);
    } else if (changes.requiresReconfigure) {
      console.log('[Config] Applying hot reconfiguration');
      this.applyHotReconfigure(newConfig, changes);
    }
    
    this.lastConfig = newConfig;
  }
  
  private diffConfig(
    oldConfig: HeartbeatConfig | null,
    newConfig: HeartbeatConfig
  ): ConfigDiff {
    const diff: ConfigDiff = {
      requiresRestart: false,
      requiresReconfigure: false,
      changedFields: []
    };
    
    if (!oldConfig) {
      return diff;
    }
    
    const restartFields = [
      'global.daemon.persistencePath',
      'global.daemon.persistenceEnabled'
    ];
    
    const reconfigureFields = [
      'global.daemon.layer1Interval',
      'global.daemon.layer2Interval',
      'global.daemon.layer3Interval',
      'global.daemon.maxRecoveryRetries',
      'global.recovery.rateLimits'
    ];
    
    for (const field of restartFields) {
      if (this.getFieldValue(oldConfig, field) !== this.getFieldValue(newConfig, field)) {
        diff.requiresRestart = true;
        diff.changedFields.push(field);
      }
    }
    
    for (const field of reconfigureFields) {
      if (this.getFieldValue(oldConfig, field) !== this.getFieldValue(newConfig, field)) {
        diff.requiresReconfigure = true;
        diff.changedFields.push(field);
      }
    }
    
    return diff;
  }
  
  private applyHotReconfigure(config: HeartbeatConfig, diff: ConfigDiff): void {
    if (diff.changedFields.some(f => f.includes('layer1Interval'))) {
      this.daemon.updateInterval('layer1', config.global.daemon.layer1Interval);
    }
    
    if (diff.changedFields.some(f => f.includes('layer2Interval'))) {
      this.daemon.updateInterval('layer2', config.global.daemon.layer2Interval);
    }
    
    if (diff.changedFields.some(f => f.includes('layer3Interval'))) {
      this.daemon.updateInterval('layer3', config.global.daemon.layer3Interval);
    }
    
    if (diff.changedFields.some(f => f.includes('maxRecoveryRetries'))) {
      this.daemon.updateMaxRetries(config.global.daemon.maxRecoveryRetries);
    }
  }
  
  private notifyRestartRequired(changes: ConfigDiff): void {
    console.warn(`[Config] Restart required for fields: ${changes.changedFields.join(', ')}`);
  }
  
  private getFieldValue(obj: any, path: string): any {
    return path.split('.').reduce((o, k) => o?.[k], obj);
  }
}

interface ConfigDiff {
  requiresRestart: boolean;
  requiresReconfigure: boolean;
  changedFields: string[];
}
```

## 8.1.8 配置优先级规则

```
优先级从高到低：

1. 环境变量覆盖 (HEARTBEAT_*)
2. policies.overrides (enabled: true)
3. policies.scenarios (匹配当前场景)
4. policies.workspaces (匹配当前工作区)
5. global 默认配置
```

### 优先级计算示例

```typescript
class ConfigPriorityResolver {
  resolve(
    global: GlobalConfig,
    workspacePolicy: WorkspacePolicy | undefined,
    scenarioPolicies: ScenarioPolicy[],
    overrides: OverrideConfig[],
    envOverrides: Record<string, any>
  ): EffectiveConfig {
    let config = { ...global };
    
    if (workspacePolicy?.config) {
      config = this.deepMerge(config, workspacePolicy.config);
    }
    
    for (const scenario of scenarioPolicies) {
      config = this.deepMerge(config, scenario.config);
    }
    
    for (const override of overrides.filter(o => o.enabled)) {
      config = this.deepMerge(config, override.config);
    }
    
    config = this.deepMerge(config, envOverrides);
    
    return config;
  }
  
  private deepMerge(target: any, source: any): any {
    const result = { ...target };
    
    for (const key of Object.keys(source)) {
      if (
        typeof source[key] === 'object' &&
        source[key] !== null &&
        !Array.isArray(source[key])
      ) {
        result[key] = this.deepMerge(result[key] || {}, source[key]);
      } else {
        result[key] = source[key];
      }
    }
    
    return result;
  }
}
```
