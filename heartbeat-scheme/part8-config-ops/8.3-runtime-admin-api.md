# 8.3 运行时动态调整接口

## 8.3.1 Admin API 概述

Admin API 提供运行时动态调整心跳系统配置的能力，无需重启服务即可调整参数、查看状态、触发诊断操作。

## 8.3.2 API 设计原则

| 原则 | 说明 |
|------|------|
| **安全优先** | 所有写操作需要认证，敏感操作需要二次确认 |
| **幂等性** | 相同请求多次执行结果一致 |
| **审计追踪** | 所有配置变更记录审计日志 |
| **优雅降级** | API 异常不影响心跳核心功能 |

## 8.3.3 API 端点总览

```
GET    /api/v1/status              获取系统状态
GET    /api/v1/health              健康检查
GET    /api/v1/config              获取当前配置
PATCH  /api/v1/config              更新配置
GET    /api/v1/metrics             获取指标数据
POST   /api/v1/daemon/start        启动 Daemon
POST   /api/v1/daemon/stop         停止 Daemon
POST   /api/v1/daemon/pause        暂停 Daemon
POST   /api/v1/daemon/resume       恢复 Daemon
GET    /api/v1/daemon/history      获取状态历史
POST   /api/v1/recovery/trigger    手动触发恢复
POST   /api/v1/recovery/cancel     取消恢复操作
GET    /api/v1/recovery/history    获取恢复历史
POST   /api/v1/diagnostic/run      运行诊断
GET    /api/v1/diagnostic/report   获取诊断报告
```

## 8.3.5 Admin API 服务器实现

```typescript
import express, { Request, Response, NextFunction } from 'express';
import { v4 as uuidv4 } from 'uuid';

class AdminApiServer {
  private app: express.Application;
  private daemon: HeartbeatDaemon;
  private configLoader: ConfigLoader;
  private metricsCollector: MetricsCollector;
  private auditLogger: AuditLogger;
  private authenticator: Authenticator;
  
  constructor(
    daemon: HeartbeatDaemon,
    configLoader: ConfigLoader,
    metricsCollector: MetricsCollector
  ) {
    this.app = express();
    this.daemon = daemon;
    this.configLoader = configLoader;
    this.metricsCollector = metricsCollector;
    this.auditLogger = new AuditLogger();
    this.authenticator = new Authenticator();
    
    this.setupMiddleware();
    this.setupRoutes();
  }
  
  private setupMiddleware(): void {
    this.app.use(express.json());
    this.app.use(this.auditMiddleware.bind(this));
    this.app.use(this.authMiddleware.bind(this));
  }
  
  private setupRoutes(): void {
    this.app.get('/api/v1/status', this.getStatus.bind(this));
    this.app.get('/api/v1/health', this.getHealth.bind(this));
    
    this.app.get('/api/v1/config', this.getConfig.bind(this));
    this.app.patch('/api/v1/config', this.updateConfig.bind(this));
    
    this.app.get('/api/v1/metrics', this.getMetrics.bind(this));
    
    this.app.post('/api/v1/daemon/start', this.startDaemon.bind(this));
    this.app.post('/api/v1/daemon/stop', this.stopDaemon.bind(this));
    this.app.post('/api/v1/daemon/pause', this.pauseDaemon.bind(this));
    this.app.post('/api/v1/daemon/resume', this.resumeDaemon.bind(this));
    this.app.get('/api/v1/daemon/history', this.getDaemonHistory.bind(this));
    
    this.app.post('/api/v1/recovery/trigger', this.triggerRecovery.bind(this));
    this.app.post('/api/v1/recovery/cancel', this.cancelRecovery.bind(this));
    this.app.get('/api/v1/recovery/history', this.getRecoveryHistory.bind(this));
    
    this.app.post('/api/v1/diagnostic/run', this.runDiagnostic.bind(this));
    this.app.get('/api/v1/diagnostic/report', this.getDiagnosticReport.bind(this));
    
    this.app.use(this.errorHandler.bind(this));
  }
  
  private auditMiddleware(req: Request, res: Response, next: NextFunction): void {
    const startTime = Date.now();
    
    res.on('finish', () => {
      this.auditLogger.log({
        timestamp: new Date().toISOString(),
        method: req.method,
        path: req.path,
        statusCode: res.statusCode,
        durationMs: Date.now() - startTime,
        ip: req.ip,
        userAgent: req.get('user-agent')
      });
    });
    
    next();
  }
  
  private authMiddleware(req: Request, res: Response, next: NextFunction): void {
    if (req.method === 'GET' && req.path === '/api/v1/health') {
      return next();
    }
    
    const token = req.get('Authorization')?.replace('Bearer ', '');
    
    if (!token) {
      return res.status(401).json({ error: 'Missing authorization token' });
    }
    
    const isValid = this.authenticator.validateToken(token);
    
    if (!isValid) {
      return res.status(401).json({ error: 'Invalid authorization token' });
    }
    
    next();
  }
  
  private async getStatus(req: Request, res: Response): Promise<void> {
    try {
      const status: StatusResponse = {
        status: this.daemon.isRunning() ? 'running' : 
                this.daemon.isPaused() ? 'paused' : 'stopped',
        version: process.env.npm_package_version || '1.0.0',
        uptime: this.daemon.getUptime(),
        currentTask: this.getCurrentTask(),
        daemon: {
          state: this.daemon.state,
          uptime: this.daemon.getUptime(),
          timerStatus: this.daemon.getTimerStatus()
        },
        lastHealthChange: this.getLastHealthChange(),
        recoveryAttempts: this.daemon.daemonState.recoveryAttempts
      };
      
      res.json(status);
    } catch (error) {
      res.status(500).json({ error: 'Failed to get status' });
    }
  }
  
  private async getHealth(req: Request, res: Response): Promise<void> {
    const checks = {
      daemon: this.checkDaemonHealth(),
      collectors: this.checkCollectorsHealth(),
      persistence: this.checkPersistenceHealth(),
      notifications: this.checkNotificationsHealth()
    };
    
    const healthy = Object.values(checks).every(c => c.status === 'ok');
    
    res.status(healthy ? 200 : 503).json({ healthy, checks });
  }
  
  private async getConfig(req: Request, res: Response): Promise<void> {
    const layer = req.query.layer as string || 'effective';
    const workspace = req.query.workspace as string;
    
    try {
      let config;
      let source;
      
      switch (layer) {
        case 'global':
          config = this.configLoader.load().global;
          source = 'heartbeat.yaml';
          break;
        case 'effective':
          config = this.configLoader.getEffectiveConfig(workspace || 'default', {});
          source = 'merged';
          break;
        case 'all':
          config = this.configLoader.load();
          source = 'heartbeat.yaml';
          break;
        default:
          res.status(400).json({ error: 'Invalid layer parameter' });
          return;
      }
      
      res.json({
        layer,
        config,
        source,
        lastModified: new Date().toISOString()
      });
    } catch (error) {
      res.status(500).json({ error: 'Failed to get config' });
    }
  }
  
  private async updateConfig(req: Request, res: Response): Promise<void> {
    const { updates, reason, confirmationToken } = req.body;
    
    const sensitiveUpdates = updates.filter((u: ConfigUpdate) => 
      this.isSensitivePath(u.path)
    );
    
    if (sensitiveUpdates.length > 0 && !confirmationToken) {
      const token = uuidv4();
      this.storeConfirmationToken(token, sensitiveUpdates);
      
      res.json({
        success: false,
        requiresConfirmation: true,
        confirmationToken: token,
        message: 'Sensitive configuration changes require confirmation'
      });
      return;
    }
    
    const result = this.applyConfigUpdates(updates, reason);
    
    res.json(result);
  }
  
  private async getMetrics(req: Request, res: Response): Promise<void> {
    const period = req.query.period as string || '1h';
    const metrics = req.query.metrics as string;
    
    const periodMs = this.parsePeriod(period);
    const endTime = Date.now();
    const startTime = endTime - periodMs;
    
    const metricsData = this.metricsCollector.getMetrics({
      start: startTime,
      end: endTime,
      names: metrics?.split(',')
    });
    
    res.json({
      period: {
        start: new Date(startTime).toISOString(),
        end: new Date(endTime).toISOString()
      },
      metrics: metricsData
    });
  }
  
  private async startDaemon(req: Request, res: Response): Promise<void> {
    const { taskId, workspaceId, config } = req.body;
    
    try {
      const page = await this.getPage();
      await this.daemon.start(taskId, page);
      
      res.json({
        success: true,
        taskId,
        startedAt: new Date().toISOString(),
        config: this.daemon.getConfig()
      });
    } catch (error) {
      res.status(500).json({ 
        success: false, 
        error: error instanceof Error ? error.message : 'Failed to start daemon' 
      });
    }
  }
  
  private async stopDaemon(req: Request, res: Response): Promise<void> {
    try {
      const summary = await this.daemon.stop();
      
      res.json({
        success: true,
        summary
      });
    } catch (error) {
      res.status(500).json({ 
        success: false, 
        error: error instanceof Error ? error.message : 'Failed to stop daemon' 
      });
    }
  }
  
  private async pauseDaemon(req: Request, res: Response): Promise<void> {
    try {
      this.daemon.pause();
      
      res.json({
        success: true,
        pausedAt: new Date().toISOString(),
        state: 'paused'
      });
    } catch (error) {
      res.status(500).json({ 
        success: false, 
        error: error instanceof Error ? error.message : 'Failed to pause daemon' 
      });
    }
  }
  
  private async resumeDaemon(req: Request, res: Response): Promise<void> {
    try {
      const page = await this.getPage();
      await this.daemon.resume(page);
      
      res.json({
        success: true,
        resumedAt: new Date().toISOString(),
        pausedDuration: this.daemon.daemonState.totalPausedDuration,
        state: 'running'
      });
    } catch (error) {
      res.status(500).json({ 
        success: false, 
        error: error instanceof Error ? error.message : 'Failed to resume daemon' 
      });
    }
  }
  
  private async triggerRecovery(req: Request, res: Response): Promise<void> {
    const { type, reason, force } = req.body;
    
    const recoveryId = uuidv4();
    
    try {
      await this.daemon.triggerRecovery(type, { force, reason, manual: true });
      
      res.json({
        success: true,
        recoveryId,
        type,
        triggeredAt: new Date().toISOString(),
        status: 'pending'
      });
    } catch (error) {
      res.status(500).json({ 
        success: false, 
        error: error instanceof Error ? error.message : 'Failed to trigger recovery' 
      });
    }
  }
  
  private async runDiagnostic(req: Request, res: Response): Promise<void> {
    const { checks, timeout } = req.body;
    
    const runId = uuidv4();
    
    this.runDiagnosticChecks(runId, checks, timeout);
    
    res.json({
      runId,
      status: 'running',
      startedAt: new Date().toISOString()
    });
  }
  
  private errorHandler(err: Error, req: Request, res: Response, next: NextFunction): void {
    console.error('[AdminAPI] Error:', err);
    
    res.status(500).json({
      error: 'Internal server error',
      message: process.env.NODE_ENV === 'development' ? err.message : undefined
    });
  }
  
  listen(port: number): void {
    this.app.listen(port, () => {
      console.log(`[AdminAPI] Server listening on port ${port}`);
    });
  }
}
```

## 8.3.6 认证与授权

### 8.3.6.1 Token 认证

```typescript
interface TokenPayload {
  sub: string;
  iat: number;
  exp: number;
  permissions: Permission[];
}

type Permission = 
  | 'read:status'
  | 'read:config'
  | 'write:config'
  | 'control:daemon'
  | 'trigger:recovery'
  | 'run:diagnostic';

class Authenticator {
  private secret: string;
  private tokens: Map<string, TokenPayload> = new Map();
  
  constructor(secret: string = process.env.HEARTBEAT_ADMIN_SECRET || 'default-secret') {
    this.secret = secret;
  }
  
  generateToken(subject: string, permissions: Permission[], expiresIn: number = 3600): string {
    const now = Math.floor(Date.now() / 1000);
    
    const payload: TokenPayload = {
      sub: subject,
      iat: now,
      exp: now + expiresIn,
      permissions
    };
    
    const token = this.encodeToken(payload);
    this.tokens.set(token, payload);
    
    return token;
  }
  
  validateToken(token: string): boolean {
    const payload = this.tokens.get(token);
    
    if (!payload) {
      return false;
    }
    
    if (payload.exp < Math.floor(Date.now() / 1000)) {
      this.tokens.delete(token);
      return false;
    }
    
    return true;
  }
  
  getPermissions(token: string): Permission[] {
    const payload = this.tokens.get(token);
    return payload?.permissions || [];
  }
  
  hasPermission(token: string, permission: Permission): boolean {
    const permissions = this.getPermissions(token);
    return permissions.includes(permission);
  }
  
  revokeToken(token: string): void {
    this.tokens.delete(token);
  }
  
  private encodeToken(payload: TokenPayload): string {
    return Buffer.from(JSON.stringify(payload)).toString('base64');
  }
}
```

### 8.3.6.2 权限中间件

```typescript
class AuthorizationMiddleware {
  private authenticator: Authenticator;
  
  constructor(authenticator: Authenticator) {
    this.authenticator = authenticator;
  }
  
  requirePermission(permission: Permission) {
    return (req: Request, res: Response, next: NextFunction) => {
      const token = req.get('Authorization')?.replace('Bearer ', '');
      
      if (!token) {
        return res.status(401).json({ error: 'Missing authorization token' });
      }
      
      if (!this.authenticator.hasPermission(token, permission)) {
        return res.status(403).json({ error: 'Insufficient permissions' });
      }
      
      next();
    };
  }
}
```

```typescript
    };
    
    this.logs.push(log);
    
    if (this.logs.length > this.MAX_LOGS) {
      this.logs = this.logs.slice(-this.MAX_LOGS);
    }
    
    this.persistLog(log);
  }
  
  private persistLog(log: ConfigAuditLog): void {
    const logPath = `logs/config-audit-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, JSON.stringify(log) + '\n');
  }
  
  getLogs(filter?: {
    user?: string;
    path?: string;
    since?: Date;
  }): ConfigAuditLog[] {
    let filtered = [...this.logs];
    
    if (filter?.user) {
      filtered = filtered.filter(log => log.user === filter.user);
    }
    
    if (filter?.path) {
      filtered = filtered.filter(log => log.path.startsWith(filter.path!));
    }
    
    if (filter?.since) {
      filtered = filtered.filter(log => new Date(log.timestamp) >= filter.since!);
    }
    
    return filtered;
  }
}
```
