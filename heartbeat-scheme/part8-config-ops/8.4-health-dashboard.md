# 8.4 心跳健康度仪表盘

## 8.4.1 仪表盘概述

心跳健康度仪表盘是心跳系统的可视化监控界面，提供实时状态展示、历史趋势分析、告警通知和诊断工具入口。

## 8.4.3 核心指标定义

### 8.4.3.1 实时状态指标

| 指标名称 | 类型 | 说明 | 更新频率 |
|---------|------|------|---------|
| `current_health_mode` | gauge | 当前健康模式 | 实时 |
| `daemon_state` | gauge | Daemon 状态 | 实时 |
| `uptime_seconds` | gauge | 运行时长（秒） | 1s |
| `recovery_attempts_total` | counter | 恢复尝试次数 | 实时 |
| `last_collect_timestamp` | gauge | 最后采集时间 | 采集时 |

### 8.4.3.2 采集层指标

| 指标名称 | 类型 | 说明 | 标签 |
|---------|------|------|------|
| `layer_collect_total` | counter | 采集总次数 | `layer` |
| `layer_collect_duration_ms` | histogram | 采集耗时分布 | `layer` |
| `layer_collect_errors_total` | counter | 采集错误次数 | `layer`, `error_type` |
| `layer_signal_detected_total` | counter | 检测到信号次数 | `layer`, `signal_type` |

### 8.4.3.3 健康状态指标

| 指标名称 | 类型 | 说明 | 标签 |
|---------|------|------|------|
| `health_mode_duration_seconds` | histogram | 各模式持续时间 | `mode` |
| `health_transitions_total` | counter | 状态转换次数 | `from`, `to` |
| `time_to_recovery_seconds` | histogram | 恢复耗时 | `type` |
| `recovery_success_rate` | gauge | 恢复成功率 | `type` |

### 8.4.3.4 告警指标

| 指标名称 | 类型 | 说明 | 标签 |
|---------|------|------|------|
| `alerts_total` | counter | 告警总数 | `level`, `type` |
| `alerts_active` | gauge | 活跃告警数 | `level` |
| `alert_notification_sent_total` | counter | 通知发送次数 | `channel`, `level` |
| `alert_notification_failed_total` | counter | 通知失败次数 | `channel`, `level` |

```typescript
interface DashboardDataService {
  getStatus(): Promise<DashboardStatus>;
  getMetrics(period: string): Promise<DashboardMetrics>;
  getHealthTrend(period: string): Promise<HealthTrendDataPoint[]>;
  getRecoveryHistory(limit: number): Promise<RecoveryRecord[]>;
  getActiveAlerts(): Promise<Alert[]>;
}

class DashboardDataServiceImpl implements DashboardDataService {
  private metricsCollector: MetricsCollector;
  private daemon: HeartbeatDaemon;
  private alertManager: AlertManager;
  private recoveryHistory: RecoveryHistoryStore;
  
  constructor(
    metricsCollector: MetricsCollector,
    daemon: HeartbeatDaemon,
    alertManager: AlertManager,
    recoveryHistory: RecoveryHistoryStore
  ) {
    this.metricsCollector = metricsCollector;
    this.daemon = daemon;
    this.alertManager = alertManager;
    this.recoveryHistory = recoveryHistory;
  }
  
  async getStatus(): Promise<DashboardStatus> {
    const daemonState = this.daemon.daemonState;
    
    return {
      healthMode: daemonState.currentHealthMode,
      daemonState: this.daemon.state,
      uptime: this.daemon.getUptime(),
      taskId: daemonState.currentTaskId,
      workspaceId: daemonState.workspaceId,
      startedAt: daemonState.startedAt,
      recoveryAttempts: daemonState.recoveryAttempts,
      lastCollectTime: daemonState.lastCollectTime
    };
  }
  
  async getMetrics(period: string): Promise<DashboardMetrics> {
    const periodMs = this.parsePeriod(period);
    const endTime = Date.now();
    const startTime = endTime - periodMs;
    
    const layerMetrics = await this.metricsCollector.getLayerMetrics(startTime, endTime);
    
    return {
      layer1: {
        collectTotal: layerMetrics.layer1.collectTotal,
        collectDuration: layerMetrics.layer1.duration,
        errorCount: layerMetrics.layer1.errors,
        lastCollectTime: layerMetrics.layer1.lastCollectTime
      },
      layer2: {
        collectTotal: layerMetrics.layer2.collectTotal,
        collectDuration: layerMetrics.layer2.duration,
        errorCount: layerMetrics.layer2.errors,
        lastCollectTime: layerMetrics.layer2.lastCollectTime
      },
      layer3: {
        collectTotal: layerMetrics.layer3.collectTotal,
        collectDuration: layerMetrics.layer3.duration,
        errorCount: layerMetrics.layer3.errors,
        lastCollectTime: layerMetrics.layer3.lastCollectTime
      }
    };
  }
  
  async getHealthTrend(period: string): Promise<HealthTrendDataPoint[]> {
    const periodMs = this.parsePeriod(period);
    const endTime = Date.now();
    const startTime = endTime - periodMs;
    
    const history = this.daemon.getStateHistory();
    
    return history
      .filter(point => point.timestamp >= startTime)
      .map(point => ({
        timestamp: point.timestamp,
        healthMode: point.healthMode,
        duration: point.duration || 0
      }));
  }
  
  async getRecoveryHistory(limit: number): Promise<RecoveryRecord[]> {
    return this.recoveryHistory.getRecent(limit);
  }
  
  async getActiveAlerts(): Promise<Alert[]> {
    return this.alertManager.getActiveAlerts();
  }
  
  private parsePeriod(period: string): number {
    const periodMap: Record<string, number> = {
      '1h': 3600000,
      '6h': 21600000,
      '24h': 86400000,
      '7d': 604800000
    };
    
    return periodMap[period] || periodMap['1h'];
  }
}
```

### 8.4.5.2 WebSocket 实时推送

```typescript
import { Server as WebSocketServer, WebSocket } from 'ws';

class DashboardWebSocketServer {
  private wss: WebSocketServer;
  private clients: Set<WebSocket> = new Set();
  private dataService: DashboardDataService;
  private pushInterval: ReturnType<typeof setInterval> | null = null;
  
  constructor(port: number, dataService: DashboardDataService) {
    this.wss = new WebSocketServer({ port });
    this.dataService = dataService;
    
    this.setupServer();
  }
  
  private setupServer(): void {
    this.wss.on('connection', (ws) => {
      this.clients.add(ws);
      
      console.log(`[DashboardWS] Client connected. Total: ${this.clients.size}`);
      
      this.sendInitialData(ws);
      
      ws.on('close', () => {
        this.clients.delete(ws);
        console.log(`[DashboardWS] Client disconnected. Total: ${this.clients.size}`);
      });
      
      ws.on('message', (message) => {
        this.handleMessage(ws, message.toString());
      });
    });
    
    this.startPushInterval();
  }
  
  private async sendInitialData(ws: WebSocket): Promise<void> {
    const status = await this.dataService.getStatus();
    const metrics = await this.dataService.getMetrics('1h');
    const alerts = await this.dataService.getActiveAlerts();
    
    this.send(ws, {
      type: 'initial',
      data: { status, metrics, alerts }
    });
  }
  
  private startPushInterval(): void {
    this.pushInterval = setInterval(async () => {
      if (this.clients.size === 0) return;
      
      const status = await this.dataService.getStatus();
      
      this.broadcast({
        type: 'status_update',
        data: status
      });
    }, 1000);
  }
  
  private handleMessage(ws: WebSocket, message: string): void {
    try {
      const parsed = JSON.parse(message);
      
      switch (parsed.type) {
        case 'subscribe':
          this.handleSubscribe(ws, parsed.channels);
          break;
        case 'unsubscribe':
          this.handleUnsubscribe(ws, parsed.channels);
          break;
        default:
          console.warn(`[DashboardWS] Unknown message type: ${parsed.type}`);
      }
    } catch (error) {
      console.error('[DashboardWS] Failed to parse message:', error);
    }
  }
  
  private handleSubscribe(ws: WebSocket, channels: string[]): void {
    console.log(`[DashboardWS] Client subscribed to: ${channels.join(', ')}`);
  }
  
  private handleUnsubscribe(ws: WebSocket, channels: string[]): void {
    console.log(`[DashboardWS] Client unsubscribed from: ${channels.join(', ')}`);
  }
  
  private send(ws: WebSocket, data: any): void {
    if (ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify(data));
    }
  }
  
  private broadcast(data: any): void {
    const message = JSON.stringify(data);
    
    for (const client of this.clients) {
      if (client.readyState === WebSocket.OPEN) {
        client.send(message);
      }
    }
  }
  
  broadcastAlert(alert: Alert): void {
    this.broadcast({
      type: 'alert',
      data: alert
    });
  }
  
  broadcastHealthChange(from: HealthMode, to: HealthMode): void {
    this.broadcast({
      type: 'health_change',
      data: { from, to, timestamp: Date.now() }
    });
  }
  
  close(): void {
    if (this.pushInterval) {
      clearInterval(this.pushInterval);
    }
    
    this.wss.close();
  }
}
```

```typescript
  const [metrics, setMetrics] = useState<DashboardMetrics | null>(null);

        case 'initial':
          setStatus(message.data.status);
          setMetrics(message.data.metrics);
          setAlerts(message.data.alerts);
          break;
        case 'status_update':
          setStatus(message.data);
          break;
        case 'alert':
          setAlerts(prev => [message.data, ...prev]);
          break;
        case 'health_change':
          break;

        <div className="col-span-3">

            daemonState={status?.daemonState}
            className="mt-4"

            alerts={alerts}

            className="mt-4"

      </div>
    </div>
  );
```
