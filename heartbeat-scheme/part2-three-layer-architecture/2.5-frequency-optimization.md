# 2.5 心跳频率动态调优

## 2.5.1 设计目标

频率调优机制旨在根据系统状态动态调整心跳采集频率，实现：

1. **高压加速**：在异常或高风险状态下提高采集频率，快速响应问题
2. **空闲降频**：在正常或空闲状态下降低采集频率，节省资源
3. **平滑过渡**：频率调整平滑进行，避免剧烈波动
4. **成本优化**：在保证监控效果的前提下最小化采集开销

## 2.5.4 高压加速策略

```typescript
class AccelerationStrategy {
  private readonly ACCELERATION_TRIGGERS: Record<Layer1Mode | Layer2Mode, boolean> = {
    'normal': false,
    'task-interrupted': false,
    'terminal-hang': true,
    'modal-blocking': false,
    'task-completed': false,
    'model-stalled': true,
    'model-queuing': false,
    'model-streaming': false,
    'no-active-turn': false
  };

  shouldAccelerate(layer1Mode: Layer1Mode, layer2Mode: Layer2Mode): boolean {
    if (this.ACCELERATION_TRIGGERS[layer1Mode]) {
      return true;
    }
    
    if (this.ACCELERATION_TRIGGERS[layer2Mode]) {
      return true;
    }
    
    return false;
  }

  getAccelerationReason(layer1Mode: Layer1Mode, layer2Mode: Layer2Mode): string {
    if (layer1Mode === 'terminal-hang') {
      return 'terminal-hang: 终端阻塞，需要快速监控恢复';
    }
    
    if (layer2Mode === 'model-stalled') {
      return 'model-stalled: 模型卡死，需要快速检测恢复';
    }
    
    return 'unknown';
  }

  getAcceleratedFrequencies(): BaseFrequencyConfig {
    return { ...ACCELERATED_FREQUENCIES };
  }
}
```

## 2.5.5 空闲降频策略

```typescript
class IdleStrategy {
  private readonly IDLE_THRESHOLD = 5 * 60 * 1000;
  private readonly ACTIVITY_INDICATORS = [
    'user_input',
    'model_streaming',
    'task_executing',
    'file_operation'
  ];

  private lastActivityTime = Date.now();

  recordActivity(type: string): void {
    if (this.ACTIVITY_INDICATORS.includes(type)) {
      this.lastActivityTime = Date.now();
    }
  }

  shouldEnterIdle(): boolean {
    const idleTime = Date.now() - this.lastActivityTime;
    return idleTime >= this.IDLE_THRESHOLD;
  }

  shouldExitIdle(layer1Result: Layer1Result, layer2Result: Layer2Result): boolean {
    if (layer1Result.mode !== 'normal') {
      return true;
    }
    
    if (layer2Result.mode === 'model-streaming') {
      return true;
    }
    
    if (layer2Result.payload.charsDelta > 0) {
      return true;
    }
    
    return false;
  }

  getIdleFrequencies(): BaseFrequencyConfig {
    return { ...IDLE_FREQUENCIES };
  }

  getIdleDuration(): number {
    return Date.now() - this.lastActivityTime;
  }
}
```

```typescript
const IDLE_CONDITIONS: IdleCondition[] = [
  {

  },
  {
    check: (ctx) => ctx.layer2?.payload.charsDelta === 0,
    description: '无文本增量'
  },
  {
    check: (ctx) => !ctx.layer1?.payload.hasCancelBtn,
    description: '无执行中按钮'
  },
  {
    check: (ctx) => ctx.layer2?.mode !== 'model-streaming',
    description: '模型未输出'
  }
];

function checkIdleConditions(context: SignalContext): {
  isIdle: boolean;
  matchedConditions: string[];
} {
  const matchedConditions = IDLE_CONDITIONS
    .filter(c => c.check(context))
    .map(c => c.description);
  
  return {
    isIdle: matchedConditions.length === IDLE_CONDITIONS.length,
    matchedConditions
  };
}
```

## 2.5.6 频率管理器

```typescript
class FrequencyManager {
  private state: FrequencyState = {
    mode: 'normal',
    currentFrequencies: { ...DEFAULT_FREQUENCIES },
    lastModeChange: Date.now(),
    modeChangeReason: '初始化',
    consecutiveNormalCount: 0,
    idleStartTime: null
  };

  private metrics: FrequencyMetrics = {
    modeDistribution: {
      'normal': 0,
      'accelerated': 0,
      'idle': 0,
      'paused': 0
    },
    adjustmentCount: 0,
    avgTimeInMode: {
      'normal': 0,
      'accelerated': 0,
      'idle': 0,
      'paused': 0
    },
    lastAdjustment: null
  };

  private accelerationStrategy: AccelerationStrategy;
  private idleStrategy: IdleStrategy;
  private readonly NORMAL_THRESHOLD = 3;

  constructor() {
    this.accelerationStrategy = new AccelerationStrategy();
    this.idleStrategy = new IdleStrategy();
  }

  update(layer1Result: Layer1Result, layer2Result: Layer2Result): FrequencyAdjustment | null {
    const newMode = this.determineMode(layer1Result, layer2Result);
    
    if (newMode !== this.state.mode) {
      return this.adjustMode(newMode, layer1Result, layer2Result);
    }
    
    this.updateMetrics();
    return null;
  }

  private determineMode(layer1Result: Layer1Result, layer2Result: Layer2Result): FrequencyMode {
    if (layer1Result.mode === 'task-interrupted') {
      return 'paused';
    }
    
    if (this.accelerationStrategy.shouldAccelerate(layer1Result.mode, layer2Result.mode)) {
      return 'accelerated';
    }
    
    if (this.idleStrategy.shouldEnterIdle()) {
      return 'idle';
    }
    
    return 'normal';
  }

  private adjustMode(
    newMode: FrequencyMode,
    layer1Result: Layer1Result,
    layer2Result: Layer2Result
  ): FrequencyAdjustment {
    const oldMode = this.state.mode;
    const now = Date.now();
    
    let newFrequencies: BaseFrequencyConfig;
    let reason: string;
    
    switch (newMode) {
      case 'accelerated':
        newFrequencies = this.accelerationStrategy.getAcceleratedFrequencies();
        reason = this.accelerationStrategy.getAccelerationReason(
          layer1Result.mode,
          layer2Result.mode
        );
        break;
      case 'idle':
        newFrequencies = this.idleStrategy.getIdleFrequencies();
        reason = `空闲超过 ${Math.round(this.idleStrategy.getIdleDuration() / 1000)} 秒`;
        break;
      case 'paused':
        newFrequencies = { layer1: 0, layer2: 0, layer3: 0 };
        reason = '任务中断';
        break;
      default:
        newFrequencies = { ...DEFAULT_FREQUENCIES };
        reason = '恢复正常频率';
    }
    
    const adjustment: FrequencyAdjustment = {
      from: oldMode,
      to: newMode,
      reason,
      timestamp: now,
      newFrequencies
    };
    
    this.state = {
      mode: newMode,
      currentFrequencies: newFrequencies,
      lastModeChange: now,
      modeChangeReason: reason,
      consecutiveNormalCount: newMode === 'normal' ? this.state.consecutiveNormalCount + 1 : 0,
      idleStartTime: newMode === 'idle' ? now : null
    };
    
    this.metrics.adjustmentCount++;
    this.metrics.lastAdjustment = adjustment;
    
    console.log(`[Frequency] ${oldMode} -> ${newMode}: ${reason}`);
    
    return adjustment;
  }

  private updateMetrics(): void {
    this.metrics.modeDistribution[this.state.mode]++;
  }

  getCurrentFrequencies(): BaseFrequencyConfig {
    return { ...this.state.currentFrequencies };
  }

  getCurrentMode(): FrequencyMode {
    return this.state.mode;
  }

  getMetrics(): FrequencyMetrics {
    return { ...this.metrics };
  }

  forceMode(mode: FrequencyMode, reason: string): FrequencyAdjustment | null {
    if (mode === this.state.mode) {
      return null;
    }
    
    const mockLayer1: Layer1Result = {
      layer: 1,
      payload: {} as any,
      mode: 'normal',
      cost: 0
    };
    
    const mockLayer2: Layer2Result = {
      layer: 2,
      payload: {} as any,
      mode: 'normal',
      cost: 0
    };
    
    return this.adjustMode(mode, mockLayer1, mockLayer2);
  }
}
```

## 2.5.7 平滑过渡机制

```typescript
class SmoothTransition {
  private readonly TRANSITION_DURATION = 30000;
  private transitionInProgress = false;
  private transitionStart: number | null = null;
  private fromFrequencies: BaseFrequencyConfig | null = null;
  private toFrequencies: BaseFrequencyConfig | null = null;

  startTransition(
    from: BaseFrequencyConfig,
    to: BaseFrequencyConfig
  ): void {
    this.transitionInProgress = true;
    this.transitionStart = Date.now();
    this.fromFrequencies = from;
    this.toFrequencies = to;
  }

  getCurrentFrequencies(): BaseFrequencyConfig | null {
    if (!this.transitionInProgress || !this.fromFrequencies || !this.toFrequencies) {
      return null;
    }
    
    const elapsed = Date.now() - this.transitionStart!;
    const progress = Math.min(1, elapsed / this.TRANSITION_DURATION);
    
    return {
      layer1: this.interpolate(this.fromFrequencies.layer1, this.toFrequencies.layer1, progress),
      layer2: this.interpolate(this.fromFrequencies.layer2, this.toFrequencies.layer2, progress),
      layer3: this.interpolate(this.fromFrequencies.layer3, this.toFrequencies.layer3, progress)
    };
  }

  private interpolate(from: number, to: number, progress: number): number {
    if (from === to) return from;
    
    const easeProgress = this.easeInOutCubic(progress);
    return Math.round(from + (to - from) * easeProgress);
  }

  private easeInOutCubic(t: number): number {
    return t < 0.5
      ? 4 * t * t * t
      : 1 - Math.pow(-2 * t + 2, 3) / 2;
  }

  isComplete(): boolean {
    if (!this.transitionInProgress) return true;
    
    const elapsed = Date.now() - this.transitionStart!;
    if (elapsed >= this.TRANSITION_DURATION) {
      this.transitionInProgress = false;
      return true;
    }
    
    return false;
  }

  cancel(): void {
    this.transitionInProgress = false;
    this.transitionStart = null;
    this.fromFrequencies = null;
    this.toFrequencies = null;
  }
}
```

## 2.5.8 成本优化分析

### 不同频率下的成本对比

| 模式 | Layer 1 | Layer 2 | Layer 3 | 每小时采集次数 | 每小时总耗时 |
|------|---------|---------|---------|----------------|--------------|
| 正常 | 5s | 15s | 60s | 720+240+60=1020 | ~3.7s |
| 加速 | 2s | 5s | 30s | 1800+720+120=2640 | ~9.6s |
| 空闲 | 10s | 30s | 120s | 360+120+30=510 | ~1.9s |
| 暂停 | - | - | - | 0 | 0s |

### 节省计算

```typescript
interface CostSavings {
  normalCost: number;
  optimizedCost: number;
  savedPercentage: number;
  savedTime: number;
}

function calculateCostSavings(
  normalModeDuration: number,
  acceleratedModeDuration: number,
  idleModeDuration: number
): CostSavings {
  const normalCostPerHour = 3.7;
  const acceleratedCostPerHour = 9.6;
  const idleCostPerHour = 1.9;
  
  const normalCost = (normalModeDuration / 3600000) * normalCostPerHour;
  const acceleratedCost = (acceleratedModeDuration / 3600000) * acceleratedCostPerHour;
  const idleCost = (idleModeDuration / 3600000) * idleCostPerHour;
  
  const optimizedCost = normalCost + acceleratedCost + idleCost;
  const alwaysNormalCost = ((normalModeDuration + acceleratedModeDuration + idleModeDuration) / 3600000) * normalCostPerHour;
  
  const savedTime = alwaysNormalCost - optimizedCost;
  const savedPercentage = (savedTime / alwaysNormalCost) * 100;
  
  return {
    normalCost: alwaysNormalCost,
    optimizedCost,
    savedPercentage,
    savedTime
  };
}
```

```typescript
class FrequencyAdjuster {
  private frequencyManager: FrequencyManager;
  private smoothTransition: SmoothTransition;
  private layer1Timer: NodeJS.Timeout | null = null;
  private layer2Timer: NodeJS.Timeout | null = null;
  private layer3Timer: NodeJS.Timeout | null = null;
  
  private layer1Collector: Layer1Collector;
  private layer2Collector: Layer2Collector;
  private layer3Collector: Layer3Collector;
  
  private currentPage: Page | null = null;

  constructor() {
    this.frequencyManager = new FrequencyManager();
    this.smoothTransition = new SmoothTransition();
  }

  async start(page: Page): Promise<void> {
    this.currentPage = page;
    const frequencies = this.frequencyManager.getCurrentFrequencies();
    this.startTimers(frequencies);
  }

  async stop(): Promise<void> {
    this.stopTimers();
  }

  private startTimers(frequencies: BaseFrequencyConfig): void {
    this.stopTimers();
    
    if (frequencies.layer1 > 0) {
      this.layer1Timer = setInterval(
        () => this.collectLayer1(),
        frequencies.layer1
      );
    }
    
    if (frequencies.layer2 > 0) {
      this.layer2Timer = setInterval(
        () => this.collectLayer2(),
        frequencies.layer2
      );
    }
    
    if (frequencies.layer3 > 0) {
      this.layer3Timer = setInterval(
        () => this.collectLayer3(),
        frequencies.layer3
      );
    }
  }

  private stopTimers(): void {
    if (this.layer1Timer) {
      clearInterval(this.layer1Timer);
      this.layer1Timer = null;
    }
    if (this.layer2Timer) {
      clearInterval(this.layer2Timer);
      this.layer2Timer = null;
    }
    if (this.layer3Timer) {
      clearInterval(this.layer3Timer);
      this.layer3Timer = null;
    }
  }

  private async collectLayer1(): Promise<void> {
    if (!this.currentPage) return;
    
    const result = await this.layer1Collector.collect(this.currentPage);
    
    const layer2Result = this.getLastLayer2Result();
    const adjustment = this.frequencyManager.update(result, layer2Result);
    
    if (adjustment) {
      this.applyAdjustment(adjustment);
    }
  }

  private async collectLayer2(): Promise<void> {
    if (!this.currentPage) return;
    
    const layer1Context = this.getLastLayer1Result();
    const result = await this.layer2Collector.collect(this.currentPage, layer1Context);
    
    const layer1Result = this.getLastLayer1Result();
    const adjustment = this.frequencyManager.update(layer1Result!, result);
    
    if (adjustment) {
      this.applyAdjustment(adjustment);
    }
  }

  private async collectLayer3(): Promise<void> {
    if (!this.currentPage) return;
    
    const layer1Context = this.getLastLayer1Result();
    const layer2Context = this.getLastLayer2Result();
    await this.layer3Collector.collect(
      this.currentPage,
      layer1Context,
      layer2Context
    );
  }

  private applyAdjustment(adjustment: FrequencyAdjustment): void {
    console.log(`[FrequencyAdjuster] 应用频率调整: ${adjustment.from} -> ${adjustment.to}`);
    
    this.smoothTransition.startTransition(
      this.frequencyManager.getCurrentFrequencies(),
      adjustment.newFrequencies
    );
    
    this.startTimers(adjustment.newFrequencies);
  }

  private getLastLayer1Result(): Layer1Result | null {
    return null;
  }

  private getLastLayer2Result(): Layer2Result | null {
    return null;
  }
}
```

### 1. 避免频繁切换

```typescript
const MIN_MODE_DURATION = 30000;

class ModeStabilityChecker {
  private lastModeChange = 0;

  shouldAllowModeChange(): boolean {
    const now = Date.now();
    if (now - this.lastModeChange < MIN_MODE_DURATION) {
      return false;
    }
    return true;
  }

  recordModeChange(): void {
    this.lastModeChange = Date.now();
  }
}
```

### 2. 优先级规则

```typescript
function resolveModeConflict(modes: FrequencyMode[]): FrequencyMode {
  const priority: FrequencyMode[] = [
    'paused',
    'accelerated',
    'normal',
    'idle'
  ];
  
  for (const mode of priority) {
    if (modes.includes(mode)) {
      return mode;
    }
  }
  
  return 'normal';
}
```

### 3. 用户覆盖

```typescript
class UserOverride {
  private userMode: FrequencyMode | null = null;

  setUserMode(mode: FrequencyMode): void {
    this.userMode = mode;
  }

  clearUserMode(): void {
    this.userMode = null;
  }

  getEffectiveMode(systemMode: FrequencyMode): FrequencyMode {
    return this.userMode || systemMode;
  }
}
```
