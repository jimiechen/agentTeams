# 2.4 三层信号传递机制

## 2.4.1 设计目标

信号传递机制是三层心跳架构的核心协作机制，其设计目标是：

1. **状态复用**：避免重复采集，降低整体开销
2. **上下文传递**：下层采集结果为上层提供决策依据
3. **智能跳过**：根据下层状态决定是否执行上层采集
4. **一致性保证**：确保三层采集的时间戳和状态一致性

## 2.4.4 状态共享存储

```typescript
class SharedStateStore {
  private state: Map<string, any> = new Map();
  private subscribers: Map<string, Set<(value: any) => void>> = new Map();

  set<T>(key: string, value: T): void {
    const oldValue = this.state.get(key);
    this.state.set(key, value);
    
    if (oldValue !== value) {
      this.notifySubscribers(key, value);
    }
  }

  get<T>(key: string): T | undefined {
    return this.state.get(key) as T | undefined;
  }

  subscribe(key: string, handler: (value: any) => void): () => void {
    if (!this.subscribers.has(key)) {
      this.subscribers.set(key, new Set());
    }
    this.subscribers.get(key)!.add(handler);
    
    return () => {
      this.subscribers.get(key)?.delete(handler);
    };
  }

  private notifySubscribers(key: string, value: any): void {
    const handlers = this.subscribers.get(key);
    if (handlers) {
      handlers.forEach(handler => handler(value));
    }
  }

  getLayer1State(): Layer1Payload | null {
    return this.get<Layer1Payload>('layer1:payload');
  }

  getLayer2State(): Layer2Payload | null {
    return this.get<Layer2Payload>('layer2:payload');
  }

  getCurrentMode(): string {
    return this.get<string>('current:mode') || 'normal';
  }
}
```

## 2.4.5 信号传递实现

```typescript
class SignalPassingManager {
  private store: SharedStateStore;
  private signalBus: SignalBus;
  private sequenceId = 0;

  constructor() {
    this.store = new SharedStateStore();
    this.signalBus = this.createSignalBus();
    this.setupSignalHandlers();
  }

  private createSignalBus(): SignalBus {
    const handlers = new Map<SignalType, Set<SignalHandler>>();

    return {
      emit: (signal: Signal) => {
        const typeHandlers = handlers.get(signal.type);
        if (typeHandlers) {
          typeHandlers.forEach(handler => handler(signal));
        }
      },
      on: (type: SignalType, handler: SignalHandler) => {
        if (!handlers.has(type)) {
          handlers.set(type, new Set());
        }
        handlers.get(type)!.add(handler);
      },
      off: (type: SignalType, handler: SignalHandler) => {
        handlers.get(type)?.delete(handler);
      }
    };
  }

  private setupSignalHandlers(): void {
    this.signalBus.on('layer1:collected', this.handleLayer1Collected.bind(this));
    this.signalBus.on('layer2:collected', this.handleLayer2Collected.bind(this));
    this.signalBus.on('layer3:collected', this.handleLayer3Collected.bind(this));
  }

  private handleLayer1Collected(signal: Signal): void {
    const result = signal.payload as Layer1Result;
    
    this.store.set('layer1:payload', result.payload);
    this.store.set('layer1:mode', result.mode);
    this.store.set('layer1:timestamp', result.payload.timestamp);
    
    this.updateCurrentMode(result.mode);
  }

  private handleLayer2Collected(signal: Signal): void {
    const result = signal.payload as Layer2Result;
    
    this.store.set('layer2:payload', result.payload);
    this.store.set('layer2:mode', result.mode);
    this.store.set('layer2:timestamp', result.payload.timestamp);
    
    if (result.mode !== 'normal') {
      this.updateCurrentMode(result.mode);
    }
  }

  private handleLayer3Collected(signal: Signal): void {
    const result = signal.payload as Layer3Result;
    
    this.store.set('layer3:payload', result.payload);
    this.store.set('layer3:healthScore', result.healthScore);
    this.store.set('layer3:timestamp', result.payload.timestamp);
  }

  private updateCurrentMode(mode: string): void {
    const currentMode = this.store.getCurrentMode();
    
    if (currentMode !== mode) {
      this.store.set('current:mode', mode);
      this.signalBus.emit({
        type: 'layer1:mode_changed',
        source: 1,
        payload: { from: currentMode, to: mode },
        timestamp: Date.now()
      });
    }
  }

  getLayer1Context(): Layer1Result | null {
    const payload = this.store.get<Layer1Payload>('layer1:payload');
    const mode = this.store.get<Layer1Mode>('layer1:mode');
    
    if (!payload) return null;
    
    return {
      layer: 1,
      payload,
      mode: mode || 'normal',
      cost: 0
    };
  }

  getLayer2Context(): Layer2Result | null {
    const payload = this.store.get<Layer2Payload>('layer2:payload');
    const mode = this.store.get<Layer2Mode>('layer2:mode');
    
    if (!payload) return null;
    
    return {
      layer: 2,
      payload,
      mode: mode || 'normal',
      cost: 0
    };
  }

  buildContext(): SignalContext {
    return {
      layer1: this.getLayer1Context(),
      layer2: this.getLayer2Context(),
      timestamp: Date.now(),
      sequenceId: ++this.sequenceId
    };
  }
}
```

## 2.4.6 智能跳过策略

```typescript
class SkipStrategy {
  shouldSkipLayer2(layer1Result: Layer1Result): boolean {
    const skipModes: Layer1Mode[] = [
      'task-interrupted',
      'modal-blocking'
    ];
    
    return skipModes.includes(layer1Result.mode);
  }

  shouldSkipLayer3(layer1Result: Layer1Result, layer2Result: Layer2Result): boolean {
    if (layer1Result.mode === 'task-interrupted') {
      return true;
    }
    
    if (layer2Result.mode === 'model-queuing') {
      return true;
    }
    
    return false;
  }

  shouldAccelerate(layer1Result: Layer1Result, layer2Result: Layer2Result): boolean {
    const accelerateModes: (Layer1Mode | Layer2Mode)[] = [
      'terminal-hang',
      'model-stalled'
    ];
    
    return accelerateModes.includes(layer1Result.mode) ||
           accelerateModes.includes(layer2Result.mode);
  }
}
```

## 2.4.7 状态复用机制

```typescript
class StateReuseManager {
  private store: SharedStateStore;

  reuseLayer1StateForLayer2(): Partial<Layer2Payload> {
    const layer1Payload = this.store.get<Layer1Payload>('layer1:payload');
    
    if (!layer1Payload) {
      return {};
    }
    
    return {
      messageCount: 0,
      lastTurnIndex: -1
    };
  }

  reuseLayer1AndLayer2ForLayer3(): Partial<Layer3Payload> {
    const layer1Payload = this.store.get<Layer1Payload>('layer1:payload');
    const layer2Payload = this.store.get<Layer2Payload>('layer2:payload');
    
    return {
      taskListSnapshot: [],
      conversationLength: layer2Payload?.messageCount || 0
    };
  }

  getSharedFields(): {
    taskStatus: string | null;
    activeTaskId: string | null;
    messageCount: number;
  } {
    const layer1 = this.store.get<Layer1Payload>('layer1:payload');
    const layer2 = this.store.get<Layer2Payload>('layer2:payload');
    
    return {
      taskStatus: layer1?.taskStatus || null,
      activeTaskId: layer1?.activeTaskId || null,
      messageCount: layer2?.messageCount || 0
    };
  }
}
```

## 2.4.8 时间戳对齐

```typescript
class TimestampAligner {
  private readonly MAX_DRIFT = 1000;

  alignTimestamps(
    layer1Timestamp: number,
    layer2Timestamp: number,
    layer3Timestamp: number
  ): {
    aligned: boolean;
    drift: number;
    referenceTimestamp: number;
  } {
    const timestamps = [layer1Timestamp, layer2Timestamp, layer3Timestamp]
      .filter(t => t > 0);
    
    if (timestamps.length === 0) {
      return {
        aligned: true,
        drift: 0,
        referenceTimestamp: Date.now()
      };
    }
    
    const referenceTimestamp = Math.max(...timestamps);
    const minTimestamp = Math.min(...timestamps);
    const drift = referenceTimestamp - minTimestamp;
    
    return {
      aligned: drift <= this.MAX_DRIFT,
      drift,
      referenceTimestamp
    };
  }

  isTimestampFresh(timestamp: number, maxAge: number): boolean {
    return Date.now() - timestamp <= maxAge;
  }

  getTimestampAge(timestamp: number): number {
    return Date.now() - timestamp;
  }
}
```

## 2.4.9 三层协调器

```typescript
class ThreeLayerCoordinator {
  private layer1Collector: Layer1Collector;
  private layer2Collector: Layer2Collector;
  private layer3Collector: Layer3Collector;
  private signalManager: SignalPassingManager;
  private skipStrategy: SkipStrategy;
  private stateReuse: StateReuseManager;

  private layer1Timer: NodeJS.Timeout | null = null;
  private layer2Timer: NodeJS.Timeout | null = null;
  private layer3Timer: NodeJS.Timeout | null = null;

  async start(page: Page): Promise<void> {
    this.startLayer1(page);
    this.startLayer2(page);
    this.startLayer3(page);
  }

  async stop(): Promise<void> {
    if (this.layer1Timer) clearInterval(this.layer1Timer);
    if (this.layer2Timer) clearInterval(this.layer2Timer);
    if (this.layer3Timer) clearInterval(this.layer3Timer);
  }

  private startLayer1(page: Page): void {
    const interval = 5000;
    
    const collect = async () => {
      try {
        const result = await this.layer1Collector.collect(page);
        this.signalManager.signalBus.emit({
          type: 'layer1:collected',
          source: 1,
          payload: result,
          timestamp: Date.now()
        });
      } catch (error) {
        console.error('[Layer1] 采集失败:', error);
      }
    };

    this.layer1Timer = setInterval(collect, interval);
    collect();
  }

  private startLayer2(page: Page): void {
    const interval = 15000;
    
    const collect = async () => {
      try {
        const layer1Context = this.signalManager.getLayer1Context();
        
        if (layer1Context && this.skipStrategy.shouldSkipLayer2(layer1Context)) {
          console.log('[Layer2] 跳过采集，原因:', layer1Context.mode);
          return;
        }
        
        const result = await this.layer2Collector.collect(page, layer1Context || undefined);
        this.signalManager.signalBus.emit({
          type: 'layer2:collected',
          source: 2,
          payload: result,
          timestamp: Date.now()
        });
      } catch (error) {
        console.error('[Layer2] 采集失败:', error);
      }
    };

    this.layer2Timer = setInterval(collect, interval);
    setTimeout(collect, 1000);
  }

  private startLayer3(page: Page): void {
    const interval = 60000;
    
    const collect = async () => {
      try {
        const layer1Context = this.signalManager.getLayer1Context();
        const layer2Context = this.signalManager.getLayer2Context();
        
        if (layer1Context && layer2Context && 
            this.skipStrategy.shouldSkipLayer3(layer1Context, layer2Context)) {
          console.log('[Layer3] 跳过采集');
          return;
        }
        
        const result = await this.layer3Collector.collect(
          page,
          layer1Context || undefined,
          layer2Context || undefined
        );
        this.signalManager.signalBus.emit({
          type: 'layer3:collected',
          source: 3,
          payload: result,
          timestamp: Date.now()
        });
      } catch (error) {
        console.error('[Layer3] 采集失败:', error);
      }
    };

    this.layer3Timer = setInterval(collect, interval);
    setTimeout(collect, 2000);
  }

  getCurrentState(): SignalContext {
    return this.signalManager.buildContext();
  }
}
```

## 2.4.10 事件驱动更新

```typescript
class EventDrivenUpdater {
  private signalBus: SignalBus;
  private handlers: Map<string, () => void> = new Map();

  constructor(signalBus: SignalBus) {
    this.signalBus = signalBus;
    this.setupEventHandlers();
  }

  private setupEventHandlers(): void {
    this.signalBus.on('layer1:mode_changed', this.handleModeChanged.bind(this));
    this.signalBus.on('layer2:stalled_detected', this.handleStalledDetected.bind(this));
    this.signalBus.on('layer3:health_degraded', this.handleHealthDegraded.bind(this));
  }

  private handleModeChanged(signal: Signal): void {
    const { from, to } = signal.payload;
    console.log(`[Signal] 模式变更: ${from} -> ${to}`);
    
    const handler = this.handlers.get(`mode:${to}`);
    if (handler) {
      handler();
    }
  }

  private handleStalledDetected(signal: Signal): void {
    console.log('[Signal] 检测到模型卡死');
    
    const handler = this.handlers.get('stalled');
    if (handler) {
      handler();
    }
  }

  private handleHealthDegraded(signal: Signal): void {
    const result = signal.payload as Layer3Result;
    console.log(`[Signal] 健康度下降: ${result.healthScore}`);
    
    const handler = this.handlers.get('health_degraded');
    if (handler) {
      handler();
    }
  }

  registerHandler(event: string, handler: () => void): void {
    this.handlers.set(event, handler);
  }

  unregisterHandler(event: string): void {
    this.handlers.delete(event);
  }
}
```

## 2.4.11 信号传递成本分析

| 操作 | 耗时 | 说明 |
|------|------|------|
| 状态存储写入 | ~0.01ms | Map 操作 |
| 状态存储读取 | ~0.01ms | Map 操作 |
| 信号发射 | ~0.1ms | 遍历订阅者 |
| 订阅者回调 | ~0.5ms | 取决于回调逻辑 |
| 时间戳对齐 | ~0.01ms | 数学计算 |
| **总计** | **~0.63ms** | 可忽略不计 |
