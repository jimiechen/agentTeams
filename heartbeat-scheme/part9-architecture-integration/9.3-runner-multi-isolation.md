# 9.3 runner-multi.ts 多工作区隔离

## 9.3.1 多工作区隔离背景

runner-multi.ts 负责管理多个工作区的并行执行。在心跳架构中，每个工作区需要：

1. 拥有独立的 HeartbeatDaemon 实例
2. 隔离的心跳状态和恢复动作
3. 独立的健康状态追踪

## 9.3.2 WorkspaceRunner 架构

### 9.3.2.1 核心概念

每个 WorkspaceRunner 封装一个工作区的完整执行环境，包括：

- 独立的 HeartbeatDaemon
- 独立的 TaskMachine
- 独立的 ChatMutexMachine
- 独立的恢复执行器

### 9.3.2.2 WorkspaceRunner 定义

```typescript
interface WorkspaceRunnerConfig {
  workspaceId: string;
  workspaceType: 'pmcli' | 'devcli' | 'other';
  heartbeatConfig: Partial<HeartbeatDaemonConfig>;
  taskMachineConfig: Partial<TaskMachineConfig>;
  recoveryConfig: Partial<RecoveryConfig>;
}

interface WorkspaceRunnerState {
  workspaceId: string;
  status: 'idle' | 'running' | 'paused' | 'stopped' | 'error';
  currentTaskId: string | null;
  healthMode: HealthMode;
  startedAt: number | null;
  lastHeartbeatAt: number | null;
  recoveryAttempts: number;
}

class WorkspaceRunner {
  private _config: WorkspaceRunnerConfig;
  private _state: WorkspaceRunnerState;
  
  private _heartbeatDaemon: HeartbeatDaemon;
  private _taskMachine: TaskMachine;
  private _recoveryExecutor: RecoveryExecutor;
  
  private _page: Page | null = null;

  constructor(config: WorkspaceRunnerConfig) {
    this._config = config;
    
    this._state = {
      workspaceId: config.workspaceId,
      status: 'idle',
      currentTaskId: null,
      healthMode: 'healthy',
      startedAt: null,
      lastHeartbeatAt: null,
      recoveryAttempts: 0
    };

    this._heartbeatDaemon = new HeartbeatDaemon({
      ...config.heartbeatConfig,
      workspaceId: config.workspaceId,
      onHealthChange: (mode) => this.handleHealthChange(mode),
      onRecoveryNeeded: (action) => this.handleRecoveryNeeded(action)
    });

    this._taskMachine = new TaskMachine({
      ...config.taskMachineConfig,
      workspaceId: config.workspaceId
    });

    this._recoveryExecutor = new RecoveryExecutor({
      ...config.recoveryConfig,
      workspaceId: config.workspaceId,
      workspaceType: config.workspaceType
    });
  }

  get state(): Readonly<WorkspaceRunnerState> {
    return { ...this._state };
  }

  get workspaceId(): string {
    return this._config.workspaceId;
  }

  get heartbeatDaemon(): HeartbeatDaemon {
    return this._heartbeatDaemon;
  }

  async start(taskId: string, page: Page): Promise<void> {
    if (this._state.status !== 'idle') {
      throw new Error(`Cannot start workspace runner in state: ${this._state.status}`);
    }

    this._page = page;
    this._state.currentTaskId = taskId;
    this._state.startedAt = Date.now();
    this._state.status = 'running';

    await Promise.all([
      this._heartbeatDaemon.start(taskId, page),
      this._taskMachine.start(taskId, page)
    ]);

    this.emit('runner:started', {
      workspaceId: this._config.workspaceId,
      taskId,
      timestamp: this._state.startedAt
    });
  }

  async pause(): Promise<void> {
    if (this._state.status !== 'running') {
      return;
    }

    this._heartbeatDaemon.pause();
    this._state.status = 'paused';

    this.emit('runner:paused', {
      workspaceId: this._config.workspaceId,
      timestamp: Date.now()
    });
  }

  async resume(): Promise<void> {
    if (this._state.status !== 'paused' || !this._page) {
      return;
    }

    await this._heartbeatDaemon.resume(this._page);
    this._state.status = 'running';

    this.emit('runner:resumed', {
      workspaceId: this._config.workspaceId,
      timestamp: Date.now()
    });
  }

  async stop(): Promise<WorkspaceSummary> {
    if (this._state.status === 'stopped') {
      throw new Error('Workspace runner already stopped');
    }

    const heartbeatSummary = await this._heartbeatDaemon.stop();
    const taskSummary = await this._taskMachine.stop();

    const summary: WorkspaceSummary = {
      workspaceId: this._config.workspaceId,
      taskId: this._state.currentTaskId,
      startedAt: this._state.startedAt,
      stoppedAt: Date.now(),
      totalDuration: Date.now() - (this._state.startedAt || 0),
      finalHealthMode: this._state.healthMode,
      recoveryAttempts: this._state.recoveryAttempts,
      heartbeatSummary,
      taskSummary
    };

    this._state.status = 'stopped';
    this._page = null;

    this.emit('runner:stopped', {
      workspaceId: this._config.workspaceId,
      summary
    });

    return summary;
  }

  private handleHealthChange(mode: HealthMode): void {
    this._state.healthMode = mode;
    this._state.lastHeartbeatAt = Date.now();

    if (mode === 'task-interrupted' || mode === 'completed') {
      this.stop();
    }

    this.emit('health:changed', {
      workspaceId: this._config.workspaceId,
      mode,
      timestamp: Date.now()
    });
  }

  private async handleRecoveryNeeded(action: RecoveryAction): Promise<void> {
    this._state.recoveryAttempts++;

    try {
      await this._recoveryExecutor.execute(action);
    } catch (error) {
      console.error(`[WorkspaceRunner ${this._config.workspaceId}] Recovery failed:`, error);
    }
  }
}

interface WorkspaceSummary {
  workspaceId: string;
  taskId: string | null;
  startedAt: number | null;
  stoppedAt: number;
  totalDuration: number;
  finalHealthMode: HealthMode;
  recoveryAttempts: number;
  heartbeatSummary: DaemonSummary;
  taskSummary: any;
}
```

## 9.3.3 MultiRunner 协调器

### 9.3.3.1 协调器定义

```typescript
interface MultiRunnerConfig {
  maxConcurrentWorkspaces: number;
  defaultHeartbeatConfig: Partial<HeartbeatDaemonConfig>;
  defaultTaskMachineConfig: Partial<TaskMachineConfig>;
  defaultRecoveryConfig: Partial<RecoveryConfig>;
}

class MultiRunner {
  private _config: MultiRunnerConfig;
  private _runners: Map<string, WorkspaceRunner> = new Map();
  private _activeTaskIds: Set<string> = new Set();

  constructor(config: Partial<MultiRunnerConfig> = {}) {
    this._config = {
      maxConcurrentWorkspaces: 5,
      defaultHeartbeatConfig: {},
      defaultTaskMachineConfig: {},
      defaultRecoveryConfig: {},
      ...config
    };
  }

  get activeCount(): number {
    let count = 0;
    for (const runner of this._runners.values()) {
      if (runner.state.status === 'running') {
        count++;
      }
    }
    return count;
  }

  get totalWorkspaces(): number {
    return this._runners.size;
  }

  createRunner(workspaceId: string, workspaceType: 'pmcli' | 'devcli' | 'other'): WorkspaceRunner {
    if (this._runners.has(workspaceId)) {
      throw new Error(`Workspace runner already exists: ${workspaceId}`);
    }

    const runner = new WorkspaceRunner({
      workspaceId,
      workspaceType,
      heartbeatConfig: this._config.defaultHeartbeatConfig,
      taskMachineConfig: this._config.defaultTaskMachineConfig,
      recoveryConfig: this._config.defaultRecoveryConfig
    });

    this._runners.set(workspaceId, runner);
    return runner;
  }

  getRunner(workspaceId: string): WorkspaceRunner | undefined {
    return this._runners.get(workspaceId);
  }

  async startTask(workspaceId: string, taskId: string, page: Page): Promise<void> {
    const runner = this._runners.get(workspaceId);
    if (!runner) {
      throw new Error(`Workspace runner not found: ${workspaceId}`);
    }

    if (this._activeTaskIds.has(taskId)) {
      throw new Error(`Task already active: ${taskId}`);
    }

    await runner.start(taskId, page);
    this._activeTaskIds.add(taskId);
  }

  async stopTask(workspaceId: string): Promise<WorkspaceSummary> {
    const runner = this._runners.get(workspaceId);
    if (!runner) {
      throw new Error(`Workspace runner not found: ${workspaceId}`);
    }

    const taskId = runner.state.currentTaskId;
    const summary = await runner.stop();

    if (taskId) {
      this._activeTaskIds.delete(taskId);
    }

    return summary;
  }

  async stopAll(): Promise<Map<string, WorkspaceSummary>> {
    const summaries = new Map<string, WorkspaceSummary>();

    for (const [workspaceId, runner] of this._runners) {
      if (runner.state.status !== 'stopped') {
        const summary = await runner.stop();
        summaries.set(workspaceId, summary);
      }
    }

    this._activeTaskIds.clear();
    return summaries;
  }

  getHealthOverview(): HealthOverview {
    const overview: HealthOverview = {
      total: this._runners.size,
      healthy: 0,
      recovering: 0,
      stalled: 0,
      interrupted: 0,
      completed: 0,
      byWorkspace: new Map()
    };

    for (const [workspaceId, runner] of this._runners) {
      const mode = runner.state.healthMode;
      overview.byWorkspace.set(workspaceId, mode);

      switch (mode) {
        case 'healthy':
          overview.healthy++;
          break;
        case 'model-stalled':
        case 'terminal-hang':
        case 'modal-blocking':
          overview.recovering++;
          break;
        case 'task-interrupted':
          overview.interrupted++;
          break;
        case 'completed':
          overview.completed++;
          break;
      }
    }

    return overview;
  }

  getMetrics(): MultiRunnerMetrics {
    const metrics: MultiRunnerMetrics = {
      totalWorkspaces: this._runners.size,
      activeWorkspaces: this.activeCount,
      totalRecoveryAttempts: 0,
      avgHeartbeatLatency: 0,
      workspaceMetrics: new Map()
    };

    let totalLatency = 0;
    let latencyCount = 0;

    for (const [workspaceId, runner] of this._runners) {
      const runnerMetrics = {
        status: runner.state.status,
        healthMode: runner.state.healthMode,
        recoveryAttempts: runner.state.recoveryAttempts,
        uptime: runner.state.startedAt ? Date.now() - runner.state.startedAt : 0
      };

      metrics.workspaceMetrics.set(workspaceId, runnerMetrics);
      metrics.totalRecoveryAttempts += runner.state.recoveryAttempts;

      if (runner.state.lastHeartbeatAt) {
        totalLatency += Date.now() - runner.state.lastHeartbeatAt;
        latencyCount++;
      }
    }

    if (latencyCount > 0) {
      metrics.avgHeartbeatLatency = totalLatency / latencyCount;
    }

    return metrics;
  }
}

interface HealthOverview {
  total: number;
  healthy: number;
  recovering: number;
  stalled: number;
  interrupted: number;
  completed: number;
  byWorkspace: Map<string, HealthMode>;
}

interface MultiRunnerMetrics {
  totalWorkspaces: number;
  activeWorkspaces: number;
  totalRecoveryAttempts: number;
  avgHeartbeatLatency: number;
  workspaceMetrics: Map<string, any>;
}
```

## 9.3.4 独立 HeartbeatDaemon 实例管理

### 9.3.4.1 实例隔离策略

每个 WorkspaceRunner 的 HeartbeatDaemon 完全独立：

```typescript
class HeartbeatDaemonFactory {
  private static _instances: Map<string, HeartbeatDaemon> = new Map();

  static create(workspaceId: string, config: Partial<HeartbeatDaemonConfig>): HeartbeatDaemon {
    const daemon = new HeartbeatDaemon({
      ...config,
      workspaceId,
      persistencePath: `./heartbeat-${workspaceId}.jsonl`
    });

    this._instances.set(workspaceId, daemon);
    return daemon;
  }

  static get(workspaceId: string): HeartbeatDaemon | undefined {
    return this._instances.get(workspaceId);
  }

  static destroy(workspaceId: string): void {
    const daemon = this._instances.get(workspaceId);
    if (daemon) {
      daemon.stop();
      this._instances.delete(workspaceId);
    }
  }

  static destroyAll(): void {
    for (const [workspaceId] of this._instances) {
      this.destroy(workspaceId);
    }
  }
}
```

### 9.3.6.1 跨工作区事件

```typescript
type MultiRunnerEvent =
  | 'runner:created'
  | 'runner:started'
  | 'runner:stopped'
  | 'runner:error'
  | 'health:changed'
  | 'recovery:started'
  | 'recovery:completed';

class MultiRunnerEventBus extends EventEmitter {
  private static _instance: MultiRunnerEventBus;

  static getInstance(): MultiRunnerEventBus {
    if (!this._instance) {
      this._instance = new MultiRunnerEventBus();
    }
    return this._instance;
  }

  emitWorkspaceEvent(workspaceId: string, event: MultiRunnerEvent, data: any): void {
    this.emit(event, {
      workspaceId,
      ...data,
      timestamp: Date.now()
    });
  }

  subscribeToWorkspace(workspaceId: string, event: MultiRunnerEvent, handler: (data: any) => void): () => void {
    const filteredHandler = (data: any) => {
      if (data.workspaceId === workspaceId) {
        handler(data);
      }
    };
    
    this.on(event, filteredHandler);
    return () => this.off(event, filteredHandler);
  }
}
```

### 9.3.6.2 状态聚合

```typescript
class StateAggregator {
  private _states: Map<string, WorkspaceRunnerState> = new Map();

  update(workspaceId: string, state: WorkspaceRunnerState): void {
    this._states.set(workspaceId, state);
  }

  getAggregatedState(): AggregatedState {
    const aggregated: AggregatedState = {
      totalWorkspaces: this._states.size,
      runningCount: 0,
      pausedCount: 0,
      stoppedCount: 0,
      errorCount: 0,
      healthyCount: 0,
      recoveringCount: 0,
      workspaces: []
    };

    for (const [workspaceId, state] of this._states) {
      aggregated.workspaces.push({
        workspaceId,
        status: state.status,
        healthMode: state.healthMode,
        taskId: state.currentTaskId
      });

      switch (state.status) {
        case 'running':
          aggregated.runningCount++;
          break;
        case 'paused':
          aggregated.pausedCount++;
          break;
        case 'stopped':
          aggregated.stoppedCount++;
          break;
        case 'error':
          aggregated.errorCount++;
          break;
      }

      if (state.healthMode === 'healthy') {
        aggregated.healthyCount++;
      } else if (['model-stalled', 'terminal-hang', 'modal-blocking'].includes(state.healthMode)) {
        aggregated.recoveringCount++;
      }
    }

    return aggregated;
  }
}

interface AggregatedState {
  totalWorkspaces: number;
  runningCount: number;
  pausedCount: number;
  stoppedCount: number;
  errorCount: number;
  healthyCount: number;
  recoveringCount: number;
  workspaces: Array<{
    workspaceId: string;
    status: string;
    healthMode: HealthMode;
    taskId: string | null;
  }>;
}
```
