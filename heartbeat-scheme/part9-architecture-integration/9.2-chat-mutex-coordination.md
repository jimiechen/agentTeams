# 9.2 chat-mutex-machine.ts 并发协调

## 9.2.1 并发协调背景

chat-mutex-machine.ts 负责管理多任务并发执行时的互斥访问。在引入心跳机制后，需要：

1. 协调心跳检测与任务执行的并发关系
2. 管理 `pendingEvaluates` 计数，防止评估任务堆积
3. 确保恢复动作不会与正常任务执行冲突

## 9.2.2 pendingEvaluates 计数机制

### 9.2.2.1 问题定义

在多工作区场景下，可能同时存在多个待评估的任务。`pendingEvaluates` 计数用于：

- 追踪当前等待评估的任务数量
- 防止评估队列溢出
- 为心跳检测提供任务状态参考

### 9.2.2.2 数据结构定义

```typescript
interface PendingEvaluate {
  taskId: string;
  workspaceId: string;
  enqueuedAt: number;
  priority: 'high' | 'normal' | 'low';
  status: 'pending' | 'evaluating' | 'completed' | 'failed';
  heartbeatSessionId: string | null;
}

interface PendingEvaluatesState {
  count: number;
  items: Map<string, PendingEvaluate>;
  maxPending: number;
  oldestPendingAt: number | null;
}
```

### 9.2.2.3 计数管理器

```typescript
class PendingEvaluatesManager {
  private _state: PendingEvaluatesState;
  private _config: PendingEvaluatesConfig;

  constructor(config: Partial<PendingEvaluatesConfig> = {}) {
    this._config = {
      maxPending: 10,
      pendingTimeoutMs: 300000,
      cleanupIntervalMs: 60000,
      ...config
    };

    this._state = {
      count: 0,
      items: new Map(),
      maxPending: this._config.maxPending,
      oldestPendingAt: null
    };
  }

  get count(): number {
    return this._state.count;
  }

  get canEnqueue(): boolean {
    return this._state.count < this._config.maxPending;
  }

  enqueue(evaluate: Omit<PendingEvaluate, 'enqueuedAt' | 'status' | 'heartbeatSessionId'>): {
    success: boolean;
    reason: string;
    position: number;
  } {
    if (!this.canEnqueue) {
      return {
        success: false,
        reason: '评估队列已满',
        position: -1
      };
    }

    const id = this.generateId(evaluate.taskId, evaluate.workspaceId);
    const pendingEvaluate: PendingEvaluate = {
      ...evaluate,
      enqueuedAt: Date.now(),
      status: 'pending',
      heartbeatSessionId: null
    };

    this._state.items.set(id, pendingEvaluate);
    this._state.count++;
    this.updateOldestPending();

    return {
      success: true,
      reason: '已加入评估队列',
      position: this._state.count
    };
  }

  startEvaluating(taskId: string, workspaceId: string): boolean {
    const id = this.generateId(taskId, workspaceId);
    const item = this._state.items.get(id);

    if (!item || item.status !== 'pending') {
      return false;
    }

    item.status = 'evaluating';
    item.heartbeatSessionId = this.generateSessionId();
    return true;
  }

  complete(taskId: string, workspaceId: string, success: boolean): void {
    const id = this.generateId(taskId, workspaceId);
    const item = this._state.items.get(id);

    if (!item) return;

    item.status = success ? 'completed' : 'failed';
    
    setTimeout(() => {
      this._state.items.delete(id);
      this._state.count--;
      this.updateOldestPending();
    }, 1000);
  }

  getPendingByWorkspace(workspaceId: string): PendingEvaluate[] {
    const result: PendingEvaluate[] = [];
    for (const item of this._state.items.values()) {
      if (item.workspaceId === workspaceId) {
        result.push(item);
      }
    }
    return result;
  }

  getStats(): PendingEvaluatesStats {
    const now = Date.now();
    let oldestAge = 0;
    let pendingCount = 0;
    let evaluatingCount = 0;

    for (const item of this._state.items.values()) {
      if (item.status === 'pending') {
        pendingCount++;
        const age = now - item.enqueuedAt;
        if (age > oldestAge) {
          oldestAge = age;
        }
      } else if (item.status === 'evaluating') {
        evaluatingCount++;
      }
    }

    return {
      total: this._state.count,
      pending: pendingCount,
      evaluating: evaluatingCount,
      oldestAgeMs: oldestAge,
      maxPending: this._config.maxPending
    };
  }

  private updateOldestPending(): void {
    let oldest: number | null = null;
    for (const item of this._state.items.values()) {
      if (item.status === 'pending') {
        if (oldest === null || item.enqueuedAt < oldest) {
          oldest = item.enqueuedAt;
        }
      }
    }
    this._state.oldestPendingAt = oldest;
  }

  private generateId(taskId: string, workspaceId: string): string {
    return `${workspaceId}:${taskId}`;
  }

  private generateSessionId(): string {
    return `eval-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }
}

interface PendingEvaluatesStats {
  total: number;
  pending: number;
  evaluating: number;
  oldestAgeMs: number;
  maxPending: number;
}
```

## 9.2.3 ChatMutex 状态机改造

### 9.2.3.1 状态定义

```typescript
type ChatMutexState = 
  | 'idle'
  | 'acquired'
  | 'acquired:evaluating'
  | 'acquired:recovering'
  | 'released';

interface ChatMutexContext {
  workspaceId: string;
  taskId: string | null;
  acquiredAt: number | null;
  pendingEvaluates: PendingEvaluatesManager;
  heartbeatEnabled: boolean;
}
```

### 9.2.3.2 状态转移矩阵

| 当前状态 | 事件 | 目标状态 | 条件 | 动作 |
|---------|------|---------|------|------|
| `idle` | `ACQUIRE` | `acquired` | 无待评估任务 | 获取锁，启动心跳 |
| `idle` | `ACQUIRE` | `acquired:evaluating` | 有待评估任务 | 获取锁，开始评估 |
| `acquired` | `START_EVALUATE` | `acquired:evaluating` | - | 开始评估任务 |
| `acquired:evaluating` | `EVALUATE_COMPLETE` | `acquired` | - | 完成评估 |
| `acquired` | `RECOVERY_NEEDED` | `acquired:recovering` | - | 执行恢复动作 |
| `acquired:recovering` | `RECOVERY_COMPLETE` | `acquired` | - | 恢复完成 |
| `acquired:*` | `RELEASE` | `released` | - | 释放锁，停止心跳 |
| `released` | `RESET` | `idle` | - | 重置状态 |

### 9.2.3.3 ChatMutex 机器实现

```typescript
interface ChatMutexConfig {
  maxPendingEvaluates: number;
  acquireTimeoutMs: number;
  heartbeatEnabled: boolean;
}

class ChatMutexMachine {
  private _state: ChatMutexState = 'idle';
  private _context: ChatMutexContext;
  private _config: ChatMutexConfig;
  private _heartbeatDaemon: HeartbeatDaemon | null = null;

  constructor(workspaceId: string, config: Partial<ChatMutexConfig> = {}) {
    this._config = {
      maxPendingEvaluates: 10,
      acquireTimeoutMs: 30000,
      heartbeatEnabled: true,
      ...config
    };

    this._context = {
      workspaceId,
      taskId: null,
      acquiredAt: null,
      pendingEvaluates: new PendingEvaluatesManager({
        maxPending: this._config.maxPendingEvaluates
      }),
      heartbeatEnabled: this._config.heartbeatEnabled
    };
  }

  get state(): ChatMutexState {
    return this._state;
  }

  get pendingEvaluates(): PendingEvaluatesManager {
    return this._context.pendingEvaluates;
  }

  async acquire(taskId: string, page: Page): Promise<{
    success: boolean;
    reason: string;
  }> {
    if (this._state !== 'idle') {
      return {
        success: false,
        reason: `Mutex already in state: ${this._state}`
      };
    }

    this._context.taskId = taskId;
    this._context.acquiredAt = Date.now();

    const pendingStats = this._context.pendingEvaluates.getStats();
    
    if (pendingStats.pending > 0) {
      this._state = 'acquired:evaluating';
      this.startEvaluation(taskId);
    } else {
      this._state = 'acquired';
    }

    if (this._context.heartbeatEnabled) {
      await this.startHeartbeat(taskId, page);
    }

    return {
      success: true,
      reason: 'Mutex acquired successfully'
    };
  }

  async release(): Promise<void> {
    if (this._state === 'idle' || this._state === 'released') {
      return;
    }

    await this.stopHeartbeat();
    this._state = 'released';
    this._context.taskId = null;
    this._context.acquiredAt = null;
  }

  reset(): void {
    this._state = 'idle';
  }

  private startEvaluation(taskId: string): void {
    this._context.pendingEvaluates.startEvaluating(
      taskId,
      this._context.workspaceId
    );
  }

  completeEvaluation(success: boolean): void {
    if (this._state !== 'acquired:evaluating') {
      return;
    }

    this._context.pendingEvaluates.complete(
      this._context.taskId!,
      this._context.workspaceId,
      success
    );

    this._state = 'acquired';
  }

  enterRecovery(): void {
    if (this._state === 'acquired' || this._state === 'acquired:evaluating') {
      this._state = 'acquired:recovering';
    }
  }

  exitRecovery(success: boolean): void {
    if (this._state !== 'acquired:recovering') {
      return;
    }

    this._state = success ? 'acquired' : 'acquired';
  }

  private async startHeartbeat(taskId: string, page: Page): Promise<void> {
    this._heartbeatDaemon = new HeartbeatDaemon({
      taskId,
      workspaceId: this._context.workspaceId,
      onHealthChange: (mode) => this.handleHealthChange(mode),
      onRecoveryNeeded: (action) => this.handleRecoveryNeeded(action)
    });

    await this._heartbeatDaemon.start(taskId, page);
  }

  private async stopHeartbeat(): Promise<void> {
    if (this._heartbeatDaemon) {
      await this._heartbeatDaemon.stop();
      this._heartbeatDaemon = null;
    }
  }

  private handleHealthChange(mode: HealthMode): void {
    switch (mode) {
      case 'task-interrupted':
        this.release();
        break;
      case 'completed':
        this.release();
        break;
    }
  }

  private async handleRecoveryNeeded(action: RecoveryAction): Promise<void> {
    this.enterRecovery();
    
    try {
      await this._heartbeatDaemon?.executeRecovery(action);
      this.exitRecovery(true);
    } catch (error) {
      this.exitRecovery(false);
    }
  }
}
```

## 9.2.4 多工作区并发协调

### 9.2.4.1 协调器架构

```typescript
interface WorkspaceMutexEntry {
  workspaceId: string;
  mutex: ChatMutexMachine;
  lastActivity: number;
  isActive: boolean;
}

class ChatMutexCoordinator {
  private _workspaces: Map<string, WorkspaceMutexEntry> = new Map();
  private _config: CoordinatorConfig;

  constructor(config: Partial<CoordinatorConfig> = {}) {
    this._config = {
      maxConcurrentWorkspaces: 5,
      inactiveTimeoutMs: 600000,
      ...config
    };
  }

  getWorkspace(workspaceId: string): ChatMutexMachine {
    let entry = this._workspaces.get(workspaceId);
    
    if (!entry) {
      entry = {
        workspaceId,
        mutex: new ChatMutexMachine(workspaceId),
        lastActivity: Date.now(),
        isActive: false
      };
      this._workspaces.set(workspaceId, entry);
    }

    entry.lastActivity = Date.now();
    return entry.mutex;
  }

  getActiveWorkspaces(): string[] {
    const active: string[] = [];
    for (const [id, entry] of this._workspaces) {
      if (entry.isActive) {
        active.push(id);
      }
    }
    return active;
  }

  getTotalPendingEvaluates(): number {
    let total = 0;
    for (const entry of this._workspaces.values()) {
      total += entry.mutex.pendingEvaluates.count;
    }
    return total;
  }

  getCoordinatorStats(): CoordinatorStats {
    const workspaceStats: Map<string, PendingEvaluatesStats> = new Map();
    let totalPending = 0;
    let totalEvaluating = 0;

    for (const [id, entry] of this._workspaces) {
      const stats = entry.mutex.pendingEvaluates.getStats();
      workspaceStats.set(id, stats);
      totalPending += stats.pending;
      totalEvaluating += stats.evaluating;
    }

    return {
      totalWorkspaces: this._workspaces.size,
      activeWorkspaces: this.getActiveWorkspaces().length,
      totalPendingEvaluates: totalPending,
      totalEvaluating: totalEvaluating,
      workspaceStats
    };
  }

  cleanupInactive(): void {
    const now = Date.now();
    const threshold = now - this._config.inactiveTimeoutMs;

    for (const [id, entry] of this._workspaces) {
      if (!entry.isActive && entry.lastActivity < threshold) {
        this._workspaces.delete(id);
      }
    }
  }
}

interface CoordinatorConfig {
  maxConcurrentWorkspaces: number;
  inactiveTimeoutMs: number;
}

interface CoordinatorStats {
  totalWorkspaces: number;
  activeWorkspaces: number;
  totalPendingEvaluates: number;
  totalEvaluating: number;
  workspaceStats: Map<string, PendingEvaluatesStats>;
}
```

## 9.2.5 心跳与评估的并发控制

### 9.2.5.1 并发规则

| 场景 | 心跳状态 | 评估状态 | 处理策略 |
|------|---------|---------|---------|
| 正常执行 | 运行中 | 无 | 继续心跳监控 |
| 等待评估 | 暂停 | 待处理 | 评估完成后恢复心跳 |
| 评估中 | 暂停 | 进行中 | 等待评估完成 |
| 恢复中 | 运行中 | 暂停 | 优先完成恢复 |
| 恢复完成 | 运行中 | 恢复 | 恢复评估流程 |

### 9.2.5.2 并发控制器

```typescript
class ConcurrencyController {
  private _heartbeatPaused: boolean = false;
  private _evaluationPaused: boolean = false;
  private _recoveryInProgress: boolean = false;

  canRunHeartbeat(): boolean {
    return !this._heartbeatPaused && !this._recoveryInProgress;
  }

  canRunEvaluation(): boolean {
    return !this._evaluationPaused && !this._recoveryInProgress;
  }

  pauseHeartbeat(): void {
    this._heartbeatPaused = true;
  }

  resumeHeartbeat(): void {
    this._heartbeatPaused = false;
  }

  pauseEvaluation(): void {
    this._evaluationPaused = true;
  }

  resumeEvaluation(): void {
    this._evaluationPaused = false;
  }

  startRecovery(): void {
    this._recoveryInProgress = true;
    this.pauseEvaluation();
  }

  endRecovery(): void {
    this._recoveryInProgress = false;
    this.resumeEvaluation();
  }

  getState(): ConcurrencyState {
    return {
      heartbeatPaused: this._heartbeatPaused,
      evaluationPaused: this._evaluationPaused,
      recoveryInProgress: this._recoveryInProgress
    };
  }
}

interface ConcurrencyState {
  heartbeatPaused: boolean;
  evaluationPaused: boolean;
  recoveryInProgress: boolean;
}
```

## 9.2.9 监控指标

```typescript
interface ChatMutexMetrics {
  totalAcquisitions: number;
  totalReleases: number;
  avgHoldTime: number;
  pendingEvaluatesCount: number;
  evaluatingCount: number;
  recoveryCount: number;
  overflowCount: number;
}

class ChatMutexMetricsCollector {
  private _metrics: ChatMutexMetrics = {
    totalAcquisitions: 0,
    totalReleases: 0,
    avgHoldTime: 0,
    pendingEvaluatesCount: 0,
    evaluatingCount: 0,
    recoveryCount: 0,
    overflowCount: 0
  };

  recordAcquisition(): void {
    this._metrics.totalAcquisitions++;
  }

  recordRelease(holdTime: number): void {
    this._metrics.totalReleases++;
    this.updateAvgHoldTime(holdTime);
  }

  recordOverflow(): void {
    this._metrics.overflowCount++;
  }

  private updateAvgHoldTime(holdTime: number): void {
    const n = this._metrics.totalReleases;
    this._metrics.avgHoldTime = 
      (this._metrics.avgHoldTime * (n - 1) + holdTime) / n;
  }

  getMetrics(): ChatMutexMetrics {
    return { ...this._metrics };
  }
}
```
