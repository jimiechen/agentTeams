# 9.1 task-machine.ts 改造点

## 9.1.1 改造背景

task-machine.ts 是任务执行的核心状态机。为支持心跳机制，需对其进行并行状态改造，使其能够：

1. 在任务执行期间并行运行心跳检测
2. 根据心跳检测结果触发状态转移
3. 支持恢复动作后的状态验证

## 9.1.3 并行状态改造方案

### 9.1.3.1 新增并行状态维度

```typescript
type TaskState = 
  | 'idle'
  | 'running'
  | 'running:healthy'
  | 'running:recovering'
  | 'running:verifying'
  | 'paused'
  | 'completed'
  | 'failed'
  | 'interrupted';

type HealthSubstate = 
  | 'healthy'
  | 'recovering'
  | 'verifying'
  | 'none';

interface TaskMachineState {
  primary: TaskState;
  health: HealthSubstate;
  heartbeatSessionId: string | null;
  recoveryAttempts: number;
  lastHealthCheck: number | null;
}
```

### 9.1.3.2 并行状态转移矩阵

| 当前状态 | 心跳事件 | 目标状态 | 动作 |
|---------|---------|---------|------|
| `running` | `health:healthy` | `running:healthy` | 更新健康时间戳 |
| `running:healthy` | `health:model-stalled` | `running:recovering` | 触发恢复动作 |
| `running:healthy` | `health:terminal-hang` | `running:recovering` | 点击后台运行 |
| `running:healthy` | `health:modal-blocking` | `running:recovering` | 处理弹窗 |
| `running:recovering` | `recovery:started` | `running:verifying` | 启动验证窗口 |
| `running:verifying` | `verify:success` | `running:healthy` | 重置恢复计数 |
| `running:verifying` | `verify:failed` | `running:recovering` | 增加恢复计数 |
| `running:*` | `health:task-interrupted` | `interrupted` | 停止心跳，发送告警 |
| `running:*` | `health:completed` | `completed` | 停止心跳，记录结果 |

### 9.1.3.3 状态机改造代码

```typescript
interface TaskMachineConfig {
  maxRecoveryAttempts: number;
  verificationWindowMs: number;
  healthCheckIntervalMs: number;
}

class TaskMachine {
  private _state: TaskMachineState;
  private _config: TaskMachineConfig;
  private _heartbeatDaemon: HeartbeatDaemon | null = null;
  private _stateHistory: StateTransition[] = [];

  constructor(config: Partial<TaskMachineConfig> = {}) {
    this._config = {
      maxRecoveryAttempts: 3,
      verificationWindowMs: 30000,
      healthCheckIntervalMs: 5000,
      ...config
    };

    this._state = {
      primary: 'idle',
      health: 'none',
      heartbeatSessionId: null,
      recoveryAttempts: 0,
      lastHealthCheck: null
    };
  }

  get state(): Readonly<TaskMachineState> {
    return { ...this._state };
  }

  async start(taskId: string, page: Page): Promise<void> {
    if (this._state.primary !== 'idle') {
      throw new Error(`Cannot start task in state: ${this._state.primary}`);
    }

    const sessionId = this.generateSessionId();
    this._state.heartbeatSessionId = sessionId;

    this._heartbeatDaemon = new HeartbeatDaemon({
      taskId,
      sessionId,
      onHealthChange: (mode) => this.handleHealthChange(mode),
      onRecoveryNeeded: (action) => this.handleRecoveryNeeded(action)
    });

    await this._heartbeatDaemon.start(taskId, page);

    this.transitionTo('running', 'healthy');
    
    this.recordTransition('start', {
      taskId,
      sessionId
    });
  }

  private handleHealthChange(mode: HealthMode): void {
    this._state.lastHealthCheck = Date.now();

    switch (mode) {
      case 'healthy':
        if (this._state.health === 'verifying') {
          this.transitionTo('running', 'healthy');
          this._state.recoveryAttempts = 0;
        }
        break;

      case 'model-stalled':
      case 'terminal-hang':
      case 'modal-blocking':
        if (this._state.health === 'healthy' || this._state.health === 'none') {
          this.transitionTo('running', 'recovering');
        }
        break;

      case 'task-interrupted':
        this.transitionTo('interrupted', 'none');
        this.stopHeartbeat();
        this.emitAlert('critical', '任务被中断');
        break;

      case 'completed':
        this.transitionTo('completed', 'none');
        this.stopHeartbeat();
        break;

      case 'model-queuing':
        break;
    }
  }

  private async handleRecoveryNeeded(action: RecoveryAction): Promise<void> {
    if (this._state.recoveryAttempts >= this._config.maxRecoveryAttempts) {
      this.transitionTo('failed', 'none');
      this.stopHeartbeat();
      this.emitAlert('critical', '恢复尝试次数已达上限');
      return;
    }

    this._state.recoveryAttempts++;
    this.transitionTo('running', 'verifying');

    await this.executeRecovery(action);
  }

  private async executeRecovery(action: RecoveryAction): Promise<void> {
    try {
      await this._heartbeatDaemon?.executeRecovery(action);
    } catch (error) {
      console.error('[TaskMachine] 恢复执行失败:', error);
    }
  }

  private transitionTo(primary: TaskState, health: HealthSubstate): void {
    const previousState = { ...this._state };
    
    this._state.primary = primary;
    this._state.health = health;

    this.recordTransition('state_change', {
      from: previousState,
      to: { primary, health }
    });

    this.emit('state_changed', {
      previous: previousState,
      current: { primary, health }
    });
  }

  private stopHeartbeat(): void {
    if (this._heartbeatDaemon) {
      this._heartbeatDaemon.stop();
      this._heartbeatDaemon = null;
    }
  }

  private generateSessionId(): string {
    return `session-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  private recordTransition(type: string, data: any): void {
    this._stateHistory.push({
      timestamp: Date.now(),
      type,
      state: { ...this._state },
      data
    });
  }
}
```

## 9.1.4 心跳事件订阅

### 9.1.4.1 事件类型定义

```typescript
type TaskMachineEvent = 
  | 'state_changed'
  | 'recovery_started'
  | 'recovery_completed'
  | 'recovery_failed'
  | 'health_degraded'
  | 'health_restored'
  | 'task_interrupted'
  | 'task_completed';

interface TaskMachineEventData {
  timestamp: number;
  sessionId: string;
  taskId: string;
  [key: string]: any;
}
```

### 9.1.4.2 事件订阅接口

```typescript
class TaskMachine extends EventEmitter {
  subscribe(event: TaskMachineEvent, handler: (data: TaskMachineEventData) => void): () => void {
    this.on(event, handler);
    return () => this.off(event, handler);
  }

  onHealthDegraded(handler: (data: { mode: HealthMode; duration: number }) => void): () => void {
    return this.subscribe('health_degraded', (data) => {
      handler({
        mode: data.mode,
        duration: data.duration || 0
      });
    });
  }

  onRecoveryCompleted(handler: (data: { action: RecoveryAction; attempts: number }) => void): () => void {
    return this.subscribe('recovery_completed', (data) => {
      handler({
        action: data.action,
        attempts: data.attempts || 0
      });
    });
  }
}
```

```typescript
class TaskMachine {
  async persistState(): Promise<void> {
    const state: TaskMachinePersistedState = {
      version: '1.0.0',
      taskId: this._taskId,
      sessionId: this._state.heartbeatSessionId || '',
      primary: this._state.primary,
      health: this._state.health,
      recoveryAttempts: this._state.recoveryAttempts,
      startedAt: this._startedAt,
      lastHealthCheck: this._state.lastHealthCheck,
      stateHistory: this._stateHistory
    };

    await this._persistenceManager.saveTaskState(state);
  }

  async restoreState(state: TaskMachinePersistedState, page: Page): Promise<void> {
    this._taskId = state.taskId;
    this._state.heartbeatSessionId = state.sessionId;
    this._state.primary = state.primary;
    this._state.health = state.health;
    this._state.recoveryAttempts = state.recoveryAttempts;
    this._startedAt = state.startedAt;
    this._state.lastHealthCheck = state.lastHealthCheck;
    this._stateHistory = state.stateHistory;

    if (state.primary === 'running' || state.primary.startsWith('running:')) {
      await this.restartHeartbeat(state.taskId, page);
    }
  }
}
```

## 9.1.8 测试用例

```typescript
describe('TaskMachine Parallel State', () => {
  it('应在健康状态变化时正确转移状态', async () => {
    const machine = new TaskMachine();
    const page = createMockPage();
    
    await machine.start('task-123', page);
    expect(machine.state.primary).toBe('running');
    expect(machine.state.health).toBe('healthy');
    
    machine.simulateHealthChange('model-stalled');
    expect(machine.state.health).toBe('recovering');
  });

  it('应在恢复成功后重置恢复计数', async () => {
    const machine = new TaskMachine();
    const page = createMockPage();
    
    await machine.start('task-123', page);
    machine.simulateHealthChange('model-stalled');
    machine.simulateRecoveryCompleted(true);
    
    expect(machine.state.health).toBe('healthy');
    expect(machine.state.recoveryAttempts).toBe(0);
  });

  it('应在达到最大恢复次数后进入失败状态', async () => {
    const machine = new TaskMachine({ maxRecoveryAttempts: 2 });
    const page = createMockPage();
    
    await machine.start('task-123', page);
    
    for (let i = 0; i < 3; i++) {
      machine.simulateHealthChange('model-stalled');
      machine.simulateRecoveryCompleted(false);
    }
    
    expect(machine.state.primary).toBe('failed');
  });

  it('应在任务中断时停止心跳', async () => {
    const machine = new TaskMachine();
    const page = createMockPage();
    
    await machine.start('task-123', page);
    machine.simulateHealthChange('task-interrupted');
    
    expect(machine.state.primary).toBe('interrupted');
    expect(machine.state.heartbeatSessionId).toBeNull();
  });
});
```

## 9.1.9 监控指标

```typescript
interface TaskMachineMetrics {
  totalTasks: number;
  activeTasks: number;
  completedTasks: number;
  failedTasks: number;
  interruptedTasks: number;
  totalRecoveryAttempts: number;
  successfulRecoveries: number;
  avgRecoveryTime: number;
  stateDistribution: Record<TaskState, number>;
}

class TaskMachineMonitor {
  private metrics: TaskMachineMetrics = {
    totalTasks: 0,
    activeTasks: 0,
    completedTasks: 0,
    failedTasks: 0,
    interruptedTasks: 0,
    totalRecoveryAttempts: 0,
    successfulRecoveries: 0,
    avgRecoveryTime: 0,
    stateDistribution: {} as Record<TaskState, number>
  };

  recordStateChange(from: TaskState, to: TaskState): void {
    this.metrics.stateDistribution[from] = (this.metrics.stateDistribution[from] || 0) - 1;
    this.metrics.stateDistribution[to] = (this.metrics.stateDistribution[to] || 0) + 1;
  }

  recordRecovery(success: boolean, duration: number): void {
    this.metrics.totalRecoveryAttempts++;
    if (success) {
      this.metrics.successfulRecoveries++;
      this.updateAvgRecoveryTime(duration);
    }
  }

  private updateAvgRecoveryTime(duration: number): void {
    const n = this.metrics.successfulRecoveries;
    this.metrics.avgRecoveryTime = 
      (this.metrics.avgRecoveryTime * (n - 1) + duration) / n;
  }
}
```
