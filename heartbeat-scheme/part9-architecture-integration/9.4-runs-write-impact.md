# 9.4 runs/*.md 写入逻辑影响

## 9.4.1 写入逻辑概述

runs/*.md 文件记录任务执行的详细过程。在心跳架构引入后，需要在写入逻辑中：

1. 嵌入心跳元信息
2. 记录三终态（completed/interrupted/failed）的写入时机
3. 保存恢复动作历史

## 9.4.2 三终态写入机制

### 9.4.2.1 终态定义

| 终态 | 触发条件 | 写入时机 | 写入内容 |
|------|---------|---------|---------|
| `completed` | 任务正常完成 | 任务结束时 | 完成摘要、心跳统计 |
| `interrupted` | 任务被中断 | 检测到中断状态 | 中断原因、最后心跳状态 |
| `failed` | 恢复次数耗尽 | 恢复失败后 | 失败原因、恢复历史 |

### 9.4.2.2 终态写入器

```typescript
type TerminalState = 'completed' | 'interrupted' | 'failed';

interface TerminalStateWriter {
  state: TerminalState;
  taskId: string;
  workspaceId: string;
  timestamp: number;
  reason: string;
  heartbeatSummary: HeartbeatSummary;
  recoveryHistory: RecoveryRecord[];
  metrics: TaskMetrics;
}

interface HeartbeatSummary {
  sessionId: string;
  startedAt: number;
  endedAt: number;
  totalDuration: number;
  layer1Collects: number;
  layer2Collects: number;
  layer3Collects: number;
  healthTransitions: HealthTransition[];
  recoveryCount: number;
  finalHealthMode: HealthMode;
}

interface RecoveryRecord {
  timestamp: number;
  action: RecoveryAction;
  result: 'success' | 'failed';
  durationMs: number;
  error?: string;
}

interface TaskMetrics {
  totalChars: number;
  avgCharsPerSecond: number;
  stallCount: number;
  terminalHangCount: number;
  modalBlockingCount: number;
}

class TerminalStateWriterImpl {
  private _basePath: string;

  constructor(basePath: string = './runs') {
    this._basePath = basePath;
  }

  async write(state: TerminalState, data: TerminalStateWriter): Promise<string> {
    const filePath = this.getFilePath(data.taskId, data.workspaceId);
    const content = this.formatContent(state, data);

    await this.ensureDirectory(filePath);
    await fs.writeFile(filePath, content, 'utf-8');

    return filePath;
  }

  private getFilePath(taskId: string, workspaceId: string): string {
    const date = new Date().toISOString().split('T')[0];
    return `${this._basePath}/${date}/${workspaceId}/${taskId}.md`;
  }

  private formatContent(state: TerminalState, data: TerminalStateWriter): string {
    const lines: string[] = [];

    lines.push(`# Task: ${data.taskId}`);
    lines.push('');
    lines.push(`## 终态: ${state}`);
    lines.push('');
    lines.push(`- **工作区**: ${data.workspaceId}`);
    lines.push(`- **时间戳**: ${new Date(data.timestamp).toISOString()}`);
    lines.push(`- **原因**: ${data.reason}`);
    lines.push('');

    lines.push('## 心跳摘要');
    lines.push('');
    lines.push(`- **会话ID**: ${data.heartbeatSummary.sessionId}`);
    lines.push(`- **开始时间**: ${new Date(data.heartbeatSummary.startedAt).toISOString()}`);
    lines.push(`- **结束时间**: ${new Date(data.heartbeatSummary.endedAt).toISOString()}`);
    lines.push(`- **总时长**: ${this.formatDuration(data.heartbeatSummary.totalDuration)}`);
    lines.push(`- **Layer 1 采集次数**: ${data.heartbeatSummary.layer1Collects}`);
    lines.push(`- **Layer 2 采集次数**: ${data.heartbeatSummary.layer2Collects}`);
    lines.push(`- **Layer 3 采集次数**: ${data.heartbeatSummary.layer3Collects}`);
    lines.push(`- **恢复次数**: ${data.heartbeatSummary.recoveryCount}`);
    lines.push(`- **最终健康状态**: ${data.heartbeatSummary.finalHealthMode}`);
    lines.push('');

    if (data.heartbeatSummary.healthTransitions.length > 0) {
      lines.push('## 健康状态转移历史');
      lines.push('');
      lines.push('| 时间 | 从 | 到 | 原因 |');
      lines.push('|------|----|----|------|');
      for (const transition of data.heartbeatSummary.healthTransitions) {
        lines.push(`| ${new Date(transition.timestamp).toISOString()} | ${transition.from} | ${transition.to} | ${transition.reason} |`);
      }
      lines.push('');
    }

    if (data.recoveryHistory.length > 0) {
      lines.push('## 恢复动作历史');
      lines.push('');
      lines.push('| 时间 | 动作类型 | 结果 | 耗时 |');
      lines.push('|------|---------|------|------|');
      for (const record of data.recoveryHistory) {
        lines.push(`| ${new Date(record.timestamp).toISOString()} | ${record.action.type} | ${record.result} | ${record.durationMs}ms |`);
      }
      lines.push('');
    }

    lines.push('## 任务指标');
    lines.push('');
    lines.push(`- **总字符数**: ${data.metrics.totalChars}`);
    lines.push(`- **平均字符/秒**: ${data.metrics.avgCharsPerSecond.toFixed(2)}`);
    lines.push(`- **停滞次数**: ${data.metrics.stallCount}`);
    lines.push(`- **终端挂起次数**: ${data.metrics.terminalHangCount}`);
    lines.push(`- **模态阻塞次数**: ${data.metrics.modalBlockingCount}`);

    return lines.join('\n');
  }

  private formatDuration(ms: number): string {
    const seconds = Math.floor(ms / 1000);
    const minutes = Math.floor(seconds / 60);
    const hours = Math.floor(minutes / 60);

    if (hours > 0) {
      return `${hours}h ${minutes % 60}m ${seconds % 60}s`;
    } else if (minutes > 0) {
      return `${minutes}m ${seconds % 60}s`;
    } else {
      return `${seconds}s`;
    }
  }

  private async ensureDirectory(filePath: string): Promise<void> {
    const dir = path.dirname(filePath);
    await fs.mkdir(dir, { recursive: true });
  }
}
```

```typescript
    if (!this._metadata) return '';

    const lines: string[] = [];
    lines.push('---');
    lines.push(`heartbeat_version: "${this._metadata.version}"`);
    lines.push(`heartbeat_session_id: "${this._metadata.sessionId}"`);
    lines.push(`heartbeat_workspace_id: "${this._metadata.workspaceId}"`);
    lines.push(`heartbeat_task_id: "${this._metadata.taskId}"`);
    lines.push(`heartbeat_started_at: ${this._metadata.startedAt}`);
    lines.push(`heartbeat_last_updated: ${this._metadata.lastUpdatedAt}`);
    lines.push(`heartbeat_current_mode: "${this._metadata.currentHealthMode}"`);
    lines.push('---');

    return lines.join('\n');
  }

  formatAsJson(): string {
    return JSON.stringify(this._metadata, null, 2);
  }
```

```typescript
type WriteTrigger = 
  | 'task_completed'
  | 'task_interrupted'
  | 'task_failed'
  | 'health_changed'

interface WriteTriggerHandler {
  trigger: WriteTrigger;

  priority: 'high' | 'medium' | 'low';

const WRITE_TRIGGERS: WriteTriggerHandler[] = [

    condition: (state) => state.primary === 'completed',

  {

    priority: 'high',
```

### 9.4.4.2 写入调度器

```typescript
class WriteScheduler {
  private _queue: WriteRequest[] = [];
  private _processing: boolean = false;
  private _debounceTimers: Map<string, ReturnType<typeof setTimeout>> = new Map();

  schedule(request: WriteRequest): void {
    const trigger = WRITE_TRIGGERS.find(t => t.trigger === request.trigger);
    if (!trigger) return;

    if (trigger.debounceMs > 0) {
      this.scheduleWithDebounce(request, trigger);
    } else {
      this.enqueue(request, trigger.priority);
    }
  }

  private scheduleWithDebounce(request: WriteRequest, trigger: WriteTriggerHandler): void {
    const key = `${request.taskId}:${request.trigger}`;
    
    const existingTimer = this._debounceTimers.get(key);
    if (existingTimer) {
      clearTimeout(existingTimer);
    }

    const timer = setTimeout(() => {
      this._debounceTimers.delete(key);
      this.enqueue(request, trigger.priority);
    }, trigger.debounceMs);

    this._debounceTimers.set(key, timer);
  }

  private enqueue(request: WriteRequest, priority: 'high' | 'medium' | 'low'): void {
    const priorityOrder = { high: 0, medium: 1, low: 2 };
    
    this._queue.push(request);
    this._queue.sort((a, b) => {
      const priorityA = priorityOrder[WRITE_TRIGGERS.find(t => t.trigger === a.trigger)?.priority || 'low'];
      const priorityB = priorityOrder[WRITE_TRIGGERS.find(t => t.trigger === b.trigger)?.priority || 'low'];
      return priorityA - priorityB;
    });

    this.processQueue();
  }

  private async processQueue(): Promise<void> {
    if (this._processing || this._queue.length === 0) return;

    this._processing = true;

    while (this._queue.length > 0) {
      const request = this._queue.shift()!;
      await this.executeWrite(request);
    }

    this._processing = false;
  }

  private async executeWrite(request: WriteRequest): Promise<void> {
    try {
      await request.writer.write(request.state, request.data);
    } catch (error) {
      console.error('[WriteScheduler] Write failed:', error);
    }
  }
}

interface WriteRequest {
  trigger: WriteTrigger;
  taskId: string;
  state: TerminalState;
  data: TerminalStateWriter;
  writer: TerminalStateWriterImpl;
}
```

## 9.4.5 恢复历史记录

### 9.4.5.1 恢复记录结构

```typescript
interface RecoveryHistoryEntry {
  id: string;
  timestamp: number;
  taskId: string;
  workspaceId: string;
  trigger: HealthMode;
  action: RecoveryAction;
  execution: {
    startedAt: number;
    completedAt: number;
    durationMs: number;
    result: 'success' | 'failed' | 'skipped';
    error?: string;
  };
  verification: {
    windowStartedAt: number;
    windowEndedAt: number;
    verifiedAt: number | null;
    success: boolean;
  };
  context: {
    healthModeBefore: HealthMode;
    healthModeAfter: HealthMode;
    recoveryAttemptNumber: number;
    totalRecoveryAttempts: number;
  };
}

class RecoveryHistoryManager {
  private _history: Map<string, RecoveryHistoryEntry[]> = new Map();

  record(taskId: string, entry: RecoveryHistoryEntry): void {
    if (!this._history.has(taskId)) {
      this._history.set(taskId, []);
    }
    this._history.get(taskId)!.push(entry);
  }

  getHistory(taskId: string): RecoveryHistoryEntry[] {
    return this._history.get(taskId) || [];
  }

  getLatest(taskId: string): RecoveryHistoryEntry | undefined {
    const history = this._history.get(taskId);
    return history?.[history.length - 1];
  }

  getStats(taskId: string): RecoveryStats {
    const history = this.getHistory(taskId);
    
    return {
      totalAttempts: history.length,
      successfulAttempts: history.filter(e => e.execution.result === 'success').length,
      failedAttempts: history.filter(e => e.execution.result === 'failed').length,
      avgDurationMs: this.calculateAvgDuration(history),
      byTrigger: this.groupByTrigger(history)
    };
  }

  private calculateAvgDuration(history: RecoveryHistoryEntry[]): number {
    if (history.length === 0) return 0;
    const total = history.reduce((sum, e) => sum + e.execution.durationMs, 0);
    return total / history.length;
  }

  private groupByTrigger(history: RecoveryHistoryEntry[]): Record<HealthMode, number> {
    const result: Record<string, number> = {};
    for (const entry of history) {
      result[entry.trigger] = (result[entry.trigger] || 0) + 1;
    }
    return result as Record<HealthMode, number>;
  }
}

interface RecoveryStats {
  totalAttempts: number;
  successfulAttempts: number;
  failedAttempts: number;
  avgDurationMs: number;
  byTrigger: Record<HealthMode, number>;
}
```

```typescript
class RecoveryHistoryFormatter {
  formatAsMarkdown(history: RecoveryHistoryEntry[]): string {
    const lines: string[] = [];

    lines.push('## 恢复历史');
    lines.push('');

    if (history.length === 0) {
      lines.push('无恢复记录。');
      return lines.join('\n');
    }

    lines.push('| # | 时间 | 触发原因 | 动作类型 | 结果 | 耗时 |');
    lines.push('|---|------|---------|---------|------|------|');

    for (let i = 0; i < history.length; i++) {
      const entry = history[i];
      lines.push(`| ${i + 1} | ${new Date(entry.timestamp).toISOString()} | ${entry.trigger} | ${entry.action.type} | ${entry.execution.result} | ${entry.execution.durationMs}ms |`);
    }

    lines.push('');

    for (const entry of history) {
      lines.push(`### 恢复 #${history.indexOf(entry) + 1}`);
      lines.push('');
      lines.push(`- **ID**: ${entry.id}`);
      lines.push(`- **触发原因**: ${entry.trigger}`);
      lines.push(`- **动作类型**: ${entry.action.type}`);
      lines.push(`- **执行结果**: ${entry.execution.result}`);
      lines.push(`- **执行耗时**: ${entry.execution.durationMs}ms`);
      lines.push(`- **验证结果**: ${entry.verification.success ? '成功' : '失败'}`);
      lines.push(`- **状态变化**: ${entry.context.healthModeBefore} → ${entry.context.healthModeAfter}`);
      lines.push('');
    }

    return lines.join('\n');
  }

  formatAsJson(history: RecoveryHistoryEntry[]): string {
    return JSON.stringify(history, null, 2);
  }
}
```

```typescript
interface FilePathComponents {
  date: string;
  workspaceId: string;
  taskId: string;
  type: 'main' | 'heartbeat' | 'recovery';
}

class FilePathBuilder {
  static build(components: FilePathComponents): string {
    const { date, workspaceId, taskId, type } = components;
    
    switch (type) {
      case 'main':
        return `runs/${date}/${workspaceId}/${taskId}.md`;
      case 'heartbeat':
        return `runs/${date}/${workspaceId}/heartbeat/${taskId}.jsonl`;
      case 'recovery':
        return `runs/${date}/${workspaceId}/recovery/${taskId}.jsonl`;
    }
  }

  static parse(filePath: string): FilePathComponents | null {
    const match = filePath.match(/runs\/(\d{4}-\d{2}-\d{2})\/([^\/]+)\/(?:heartbeat\/|recovery\/)?([^\/]+)\.(md|jsonl)/);
    
    if (!match) return null;

    const [, date, workspaceId, taskId, extension] = match;
    
    let type: FilePathComponents['type'] = 'main';
    if (filePath.includes('/heartbeat/')) {
      type = 'heartbeat';
    } else if (filePath.includes('/recovery/')) {
      type = 'recovery';
    }

    return { date, workspaceId, taskId, type };
  }
}
```

## 9.4.9 监控指标

```typescript
interface RunsWriteMetrics {
  totalWrites: number;
  completedWrites: number;
  interruptedWrites: number;
  failedWrites: number;
  avgWriteTimeMs: number;
  totalBytesWritten: number;
  errors: number;
}

class RunsWriteMetricsCollector {
  private _metrics: RunsWriteMetrics = {
    totalWrites: 0,
    completedWrites: 0,
    interruptedWrites: 0,
    failedWrites: 0,
    avgWriteTimeMs: 0,
    totalBytesWritten: 0,
    errors: 0
  };

  recordWrite(state: TerminalState, durationMs: number, bytesWritten: number, error?: Error): void {
    this._metrics.totalWrites++;
    this._metrics.totalBytesWritten += bytesWritten;

    if (error) {
      this._metrics.errors++;
      return;
    }

    switch (state) {
      case 'completed':
        this._metrics.completedWrites++;
        break;
      case 'interrupted':
        this._metrics.interruptedWrites++;
        break;
      case 'failed':
        this._metrics.failedWrites++;
        break;
    }

    this.updateAvgWriteTime(durationMs);
  }

  private updateAvgWriteTime(durationMs: number): void {
    const n = this._metrics.totalWrites;
    this._metrics.avgWriteTimeMs = 
      (this._metrics.avgWriteTimeMs * (n - 1) + durationMs) / n;
  }

  getMetrics(): RunsWriteMetrics {
    return { ...this._metrics };
  }
}
```
