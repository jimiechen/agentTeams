# 4.5 模式判定算法

## 概述

模式判定算法是心跳检测系统的核心决策引擎，负责根据各层检测收集的信号和诊断数据，最终判定页面当前的心跳模式。该算法采用优先级链机制，确保判定的准确性和一致性。

## 心跳模式定义

```typescript
type HeartbeatMode = 
  | 'normal'      // 正常模式：页面活跃，心跳正常响应
  | 'idle'        // 空闲模式：页面可见但无用户交互
  | 'background'  // 后台模式：页面不可见或在后台标签页
  | 'frozen'      // 冻结模式：页面被浏览器冻结（CPU 暂停）
  | 'crashed';    // 崩溃模式：页面进程已崩溃

interface ModeDefinition {
  name: HeartbeatMode;
  description: string;
  signals: SignalType[];
  priority: number;
  recoveryAction?: RecoveryAction;
```

## 优先级链算法

### 核心算法实现

```typescript
class ModeDecisionEngine {
  private modeDefinitions: ModeDefinition[];
  private signalWeights: Map<SignalType, number>;
  private decisionHistory: DecisionRecord[];
  private maxHistorySize: number = 100;

  constructor() {
    this.modeDefinitions = MODE_DEFINITIONS.sort((a, b) => b.priority - a.priority);
    this.signalWeights = this.initializeSignalWeights();
    this.decisionHistory = [];
  }

  decide(signals: Signal[], diagnostics?: DeepDiagnostics): DecisionResult {
    const scoredModes = this.scoreAllModes(signals, diagnostics);
    
    const topMode = scoredModes[0];
    
    const confidence = this.calculateConfidence(topMode, scoredModes);
    
    const stabilityFactor = this.checkStability(topMode.mode);
    const adjustedConfidence = confidence * stabilityFactor;

    this.recordDecision({
      mode: topMode.mode,
      confidence: adjustedConfidence,
      signals,
      timestamp: Date.now()
    });

    return {
      mode: topMode.mode,
      confidence: adjustedConfidence,
      signals,
      matchedSignals: topMode.matchedSignals,
      layer: this.determineLayer(signals),
      alternativeModes: scoredModes.slice(1, 3).map(m => ({
        mode: m.mode,
        confidence: m.score / topMode.score
      }))
    };
  }

  private scoreAllModes(signals: Signal[], diagnostics?: DeepDiagnostics[]): ScoredMode[] {
    const scores: ScoredMode[] = [];

    for (const definition of this.modeDefinitions) {
      const score = this.scoreMode(definition, signals, diagnostics);
      scores.push({
        mode: definition.name,
        score,
        matchedSignals: score.matchedSignals,
        priority: definition.priority
      });
    }

    return scores.sort((a, b) => {
      if (b.score.total !== a.score.total) {
        return b.score.total - a.score.total;
      }
      return b.priority - a.priority;
    });
  }

  private scoreMode(
    definition: ModeDefinition, 
    signals: Signal[], 
    diagnostics?: DeepDiagnostics
  ): ModeScore {
    const matchedSignals: Signal[] = [];
    let signalScore = 0;
    let weightSum = 0;

    for (const signal of signals) {
      if (definition.signals.includes(signal.type)) {
        matchedSignals.push(signal);
        const weight = this.signalWeights.get(signal.type) || 0.5;
        signalScore += signal.weight * weight;
        weightSum += weight;
      }
    }

    const diagnosticScore = diagnostics 
      ? this.scoreDiagnostics(definition.name, diagnostics) 
      : 0;

    const normalizedSignalScore = weightSum > 0 ? signalScore / weightSum : 0;
    const total = normalizedSignalScore * 0.7 + diagnosticScore * 0.3;

    return {
      total,
      signalScore: normalizedSignalScore,
      diagnosticScore,
      matchedSignals
    };
  }

  private scoreDiagnostics(mode: HeartbeatMode, diagnostics: DeepDiagnostics): number {
    let score = 0;

    switch (mode) {
      case 'crashed':
        if (diagnostics.process?.frozen) score += 0.5;
        if (diagnostics.v8?.unresponsive) score += 0.5;
        break;

      case 'frozen':
        if (diagnostics.render?.stopped) score += 0.4;
        if (diagnostics.v8?.unresponsive) score += 0.3;
        if (diagnostics.memory?.critical) score += 0.3;
        break;

      case 'background':
        if (diagnostics.process && !diagnostics.process.attached) score += 0.5;
        break;

      case 'idle':
        if (diagnostics.network?.recentRequests === 0) score += 0.3;
        if (diagnostics.memory?.level === 'normal') score += 0.2;
        break;

      case 'normal':
        if (diagnostics.process?.attached) score += 0.3;
        if (diagnostics.render && !diagnostics.render.stopped) score += 0.3;
        if (diagnostics.v8 && !diagnostics.v8.unresponsive) score += 0.4;
        break;
    }

    return Math.min(score, 1);
  }

  private initializeSignalWeights(): Map<SignalType, number> {
    const weights = new Map<SignalType, number>();
    
    weights.set('process_frozen', 1.0);
    weights.set('render_stopped', 0.95);
    weights.set('thread_blocked', 0.9);
    weights.set('thread_responsive', 0.8);
    weights.set('visibility_hidden', 0.75);
    weights.set('user_interaction', 0.7);
    weights.set('network_active', 0.6);
    weights.set('dom_changed', 0.5);
    weights.set('memory_pressure', 0.4);

    return weights;
  }

  private calculateConfidence(topMode: ScoredMode, allModes: ScoredMode[]): number {
    if (allModes.length < 2) return topMode.score.total;

    const secondMode = allModes[1];
    const gap = topMode.score.total - secondMode.score.total;
    
    let confidence = topMode.score.total;
    
    if (gap < 0.1) {
      confidence *= 0.7;
    } else if (gap < 0.2) {
      confidence *= 0.85;
    } else if (gap < 0.3) {
      confidence *= 0.95;
    }

    return Math.min(confidence, 1);
  }

  private checkStability(mode: HeartbeatMode): number {
    if (this.decisionHistory.length < 3) return 1;

    const recentDecisions = this.decisionHistory.slice(-5);
    const sameModeCount = recentDecisions.filter(d => d.mode === mode).length;
    
    if (sameModeCount >= 4) return 1.0;
    if (sameModeCount >= 3) return 0.95;
    if (sameModeCount >= 2) return 0.9;
    
    return 0.85;
  }

  private determineLayer(signals: Signal[]): 1 | 2 | 3 {
    const layers = signals.map(s => {
      switch (s.source) {
        case 'layer1': return 1;
        case 'layer2': return 2;
        case 'layer3': return 3;
        default: return 1;
      }
    });
    
    return Math.max(...layers) as 1 | 2 | 3;
  }

  private recordDecision(record: DecisionRecord): void {
    this.decisionHistory.push(record);
    if (this.decisionHistory.length > this.maxHistorySize) {
      this.decisionHistory.shift();
    }
  }
}

interface ScoredMode {
  mode: HeartbeatMode;
  score: ModeScore;
  matchedSignals: Signal[];
  priority: number;
}

interface ModeScore {
  total: number;
  signalScore: number;
  diagnosticScore: number;
  matchedSignals: Signal[];
}

interface DecisionResult {
  mode: HeartbeatMode;
  confidence: number;
  signals: Signal[];
  matchedSignals: Signal[];
  layer: 1 | 2 | 3;
  alternativeModes: Array<{ mode: HeartbeatMode; confidence: number }>;
}

interface DecisionRecord {
  mode: HeartbeatMode;
  confidence: number;
  signals: Signal[];
  timestamp: number;
}
```

### 规则引擎

```typescript
class PriorityRuleEngine {
  private rules: PriorityRule[];

  constructor(rules: PriorityRule[] = PRIORITY_RULES) {
    this.rules = [...rules].sort((a, b) => b.priority - a.priority);
  }

  evaluate(signals: Signal[], diagnostics?: DeepDiagnostics): RuleEvaluationResult {
    const matchedRules: MatchedRule[] = [];

    for (const rule of this.rules) {
      try {
        if (rule.condition(signals, diagnostics)) {
          matchedRules.push({
            rule,
            timestamp: Date.now()
          });
        }
      } catch (error) {
        console.warn(`Rule ${rule.id} evaluation error:`, error);
      }
    }

    if (matchedRules.length === 0) {
      return {
        mode: 'normal',
        confidence: 0.5,
        matchedRules: [],
        reason: 'no_rules_matched'
      };
    }

    const topRule = matchedRules[0].rule;
    
    const sameResultRules = matchedRules.filter(m => m.rule.result === topRule.result);
    const boostedConfidence = this.boostConfidence(
      topRule.confidence, 
      sameResultRules.length
    );

    return {
      mode: topRule.result,
      confidence: boostedConfidence,
      matchedRules,
      reason: `${sameResultRules.length} rules confirmed`
    };
  }

  private boostConfidence(base: number, confirmations: number): number {
    if (confirmations <= 1) return base;
    if (confirmations === 2) return Math.min(base * 1.1, 1);
    if (confirmations >= 3) return Math.min(base * 1.15, 1);
    return base;
  }

  addRule(rule: PriorityRule): void {
    this.rules.push(rule);
    this.rules.sort((a, b) => b.priority - a.priority);
  }

  removeRule(ruleId: string): boolean {
    const index = this.rules.findIndex(r => r.id === ruleId);
    if (index >= 0) {
      this.rules.splice(index, 1);
      return true;
    }
    return false;
  }
}

interface RuleEvaluationResult {
  mode: HeartbeatMode;
  confidence: number;
  matchedRules: MatchedRule[];
  reason: string;
}

interface MatchedRule {
  rule: PriorityRule;
  timestamp: number;
}
```

## 信号融合算法

```typescript
class SignalFusionEngine {
  fuse(signals: Signal[]): FusedSignalSet {
    const grouped = this.groupByType(signals);
    const fused: FusedSignal[] = [];

    for (const [type, typeSignals] of grouped) {
      const fusedSignal = this.fuseSignalsOfType(type, typeSignals);
      fused.push(fusedSignal);
    }

    return {
      signals: fused,
      timestamp: Date.now(),
      sourceCount: signals.length,
      fusedCount: fused.length
    };
  }

  private groupByType(signals: Signal[]): Map<SignalType, Signal[]> {
    const grouped = new Map<SignalType, Signal[]>();
    
    for (const signal of signals) {
      const existing = grouped.get(signal.type) || [];
      existing.push(signal);
      grouped.set(signal.type, existing);
    }

    return grouped;
  }

  private fuseSignalsOfType(type: SignalType, signals: Signal[]): FusedSignal {
    const weights = signals.map(s => s.weight);
    const values = signals.map(s => s.value);
    const sources = signals.map(s => s.source);

    const avgWeight = weights.reduce((a, b) => a + b, 0) / weights.length;
    const maxWeight = Math.max(...weights);
    const minWeight = Math.min(...weights);

    const fusedValue = this.fuseValues(type, values);

    return {
      type,
      weight: maxWeight,
      avgWeight,
      weightVariance: this.calculateVariance(weights),
      value: fusedValue,
      sources: [...new Set(sources)],
      count: signals.length,
      timestamps: signals.map(s => s.timestamp)
    };
  }

  private fuseValues(type: SignalType, values: any[]): any {
    switch (type) {
      case 'thread_responsive':
      case 'thread_blocked':
        return this.fuseNumericValues(values);
      
      case 'visibility_hidden':
        return values[values.length - 1];
      
      case 'dom_changed':
        return {
          totalChanges: values.reduce((sum, v) => sum + (v || 0), 0),
          samples: values.length
        };
      
      default:
        return values[values.length - 1];
    }
  }

  private fuseNumericValues(values: number[]): number {
    const validValues = values.filter(v => typeof v === 'number');
    if (validValues.length === 0) return 0;
    
    const sorted = [...validValues].sort((a, b) => a - b);
    const mid = Math.floor(sorted.length / 2);
    
    if (sorted.length % 2 === 0) {
      return (sorted[mid - 1] + sorted[mid]) / 2;
    }
    return sorted[mid];
  }

  private calculateVariance(values: number[]): number {
    if (values.length === 0) return 0;
    const mean = values.reduce((a, b) => a + b, 0) / values.length;
    return values.reduce((sum, v) => sum + Math.pow(v - mean, 2), 0) / values.length;
  }
}

interface FusedSignalSet {
  signals: FusedSignal[];
  timestamp: number;
  sourceCount: number;
  fusedCount: number;
}

interface FusedSignal {
  type: SignalType;
  weight: number;
  avgWeight: number;
  weightVariance: number;
  value: any;
  sources: string[];
  count: number;
  timestamps: number[];
}
```

## 状态转换验证

```typescript
class StateTransitionValidator {
  private transitionMatrix: Map<HeartbeatMode, Set<HeartbeatMode>>;
  private lastMode: HeartbeatMode = 'normal';
  private transitionHistory: TransitionRecord[] = [];

  constructor() {
    this.transitionMatrix = this.buildTransitionMatrix();
  }

  private buildTransitionMatrix(): Map<HeartbeatMode, Set<HeartbeatMode>> {
    const matrix = new Map<HeartbeatMode, Set<HeartbeatMode>>();

    matrix.set('normal', new Set(['normal', 'idle', 'background', 'frozen', 'crashed']));
    matrix.set('idle', new Set(['normal', 'idle', 'background', 'frozen', 'crashed']));
    matrix.set('background', new Set(['normal', 'idle', 'background', 'frozen', 'crashed']));
    matrix.set('frozen', new Set(['normal', 'frozen', 'crashed']));
    matrix.set('crashed', new Set(['normal', 'crashed']));

    return matrix;
  }

  validate(newMode: HeartbeatMode): ValidationResult {
    const allowedTransitions = this.transitionMatrix.get(this.lastMode);
    
    if (!allowedTransitions) {
      return {
        valid: false,
        reason: 'Unknown current mode',
        suggestedMode: this.lastMode
      };
    }

    if (allowedTransitions.has(newMode)) {
      this.recordTransition(this.lastMode, newMode);
      this.lastMode = newMode;
      
      return {
        valid: true,
        reason: 'Valid transition',
        suggestedMode: newMode
      };
    }

    const suggestedMode = this.suggestAlternativeMode(newMode);
    
    return {
      valid: false,
      reason: `Invalid transition from ${this.lastMode} to ${newMode}`,
      suggestedMode
    };
  }

  private suggestAlternativeMode(requestedMode: HeartbeatMode): HeartbeatMode {
    const allowedTransitions = this.transitionMatrix.get(this.lastMode);
    
    if (!allowedTransitions) return this.lastMode;

    if (requestedMode === 'crashed' && allowedTransitions.has('frozen')) {
      return 'frozen';
    }

    if (requestedMode === 'normal' && allowedTransitions.has('idle')) {
      return 'idle';
    }

    return this.lastMode;
  }

  private recordTransition(from: HeartbeatMode, to: HeartbeatMode): void {
    this.transitionHistory.push({
      from,
      to,
      timestamp: Date.now()
    });

    if (this.transitionHistory.length > 100) {
      this.transitionHistory.shift();
    }
  }

  getTransitionHistory(): TransitionRecord[] {
    return [...this.transitionHistory];
  }
}

interface ValidationResult {
  valid: boolean;
  reason: string;
  suggestedMode: HeartbeatMode;
}

interface TransitionRecord {
  from: HeartbeatMode;
  to: HeartbeatMode;
  timestamp: number;
}
```

## 综合决策流程

```typescript
class ComprehensiveDecisionMaker {
  private modeEngine: ModeDecisionEngine;
  private ruleEngine: PriorityRuleEngine;
  private fusionEngine: SignalFusionEngine;
  private transitionValidator: StateTransitionValidator;

  constructor() {
    this.modeEngine = new ModeDecisionEngine();
    this.ruleEngine = new PriorityRuleEngine();
    this.fusionEngine = new SignalFusionEngine();
    this.transitionValidator = new StateTransitionValidator();
  }

  async decide(
    signals: Signal[], 
    diagnostics?: DeepDiagnostics
  ): Promise<FinalDecision> {
    const fusedSignals = this.fusionEngine.fuse(signals);

    const modeResult = this.modeEngine.decide(signals, diagnostics);
    const ruleResult = this.ruleEngine.evaluate(signals, diagnostics);

    const combinedMode = this.combineResults(modeResult, ruleResult);

    const validation = this.transitionValidator.validate(combinedMode.mode);

    const finalMode = validation.valid 
      ? combinedMode.mode 
      : validation.suggestedMode;

    return {
      mode: finalMode,
      confidence: combinedMode.confidence,
      signals: fusedSignals,
      matchedRules: ruleResult.matchedRules.map(m => m.rule.id),
      layer: modeResult.layer,
      alternatives: modeResult.alternativeModes,
      transitionValid: validation.valid,
      transitionReason: validation.reason,
      timestamp: Date.now()
    };
  }

  private combineResults(
    modeResult: DecisionResult, 
    ruleResult: RuleEvaluationResult
  ): { mode: HeartbeatMode; confidence: number } {
    if (modeResult.mode === ruleResult.mode) {
      return {
        mode: modeResult.mode,
        confidence: Math.max(modeResult.confidence, ruleResult.confidence)
      };
    }

    const modePriority = MODE_DEFINITIONS.find(d => d.name === modeResult.mode)?.priority || 0;
    const rulePriority = MODE_DEFINITIONS.find(d => d.name === ruleResult.mode)?.priority || 0;

    if (rulePriority > modePriority && ruleResult.confidence > 0.8) {
      return {
        mode: ruleResult.mode,
        confidence: ruleResult.confidence * 0.9
      };
    }

    return {
      mode: modeResult.mode,
      confidence: modeResult.confidence * 0.85
    };
  }
}

interface FinalDecision {
  mode: HeartbeatMode;
  confidence: number;
  signals: FusedSignalSet;
  matchedRules: string[];
  layer: 1 | 2 | 3;
  alternatives: Array<{ mode: HeartbeatMode; confidence: number }>;
  transitionValid: boolean;
  transitionReason: string;
  timestamp: number;
}
```

## 算法性能优化

```typescript
class DecisionOptimizer {
  private cache: Map<string, CachedDecision> = new Map();
  private cacheTimeout: number = 5000;

  optimize(decision: FinalDecision): FinalDecision {
    const cacheKey = this.generateCacheKey(decision.signals);
    const cached = this.cache.get(cacheKey);

    if (cached && Date.now() - cached.timestamp < this.cacheTimeout) {
      return {
        ...decision,
        confidence: Math.max(decision.confidence, cached.decision.confidence)
      };
    }

    this.cache.set(cacheKey, {
      decision,
      timestamp: Date.now()
    });

    this.cleanupCache();

    return decision;
  }

  private generateCacheKey(signals: FusedSignalSet): string {
    const typeStrings = signals.signals
      .map(s => `${s.type}:${s.weight.toFixed(2)}`)
      .sort()
      .join('|');
    return `signals:${typeStrings}`;
  }

  private cleanupCache(): void {
    const now = Date.now();
    for (const [key, cached] of this.cache) {
      if (now - cached.timestamp > this.cacheTimeout * 2) {
        this.cache.delete(key);
      }
    }
  }
}

interface CachedDecision {
  decision: FinalDecision;
  timestamp: number;
}
```
