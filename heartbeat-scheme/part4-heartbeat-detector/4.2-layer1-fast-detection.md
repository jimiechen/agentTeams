# 4.2 Layer 1 快速探测脚本

## 概述

Layer 1 快速探测是心跳检测的第一道防线，目标是快速判断页面主线程的响应性。该层设计为轻量级、高频执行，能够在毫秒级完成检测。

## 设计目标

| 指标 | 目标值 | 说明 |
|------|--------|------|
| 执行时间 | < 50ms | 不影响主线程性能 |
| 内存占用 | < 1MB | 极低内存开销 |
| CPU 使用 | < 5% | 最小化计算开销 |
| 检测频率 | 1-10 Hz | 支持高频轮询 |

## 核心检测方法

### fastCheck() 实现

```typescript
class HeartbeatDetector {
  private lastHeartbeatTime: number = Date.now();
  private heartbeatInterval: number = 1000;
  private responsivenessThreshold: number = 100;

  async fastCheck(): Promise<DetectionResult> {
    const startTime = performance.now();
    const signals: Signal[] = [];

    const threadCheck = await this.checkThreadResponsiveness();
    signals.push({
      type: threadCheck.responsive ? 'thread_responsive' : 'thread_blocked',
      source: 'layer1',
      value: threadCheck.latency,
      timestamp: Date.now(),
      weight: threadCheck.responsive ? 0.8 : 0.9
    });

    const visibilityCheck = this.checkVisibility();
    if (visibilityCheck.hidden) {
      signals.push({
        type: 'visibility_hidden',
        source: 'layer1',
        value: visibilityCheck.reason,
        timestamp: Date.now(),
        weight: 0.7
      });
    }

    const basicDomCheck = this.checkBasicDomState();
    if (basicDomCheck.hasChanges) {
      signals.push({
        type: 'dom_changed',
        source: 'layer1',
        value: basicDomCheck.changeCount,
        timestamp: Date.now(),
        weight: 0.5
      });
    }

    const mode = this.determineModeFromSignals(signals);
    const confidence = this.calculateConfidence(signals);
    const executionTime = performance.now() - startTime;

    return {
      mode,
      confidence,
      signals,
      timestamp: Date.now(),
      layer: 1
    };
  }

  private async checkThreadResponsiveness(): Promise<{
    responsive: boolean;
    latency: number;
  }> {
    return new Promise((resolve) => {
      const start = performance.now();
      const timeout = this.config.layer1Timeout;

      const timer = setTimeout(() => {
        resolve({
          responsive: false,
          latency: timeout
        });
      }, timeout);

      requestAnimationFrame(() => {
        clearTimeout(timer);
        const latency = performance.now() - start;
        resolve({
          responsive: latency < this.responsivenessThreshold,
          latency
        });
      });
    });
  }

  private checkVisibility(): { hidden: boolean; reason?: string } {
    if (document.hidden) {
      return {
        hidden: true,
        reason: document.visibilityState
      };
    }

    if (document.hasFocus() === false) {
      return {
        hidden: true,
        reason: 'no_focus'
      };
    }

    return { hidden: false };
  }

  private checkBasicDomState(): { hasChanges: boolean; changeCount: number } {
    const currentMutations = this.getMutationCount();
    const hasChanges = currentMutations > this.lastMutationCount;
    const changeCount = currentMutations - this.lastMutationCount;
    
    this.lastMutationCount = currentMutations;
    
    return { hasChanges, changeCount };
  }

  private lastMutationCount: number = 0;

  private getMutationCount(): number {
    return (document as any).__mutationCount || 0;
  }
}
```

## 主线程响应性检测

### requestAnimationFrame 测量法

```typescript
class ThreadResponsivenessChecker {
  private samples: number[] = [];
  private sampleSize: number = 10;
  private threshold: number = 50;

  measure(): Promise<number> {
    return new Promise((resolve) => {
      const start = performance.now();
      
      requestAnimationFrame(() => {
        const latency = performance.now() - start;
        this.samples.push(latency);
        
        if (this.samples.length > this.sampleSize) {
          this.samples.shift();
        }
        
        resolve(latency);
      });
    });
  }

  getAverageLatency(): number {
    if (this.samples.length === 0) return 0;
    return this.samples.reduce((a, b) => a + b, 0) / this.samples.length;
  }

  isResponsive(): boolean {
    const avg = this.getAverageLatency();
    return avg < this.threshold;
  }

  getResponsivenessScore(): number {
    const avg = this.getAverageLatency();
    if (avg < 16) return 1.0;
    if (avg < 33) return 0.8;
    if (avg < 50) return 0.6;
    if (avg < 100) return 0.4;
    return 0.2;
  }
}
```

### MessageChannel 测量法

```typescript
class MessageChannelChecker {
  private channel: MessageChannel;
  private pendingResolve: ((value: number) => void) | null = null;
  private startTime: number = 0;

  constructor() {
    this.channel = new MessageChannel();
    this.channel.port2.onmessage = () => {
      if (this.pendingResolve) {
        const latency = performance.now() - this.startTime;
        this.pendingResolve(latency);
        this.pendingResolve = null;
      }
    };
  }

  measure(): Promise<number> {
    return new Promise((resolve) => {
      this.pendingResolve = resolve;
      this.startTime = performance.now();
      this.channel.port1.postMessage(null);
    });
  }
}
```

## 页面可见性检测

```typescript
class VisibilityChecker {
  private listeners: Array<(state: VisibilityState) => void> = [];

  constructor() {
    this.setupListeners();
  }

  private setupListeners(): void {
    document.addEventListener('visibilitychange', () => {
      this.notifyListeners(document.visibilityState);
    });

    window.addEventListener('blur', () => {
      this.notifyListeners('unfocused');
    });

    window.addEventListener('focus', () => {
      this.notifyListeners('focused');
    });
  }

  getState(): VisibilityState {
    return {
      hidden: document.hidden,
      visibilityState: document.visibilityState,
      hasFocus: document.hasFocus(),
      reason: this.determineReason()
    };
  }

  private determineReason(): string {
    if (document.hidden) {
      return `visibility:${document.visibilityState}`;
    }
    if (!document.hasFocus()) {
      return 'no_focus';
    }
    return 'visible';
  }

  onStateChange(callback: (state: VisibilityState) => void): void {
    this.listeners.push(callback);
  }

  private notifyListeners(state: VisibilityState): void {
    this.listeners.forEach(cb => cb(this.getState()));
  }
}

interface VisibilityState {
  hidden: boolean;
  visibilityState: DocumentVisibilityState;
  hasFocus: boolean;
  reason: string;
}
```

## DOM 变化快速检测

```typescript
class FastDomChangeDetector {
  private observer: MutationObserver;
  private changeCount: number = 0;
  private lastCheckTime: number = Date.now();
  private changeRate: number = 0;

  constructor() {
    this.observer = new MutationObserver((mutations) => {
      this.changeCount += mutations.length;
    });

    this.observer.observe(document.body, {
      childList: true,
      subtree: true,
      attributes: true,
      characterData: true
    });
  }

  check(): { hasChanges: boolean; rate: number } {
    const now = Date.now();
    const elapsed = now - this.lastCheckTime;
    
    this.changeRate = this.changeCount / (elapsed / 1000);
    
    const result = {
      hasChanges: this.changeCount > 0,
      rate: this.changeRate
    };

    this.changeCount = 0;
    this.lastCheckTime = now;

    return result;
  }

  getChangeRate(): number {
    return this.changeRate;
  }

  disconnect(): void {
    this.observer.disconnect();
  }
}
```

## 快速探测决策逻辑

```typescript
class Layer1DecisionMaker {
  decide(signals: Signal[]): { 
    mode: HeartbeatMode | 'unknown'; 
    confidence: number;
    needsDeeperCheck: boolean;
  } {
    const hasBlockedThread = signals.some(s => s.type === 'thread_blocked');
    const hasVisibilityHidden = signals.some(s => s.type === 'visibility_hidden');
    const hasDomChanges = signals.some(s => s.type === 'dom_changed');

    if (hasBlockedThread) {
      const blockedSignal = signals.find(s => s.type === 'thread_blocked');
      if (blockedSignal && blockedSignal.value > 1000) {
        return {
          mode: 'frozen',
          confidence: 0.7,
          needsDeeperCheck: true
        };
      }
    }

    if (hasVisibilityHidden) {
      const hiddenSignal = signals.find(s => s.type === 'visibility_hidden');
      return {
        mode: 'background',
        confidence: 0.8,
        needsDeeperCheck: false
      };
    }

    if (hasDomChanges) {
      return {
        mode: 'normal',
        confidence: 0.6,
        needsDeeperCheck: false
      };
    }

    return {
      mode: 'unknown',
      confidence: 0.3,
      needsDeeperCheck: true
    };
  }
}
```

## 性能优化策略

### 1. 采样频率自适应

```typescript
class AdaptiveSampler {
  private baseInterval: number = 1000;
  private minInterval: number = 100;
  private maxInterval: number = 5000;
  private currentInterval: number;
  private recentResults: boolean[] = [];

  adjustInterval(isResponsive: boolean): number {
    this.recentResults.push(isResponsive);
    if (this.recentResults.length > 10) {
      this.recentResults.shift();
    }

    const responsiveRate = this.recentResults.filter(r => r).length / this.recentResults.length;

    if (responsiveRate > 0.9) {
      this.currentInterval = Math.min(this.maxInterval, this.currentInterval * 1.2);
    } else if (responsiveRate < 0.5) {
      this.currentInterval = Math.max(this.minInterval, this.currentInterval * 0.8);
    }

    return this.currentInterval;
  }
}
```

### 2. 检测项动态选择

```typescript
class DynamicCheckSelector {
  private checkWeights: Map<string, number> = new Map();

  constructor() {
    this.checkWeights.set('thread', 1.0);
    this.checkWeights.set('visibility', 0.8);
    this.checkWeights.set('dom', 0.5);
  }

  selectChecks(availableTime: number): string[] {
    const checks: string[] = [];
    let remainingTime = availableTime;

    const sortedChecks = Array.from(this.checkWeights.entries())
      .sort((a, b) => b[1] - a[1]);

    for (const [check, weight] of sortedChecks) {
      const estimatedTime = this.estimateCheckTime(check);
      if (estimatedTime <= remainingTime) {
        checks.push(check);
        remainingTime -= estimatedTime;
      }
    }

    return checks;
  }

  private estimateCheckTime(check: string): number {
    const estimates: Record<string, number> = {
      'thread': 16,
      'visibility': 1,
      'dom': 10
    };
    return estimates[check] || 10;
  }
}
```
