# 4.4 Layer 3 深度探测脚本

## 概述

Layer 3 深度探测是心跳检测的最后一道防线，当 Layer 1 和 Layer 2 都无法确定页面状态时触发。该层执行全面、深入的页面状态分析，包括进程状态、渲染管线、内存详情和网络连接等。

## 设计目标

| 指标 | 目标值 | 说明 |
|------|--------|------|
| 执行时间 | 1s - 15s | 允许较长执行时间 |
| 准确率 | > 99% | 最高准确度 |
| 完整性 | 全面 | 获取所有可用信息 |
| 检测频率 | 按需/低频 | 仅在必要时执行 |

## deepCheck() 实现

```typescript
class HeartbeatDetector {
  private cdpClient: CDPClient;
  private processMonitor: ProcessMonitor;
  private renderMonitor: RenderMonitor;

  async deepCheck(): Promise<DetectionResult> {
    const startTime = performance.now();
    const signals: Signal[] = [];
    const diagnostics: DeepDiagnostics = {};

    try {
      diagnostics.process = await this.checkProcessState();
      if (diagnostics.process.frozen) {
        signals.push({
          type: 'process_frozen',
          source: 'layer3',
          value: diagnostics.process,
          timestamp: Date.now(),
          weight: 0.95
        });
      }

      diagnostics.render = await this.checkRenderPipeline();
      if (diagnostics.render.stopped) {
        signals.push({
          type: 'render_stopped',
          source: 'layer3',
          value: diagnostics.render,
          timestamp: Date.now(),
          weight: 0.9
        });
      }

      diagnostics.memory = await this.checkMemoryDetails();
      if (diagnostics.memory.critical) {
        signals.push({
          type: 'memory_pressure',
          source: 'layer3',
          value: diagnostics.memory,
          timestamp: Date.now(),
          weight: 0.85
        });
      }

      diagnostics.network = await this.checkNetworkConnections();
      if (diagnostics.network.disconnected) {
        signals.push({
          type: 'network_active',
          source: 'layer3',
          value: diagnostics.network,
          timestamp: Date.now(),
          weight: 0.7
        });
      }

      diagnostics.v8 = await this.checkV8State();
      if (diagnostics.v8.unresponsive) {
        signals.push({
          type: 'thread_blocked',
          source: 'layer3',
          value: diagnostics.v8,
          timestamp: Date.now(),
          weight: 0.9
        });
      }

      diagnostics.dom = await this.checkDOMIntegrity();
      if (diagnostics.dom.corrupted) {
        signals.push({
          type: 'thread_blocked',
          source: 'layer3',
          value: diagnostics.dom,
          timestamp: Date.now(),
          weight: 0.8
        });
      }

      const mode = this.determineModeFromDeepCheck(diagnostics, signals);
      const confidence = this.calculateDeepConfidence(signals, diagnostics);
      const executionTime = performance.now() - startTime;

      return {
        mode,
        confidence,
        signals,
        timestamp: Date.now(),
        layer: 3
      };

    } catch (error) {
      return this.handleDeepCheckError(error as Error);
    }
  }
}

interface DeepDiagnostics {
  process?: ProcessDiagnostics;
  render?: RenderDiagnostics;
  memory?: MemoryDiagnostics;
  network?: NetworkDiagnostics;
  v8?: V8Diagnostics;
  dom?: DOMDiagnostics;
}
```

## 进程状态检测

```typescript
class ProcessMonitor {
  private cdpClient: CDPClient;

  async checkProcessState(): Promise<ProcessDiagnostics> {
    const targetInfo = await this.getTargetInfo();
    const processInfo = await this.getProcessInfo();

    return {
      targetId: targetInfo.targetId,
      type: targetInfo.type,
      url: targetInfo.url,
      attached: targetInfo.attached,
      canAccessOpener: targetInfo.canAccessOpener,
      processId: processInfo.processId,
      processType: processInfo.processType,
      cpuTime: processInfo.cpuTime,
      frozen: this.detectFrozenProcess(targetInfo, processInfo)
    };
  }

  private async getTargetInfo(): Promise<TargetInfo> {
    const result = await this.cdpClient.Target.getTargetInfo();
    return result.targetInfo;
  }

  private async getProcessInfo(): Promise<ProcessInfo> {
    try {
      const result = await this.cdpClient.SystemInfo.getProcessInfo();
      return result;
    } catch {
      return { processId: -1, processType: 'unknown', cpuTime: 0 };
    }
  }

  private detectFrozenProcess(target: TargetInfo, process: ProcessInfo): boolean {
    if (!target.attached) return true;
    
    return false;
  }

  async checkProcessHealth(): Promise<ProcessHealth> {
    const metrics = await this.cdpClient.Performance.getMetrics();
    
    const relevantMetrics = {
      timestamp: Date.now(),
      documents: this.findMetric(metrics, 'Documents'),
      frames: this.findMetric(metrics, 'Frames'),
      jsEventListeners: this.findMetric(metrics, 'JSEventListeners'),
      nodes: this.findMetric(metrics, 'Nodes'),
      layoutCount: this.findMetric(metrics, 'LayoutCount'),
      recalcStyleCount: this.findMetric(metrics, 'RecalcStyleCount'),
      layoutDuration: this.findMetric(metrics, 'LayoutDuration'),
      recalcStyleDuration: this.findMetric(metrics, 'RecalcStyleDuration')
    };

    return {
      metrics: relevantMetrics,
      healthy: this.evaluateProcessHealth(relevantMetrics)
    };
  }

  private findMetric(metrics: any[], name: string): number {
    const metric = metrics.find(m => m.name === name);
    return metric ? metric.value : 0;
  }

  private evaluateProcessHealth(metrics: Record<string, number>): boolean {
    if (metrics.nodes > 100000) return false;
    if (metrics.jsEventListeners > 10000) return false;
    
    return true;
  }
}

interface ProcessDiagnostics {
  targetId: string;
  type: string;
  url: string;
  attached: boolean;
  canAccessOpener: boolean;
  processId: number;
  processType: string;
  cpuTime: number;
  frozen: boolean;
}

interface ProcessHealth {
  metrics: Record<string, number>;
  healthy: boolean;
}
```

## 渲染管线检测

```typescript
class RenderMonitor {
  private cdpClient: CDPClient;
  private frameHistory: FrameInfo[] = [];
  private maxHistorySize: number = 60;

  async checkRenderPipeline(): Promise<RenderDiagnostics> {
    const layerTree = await this.getLayerTree();
    const compositingInfo = await this.getCompositingInfo();
    const paintEvents = await this.capturePaintEvents();

    const currentFrame = await this.captureFrame();
    this.frameHistory.push(currentFrame);
    if (this.frameHistory.length > this.maxHistorySize) {
      this.frameHistory.shift();
    }

    const frameRate = this.calculateFrameRate();
    const stopped = this.detectRenderStopped(frameRate, paintEvents);

    return {
      layerCount: layerTree?.layers?.length || 0,
      compositingEnabled: compositingInfo.enabled,
      lastPaintTime: paintEvents.lastPaintTime,
      paintCount: paintEvents.count,
      frameRate,
      stopped,
      reason: stopped ? this.determineStopReason(frameRate, paintEvents) : null
    };
  }

  private async getLayerTree(): Promise<LayerTree | null> {
    try {
      const result = await this.cdpClient.LayerTree.layerTreeDidChange({});
      return result;
    } catch {
      return null;
    }
  }

  private async getCompositingInfo(): Promise<{ enabled: boolean }> {
    try {
      const result = await this.cdpClient.Compositing.getCompositingMode();
      return { enabled: result.enabled };
    } catch {
      return { enabled: true };
    }
  }

  private async capturePaintEvents(): Promise<PaintEventsInfo> {
    const script = `
      (function() {
        const entries = performance.getEntriesByType('paint');
        return {
          count: entries.length,
          lastPaintTime: entries.length > 0 
            ? entries[entries.length - 1].startTime 
            : 0,
          firstPaint: entries.find(e => e.name === 'first-paint')?.startTime || null,
          firstContentfulPaint: entries.find(e => e.name === 'first-contentful-paint')?.startTime || null
        };
      })();
    `;

    const result = await this.cdpClient.Runtime.evaluate({
      expression: script,
      returnByValue: true
    });

    return result.result.value as PaintEventsInfo;
  }

  private async captureFrame(): Promise<FrameInfo> {
    return new Promise((resolve) => {
      const start = performance.now();
      requestAnimationFrame((timestamp) => {
        resolve({
          timestamp,
          captureTime: performance.now() - start
        });
      });
    });
  }

  private calculateFrameRate(): number {
    if (this.frameHistory.length < 2) return 0;

    const recentFrames = this.frameHistory.slice(-10);
    const timeDiff = recentFrames[recentFrames.length - 1].timestamp - recentFrames[0].timestamp;
    
    if (timeDiff === 0) return 0;
    
    return (recentFrames.length - 1) / (timeDiff / 1000);
  }

  private detectRenderStopped(frameRate: number, paintEvents: PaintEventsInfo): boolean {
    if (frameRate === 0 && this.frameHistory.length >= 10) {
      return true;
    }

    const timeSinceLastPaint = Date.now() - paintEvents.lastPaintTime;
    if (timeSinceLastPaint > 30000 && paintEvents.count > 0) {
      return true;
    }

    return false;
  }

  private determineStopReason(frameRate: number, paintEvents: PaintEventsInfo): string {
    if (frameRate === 0) {
      return 'frame_rate_zero';
    }
    if (Date.now() - paintEvents.lastPaintTime > 30000) {
      return 'no_recent_paint';
    }
    return 'unknown';
  }
}

interface RenderDiagnostics {
  layerCount: number;
  compositingEnabled: boolean;
  lastPaintTime: number;
  paintCount: number;
  frameRate: number;
  stopped: boolean;
  reason: string | null;
}

interface FrameInfo {
  timestamp: number;
  captureTime: number;
}

interface PaintEventsInfo {
  count: number;
  lastPaintTime: number;
  firstPaint: number | null;
  firstContentfulPaint: number | null;
}
```

## V8 状态检测

```typescript
class V8Monitor {
  private cdpClient: CDPClient;

  async checkV8State(): Promise<V8Diagnostics> {
    const heapStats = await this.getHeapStatistics();
    const isolateState = await this.getIsolateState();
    const scriptState = await this.getScriptState();

    return {
      heap: heapStats,
      isolate: isolateState,
      scripts: scriptState,
      unresponsive: this.detectUnresponsive(heapStats, isolateState)
    };
  }

  private async getHeapStatistics(): Promise<HeapStatistics> {
    const script = `
      (function() {
        if (typeof v8 !== 'undefined' && v8.getHeapStatistics) {
          return v8.getHeapStatistics();
        }
        
        if (performance.memory) {
          return {
            total_heap_size: performance.memory.totalJSHeapSize,
            used_heap_size: performance.memory.usedJSHeapSize,
            heap_size_limit: performance.memory.jsHeapSizeLimit,
            malloced_memory: 0,
            peak_malloced_memory: 0,
            number_of_native_contexts: 1,
            number_of_detached_contexts: 0
          };
        }
        
        return null;
      })();
    `;

    const result = await this.cdpClient.Runtime.evaluate({
      expression: script,
      returnByValue: true
    });

    return result.result.value as HeapStatistics;
  }

  private async getIsolateState(): Promise<IsolateState> {
    try {
      const result = await this.cdpClient.Runtime.evaluate({
        expression: '1+1',
        returnByValue: true,
        timeout: 5000
      });

      return {
        responsive: result.result.value === 2,
        lastCheck: Date.now()
      };
    } catch (error) {
      return {
        responsive: false,
        lastCheck: Date.now(),
        error: (error as Error).message
      };
    }
  }

  private async getScriptState(): Promise<ScriptState> {
    try {
      const result = await this.cdpClient.Debugger.getScriptSource({
        scriptId: '1'
      });
      
      return {
        accessible: true,
        count: await this.getScriptCount()
      };
    } catch {
      return {
        accessible: false,
        count: 0
      };
    }
  }

  private async getScriptCount(): Promise<number> {
    try {
      const result = await this.cdpClient.Debugger.getPossibleBreakpoints({
        start: { lineNumber: 0, columnNumber: 0, scriptId: '1' },
        end: { lineNumber: 1, columnNumber: 0, scriptId: '1' }
      });
      return 1;
    } catch {
      return 0;
    }
  }

  private detectUnresponsive(heap: HeapStatistics | null, isolate: IsolateState): boolean {
    if (!isolate.responsive) return true;
    
    if (heap) {
      const usageRatio = heap.used_heap_size / heap.heap_size_limit;
      if (usageRatio > 0.99) return true;
      
      if (heap.number_of_detached_contexts > 100) return true;
    }

    return false;
  }
}

interface V8Diagnostics {
  heap: HeapStatistics | null;
  isolate: IsolateState;
  scripts: ScriptState;
  unresponsive: boolean;
}

interface HeapStatistics {
  total_heap_size: number;
  used_heap_size: number;
  heap_size_limit: number;
  malloced_memory: number;
  peak_malloced_memory: number;
  number_of_native_contexts: number;
  number_of_detached_contexts: number;
}

interface IsolateState {
  responsive: boolean;
  lastCheck: number;
  error?: string;
}

interface ScriptState {
  accessible: boolean;
  count: number;
}
```

## DOM 完整性检测

```typescript
class DOMIntegrityChecker {
  private cdpClient: CDPClient;

  async checkDOMIntegrity(): Promise<DOMDiagnostics> {
    const structureCheck = await this.checkStructure();
    const consistencyCheck = await this.checkConsistency();
    const accessibilityCheck = await this.checkAccessibility();

    return {
      structure: structureCheck,
      consistency: consistencyCheck,
      accessibility: accessibilityCheck,
      corrupted: this.detectCorruption(structureCheck, consistencyCheck)
    };
  }

  private async checkStructure(): Promise<StructureCheck> {
    const script = `
      (function() {
        const result = {
          hasDocument: !!document,
          hasDocumentElement: !!document.documentElement,
          hasHead: !!document.head,
          hasBody: !!document.body,
          doctype: document.doctype ? document.doctype.name : null,
          readyState: document.readyState,
          characterSet: document.characterSet,
          contentType: document.contentType
        };
        
        result.valid = result.hasDocument && 
                       result.hasDocumentElement && 
                       result.hasHead && 
                       result.hasBody;
        
        return result;
      })();
    `;

    const evalResult = await this.cdpClient.Runtime.evaluate({
      expression: script,
      returnByValue: true
    });

    return evalResult.result.value as StructureCheck;
  }

  private async checkConsistency(): Promise<ConsistencyCheck> {
    const script = `
      (function() {
        const allElements = document.getElementsByTagName('*');
        const allById = document.querySelectorAll('[id]');
        const ids = new Set();
        let duplicateIds = 0;
        
        allById.forEach(el => {
          if (ids.has(el.id)) {
            duplicateIds++;
          }
          ids.add(el.id);
        });

        const forms = document.forms;
        let validForms = 0;
        let invalidForms = 0;
        
        for (let i = 0; i < forms.length; i++) {
          try {
            forms[i].checkValidity();
            validForms++;
          } catch {
            invalidForms++;
          }
        }

        return {
          totalElements: allElements.length,
          elementsWithId: allById.length,
          duplicateIds,
          formCount: forms.length,
          validForms,
          invalidForms,
          consistent: duplicateIds === 0 && invalidForms === 0
        };
      })();
    `;

    const result = await this.cdpClient.Runtime.evaluate({
      expression: script,
      returnByValue: true
    });

    return result.result.value as ConsistencyCheck;
  }

  private async checkAccessibility(): Promise<AccessibilityCheck> {
    const script = `
      (function() {
        const images = document.getElementsByTagName('img');
        let imagesWithoutAlt = 0;
        
        for (let i = 0; i < images.length; i++) {
          if (!images[i].hasAttribute('alt')) {
            imagesWithoutAlt++;
          }
        }

        const buttons = document.getElementsByTagName('button');
        let buttonsWithoutText = 0;
        
        for (let i = 0; i < buttons.length; i++) {
          if (!buttons[i].innerText && !buttons[i].getAttribute('aria-label')) {
            buttonsWithoutText++;
          }
        }

        return {
          imageCount: images.length,
          imagesWithoutAlt,
          buttonCount: buttons.length,
          buttonsWithoutText,
          accessible: imagesWithoutAlt === 0 && buttonsWithoutText === 0
        };
      })();
    `;

    const result = await this.cdpClient.Runtime.evaluate({
      expression: script,
      returnByValue: true
    });

    return result.result.value as AccessibilityCheck;
  }

  private detectCorruption(structure: StructureCheck, consistency: ConsistencyCheck): boolean {
    if (!structure.valid) return true;
    if (!consistency.consistent) return false;
    return false;
  }
}

interface DOMDiagnostics {
  structure: StructureCheck;
  consistency: ConsistencyCheck;
  accessibility: AccessibilityCheck;
  corrupted: boolean;
}

interface StructureCheck {
  hasDocument: boolean;
  hasDocumentElement: boolean;
  hasHead: boolean;
  hasBody: boolean;
  doctype: string | null;
  readyState: DocumentReadyState;
  characterSet: string;
  contentType: string;
  valid: boolean;
}

interface ConsistencyCheck {
  totalElements: number;
  elementsWithId: number;
  duplicateIds: number;
  formCount: number;
  validForms: number;
  invalidForms: number;
  consistent: boolean;
}

interface AccessibilityCheck {
  imageCount: number;
  imagesWithoutAlt: number;
  buttonCount: number;
  buttonsWithoutText: number;
  accessible: boolean;
}
```

## 深度探测决策逻辑

```typescript
class Layer3DecisionMaker {
  decide(diagnostics: DeepDiagnostics, signals: Signal[]): {
    mode: HeartbeatMode;
    confidence: number;
    details: string;
  } {
    if (diagnostics.process?.frozen) {
      return {
        mode: 'frozen',
        confidence: 0.95,
        details: 'Process detected as frozen'
      };
    }

    if (diagnostics.v8?.unresponsive) {
      return {
        mode: 'crashed',
        confidence: 0.9,
        details: 'V8 isolate unresponsive'
      };
    }

    if (diagnostics.render?.stopped) {
      return {
        mode: 'frozen',
        confidence: 0.85,
        details: `Render pipeline stopped: ${diagnostics.render.reason}`
      };
    }

    if (diagnostics.dom?.corrupted) {
      return {
        mode: 'crashed',
        confidence: 0.8,
        details: 'DOM structure corrupted'
      };
    }

    if (diagnostics.memory?.critical) {
      return {
        mode: 'frozen',
        confidence: 0.75,
        details: 'Critical memory pressure'
      };
    }

    if (signals.length === 0) {
      return {
        mode: 'normal',
        confidence: 0.7,
        details: 'No issues detected in deep check'
      };
    }

    const signalWeights = signals.map(s => s.weight);
    const avgWeight = signalWeights.reduce((a, b) => a + b, 0) / signalWeights.length;

    if (avgWeight > 0.8) {
      return {
        mode: 'frozen',
        confidence: avgWeight,
        details: 'High-weight signals indicate frozen state'
      };
    }

    return {
      mode: 'normal',
      confidence: 0.6,
      details: 'Deep check completed with minor concerns'
    };
  }
}
```

## 错误处理

```typescript
class Layer3ErrorHandler {
  handle(error: Error): DetectionResult {
    console.error('[Layer3] Deep check error:', error);

    if (error.message.includes('target crashed') || 
        error.message.includes('Target closed')) {
      return {
        mode: 'crashed',
        confidence: 0.95,
        signals: [{
          type: 'process_frozen',
          source: 'layer3',
          value: error.message,
          timestamp: Date.now(),
          weight: 0.95
        }],
        timestamp: Date.now(),
        layer: 3
      };
    }

    if (error.message.includes('timeout')) {
      return {
        mode: 'frozen',
        confidence: 0.8,
        signals: [{
          type: 'thread_blocked',
          source: 'layer3',
          value: 'deep_check_timeout',
          timestamp: Date.now(),
          weight: 0.8
        }],
        timestamp: Date.now(),
        layer: 3
      };
    }

    return {
      mode: 'unknown',
      confidence: 0,
      signals: [{
        type: 'thread_blocked',
        source: 'layer3',
        value: error.message,
        timestamp: Date.now(),
        weight: 0
      }],
      timestamp: Date.now(),
      layer: 3
    };
  }
}
```
