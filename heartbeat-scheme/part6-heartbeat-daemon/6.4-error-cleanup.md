# 6.4 异常退出与资源清理

## 6.4.1 概述

HeartbeatDaemon 在运行过程中可能遇到各种异常情况，包括进程崩溃、网络断开、超时等。本节定义异常处理策略和资源清理机制，确保系统在各种情况下都能安全退出。

## 6.4.2 异常类型定义

### 6.4.2.1 异常分类

```typescript
type HeartbeatErrorType =
  | 'connection_lost'
  | 'page_crashed'
  | 'timeout_exceeded'
  | 'recovery_failed'
  | 'resource_exhausted'
  | 'internal_error'
  | 'user_abort';

interface HeartbeatError extends Error {
  type: HeartbeatErrorType;
  code: string;
  recoverable: boolean;
  context: ErrorContext;
  timestamp: number;
}

interface ErrorContext {
  taskId: string | null;
  daemonState: DaemonState;
  healthMode: HealthMode;
  lastCollectTime: number | null;
  recoveryAttempts: number;
  timerStatus: TimerStatus;
}
```

### 6.4.2.2 异常严重程度

| 严重程度 | 类型 | 处理策略 |
|---------|------|---------|
| `low` | 单次采集失败 | 重试，记录日志 |
| `medium` | 恢复动作失败 | 再次尝试恢复 |
| `high` | 连接断开 | 尝试重连 |
| `critical` | 页面崩溃、资源耗尽 | 立即停止，清理资源 |

## 6.4.3 错误处理类

### 6.4.3.1 ErrorHandler 定义

```typescript
class HeartbeatErrorHandler {
  private _daemon: HeartbeatDaemon;
  private _errorHistory: HeartbeatError[] = [];
  private _maxHistorySize = 100;
  private _retryCounters: Map<string, number> = new Map();

  constructor(daemon: HeartbeatDaemon) {
    this._daemon = daemon;
    this.setupGlobalHandlers();
  }

  private setupGlobalHandlers(): void {
    process.on('uncaughtException', this.handleUncaughtException.bind(this));
    process.on('unhandledRejection', this.handleUnhandledRejection.bind(this));
  }

  async handleError(error: HeartbeatError): Promise<ErrorHandlerResult> {
    this.recordError(error);

    switch (error.type) {
      case 'connection_lost':
        return this.handleConnectionLost(error);
      case 'page_crashed':
        return this.handlePageCrashed(error);
      case 'timeout_exceeded':
        return this.handleTimeoutExceeded(error);
      case 'recovery_failed':
        return this.handleRecoveryFailed(error);
      case 'resource_exhausted':
        return this.handleResourceExhausted(error);
      case 'internal_error':
        return this.handleInternalError(error);
      case 'user_abort':
        return this.handleUserAbort(error);
      default:
        return this.handleUnknownError(error);
    }
  }

  private recordError(error: HeartbeatError): void {
    this._errorHistory.push(error);
    if (this._errorHistory.length > this._maxHistorySize) {
      this._errorHistory.shift();
    }
  }
}

interface ErrorHandlerResult {
  handled: boolean;
  action: 'continue' | 'retry' | 'pause' | 'stop';
  message?: string;
}
```

### 6.4.3.2 连接丢失处理

```typescript
class HeartbeatErrorHandler {
  private async handleConnectionLost(error: HeartbeatError): Promise<ErrorHandlerResult> {
    const retryKey = 'connection_lost';
    const maxRetries = 5;
    const currentRetry = this._retryCounters.get(retryKey) || 0;

    if (currentRetry >= maxRetries) {
      return {
        handled: true,
        action: 'stop',
        message: `连接丢失，重试 ${maxRetries} 次后仍无法恢复`
      };
    }

    this._retryCounters.set(retryKey, currentRetry + 1);

    const delay = Math.min(1000 * Math.pow(2, currentRetry), 16000);
    await this.sleep(delay);

    try {
      await this._daemon.reconnect();
      this._retryCounters.delete(retryKey);
      
      return {
        handled: true,
        action: 'continue',
        message: '连接已恢复'
      };
    } catch (reconnectError) {
      return {
        handled: false,
        action: 'retry',
        message: `重连失败: ${(reconnectError as Error).message}`
      };
    }
  }

  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

```typescript
class HeartbeatErrorHandler {
  private async handlePageCrashed(error: HeartbeatError): Promise<ErrorHandlerResult> {
    await this._daemon.stop();

    const crashReport = await this.generateCrashReport(error);

    await this.saveCrashReport(crashReport);

    return {
      handled: true,
      action: 'stop',
      message: '页面崩溃，已生成崩溃报告'
    };
  }

  private async generateCrashReport(error: HeartbeatError): Promise<CrashReport> {
    const daemonState = this._daemon.daemonState;

    return {
      timestamp: Date.now(),
      error: {
        type: error.type,
        message: error.message,
        stack: error.stack
      },
      context: {
        taskId: daemonState.currentTaskId,
        healthMode: daemonState.currentHealthMode,
        startedAt: daemonState.startedAt,
        recoveryAttempts: daemonState.recoveryAttempts
      },
      errorHistory: this._errorHistory.slice(-10),
      systemInfo: await this.collectSystemInfo()
    };
  }

  private async saveCrashReport(report: CrashReport): Promise<void> {
    const reportPath = `crashes/${report.timestamp}.json`;
    await fs.mkdir('crashes', { recursive: true });
    await fs.writeFile(reportPath, JSON.stringify(report, null, 2));
  }
}
```

### 6.4.3.4 恢复失败处理

```typescript
class HeartbeatErrorHandler {
  private async handleRecoveryFailed(error: HeartbeatError): Promise<ErrorHandlerResult> {
    const recoveryAttempts = this._daemon.daemonState.recoveryAttempts;
    const maxAttempts = this._daemon.config.maxRecoveryRetries;

    if (recoveryAttempts >= maxAttempts) {
      await this._daemon.transitionTo('task-interrupted');

      return {
        handled: true,
        action: 'stop',
        message: `恢复失败次数达到上限 (${maxAttempts})，任务已中断`
      };
    }

    return {
      handled: false,
      action: 'retry',
      message: `恢复失败，尝试 ${recoveryAttempts}/${maxAttempts}`
    };
  }
}
```

## 6.4.4 资源清理机制

### 6.4.4.1 ResourceManager 定义

```typescript
class ResourceManager {
  private _resources: Set<DisposableResource> = new Set();
  private _cleanupOrder: string[] = [];
  private _cleanupTimeout = 5000;

  register(resource: DisposableResource): void {
    this._resources.add(resource);
  }

  unregister(resource: DisposableResource): void {
    this._resources.delete(resource);
  }

  async cleanup(): Promise<CleanupResult> {
    const results: CleanupItemResult[] = [];
    const startTime = Date.now();

    const sortedResources = this.sortByCleanupOrder();

    for (const resource of sortedResources) {
      const itemStartTime = Date.now();

      try {
        await Promise.race([
          resource.dispose(),
          this.timeout(this._cleanupTimeout)
        ]);

        results.push({
          name: resource.name,
          success: true,
          duration: Date.now() - itemStartTime
        });
      } catch (error) {
        results.push({
          name: resource.name,
          success: false,
          error: (error as Error).message,
          duration: Date.now() - itemStartTime
        });
      }

      this._resources.delete(resource);
    }

    return {
      totalDuration: Date.now() - startTime,
      items: results,
      allSucceeded: results.every(r => r.success)
    };
  }

  private sortByCleanupOrder(): DisposableResource[] {
    return Array.from(this._resources).sort((a, b) => {
      const orderA = this._cleanupOrder.indexOf(a.type);
      const orderB = this._cleanupOrder.indexOf(b.type);
      return orderA - orderB;
    });
  }

  private timeout(ms: number): Promise<never> {
    return new Promise((_, reject) => {
      setTimeout(() => reject(new Error('Cleanup timeout')), ms);
    });
  }
}

interface DisposableResource {
  name: string;
  type: string;
  dispose(): Promise<void>;
}

interface CleanupResult {
  totalDuration: number;
  items: CleanupItemResult[];
  allSucceeded: boolean;
}

interface CleanupItemResult {
  name: string;
  success: boolean;
  duration: number;
  error?: string;
}
```

## 6.4.5 优雅退出

### 6.4.5.1 GracefulShutdown 定义

```typescript
class GracefulShutdown {
  private _daemon: HeartbeatDaemon;
  private _resourceManager: ResourceManager;
  private _isShuttingDown = false;
  private _shutdownTimeout = 10000;

  constructor(daemon: HeartbeatDaemon, resourceManager: ResourceManager) {
    this._daemon = daemon;
    this._resourceManager = resourceManager;
    this.setupSignalHandlers();
  }

  private setupSignalHandlers(): void {
    process.on('SIGTERM', () => this.shutdown('SIGTERM'));
    process.on('SIGINT', () => this.shutdown('SIGINT'));
  }

  async shutdown(signal: string): Promise<void> {
    if (this._isShuttingDown) {
      console.log(`Already shutting down, ignoring ${signal}`);
      return;
    }

    this._isShuttingDown = true;
    console.log(`Received ${signal}, starting graceful shutdown...`);

    const startTime = Date.now();

    try {
      await Promise.race([
        this.performShutdown(),
        this.timeout(this._shutdownTimeout)
      ]);

      console.log(`Shutdown completed in ${Date.now() - startTime}ms`);
    } catch (error) {
      console.error('Shutdown error:', error);
    } finally {
      process.exit(0);
    }
  }

  private async performShutdown(): Promise<void> {
    if (this._daemon.state === 'running') {
      this._daemon.pause();
    }

    const summary = await this._daemon.stop();
    await this.saveFinalSummary(summary);

    const cleanupResult = await this._resourceManager.cleanup();
    console.log('Cleanup result:', cleanupResult);

    await this.flushPendingData();
  }

  private async saveFinalSummary(summary: DaemonSummary): Promise<void> {
    const summaryPath = `runs/${summary.taskId}_final.json`;
    await fs.writeFile(summaryPath, JSON.stringify(summary, null, 2));
  }

  private async flushPendingData(): Promise<void> {
    // 刷新任何待写入的数据
  }

  private timeout(ms: number): Promise<never> {
    return new Promise((_, reject) => {
      setTimeout(() => reject(new Error('Shutdown timeout')), ms);
    });
  }
}
```

## 6.4.6 状态恢复

### 6.4.6.1 状态检查点

```typescript
class StateCheckpoint {
  private _daemon: HeartbeatDaemon;
  private _checkpoints: Map<string, DaemonSnapshot> = new Map();
  private _checkpointInterval = 30000;
  private _maxCheckpoints = 10;

  constructor(daemon: HeartbeatDaemon) {
    this._daemon = daemon;
  }

  async createCheckpoint(): Promise<string> {
    const id = generateCheckpointId();
    const snapshot: DaemonSnapshot = {
      timestamp: Date.now(),
      taskId: this._daemon.daemonState.currentTaskId,
      healthMode: this._daemon.daemonState.currentHealthMode,
      startedAt: this._daemon.daemonState.startedAt,
      totalPausedDuration: this._daemon.daemonState.totalPausedDuration,
      recoveryAttempts: this._daemon.daemonState.recoveryAttempts,
      layer1State: this._daemon.getLayer1State(),
      layer2State: this._daemon.getLayer2State(),
      metrics: this._daemon.getMetrics()
    };

    this._checkpoints.set(id, snapshot);
    this.pruneOldCheckpoints();

    await this.persistCheckpoint(id, snapshot);

    return id;
  }

  async restoreFromCheckpoint(checkpointId: string): Promise<boolean> {
    const snapshot = this._checkpoints.get(checkpointId);
    if (!snapshot) {
      const loaded = await this.loadCheckpoint(checkpointId);
      if (!loaded) return false;
    }

    try {
      await this._daemon.restoreState(snapshot);
      return true;
    } catch (error) {
      console.error('Failed to restore from checkpoint:', error);
      return false;
    }
  }

  private pruneOldCheckpoints(): void {
    if (this._checkpoints.size > this._maxCheckpoints) {
      const oldest = Array.from(this._checkpoints.keys())[0];
      this._checkpoints.delete(oldest);
    }
  }

  private async persistCheckpoint(id: string, snapshot: DaemonSnapshot): Promise<void> {
    const checkpointPath = `checkpoints/${id}.json`;
    await fs.mkdir('checkpoints', { recursive: true });
    await fs.writeFile(checkpointPath, JSON.stringify(snapshot, null, 2));
  }

  private async loadCheckpoint(id: string): Promise<DaemonSnapshot | null> {
    try {
      const checkpointPath = `checkpoints/${id}.json`;
      const content = await fs.readFile(checkpointPath, 'utf-8');
      return JSON.parse(content);
    } catch {
      return null;
    }
  }
}

interface DaemonSnapshot {
  timestamp: number;
  taskId: string | null;
  healthMode: HealthMode;
  startedAt: number | null;
  totalPausedDuration: number;
  recoveryAttempts: number;
  layer1State: any;
  layer2State: any;
  metrics: any;
}
```

### 6.4.6.2 自动恢复

```typescript
class AutoRecovery {
  private _daemon: HeartbeatDaemon;
  private _checkpoint: StateCheckpoint;
  private _recoveryAttempts = 0;
  private _maxRecoveryAttempts = 3;

  constructor(daemon: HeartbeatDaemon, checkpoint: StateCheckpoint) {
    this._daemon = daemon;
    this._checkpoint = checkpoint;
  }

  async attemptRecovery(error: HeartbeatError): Promise<RecoveryResult> {
    if (this._recoveryAttempts >= this._maxRecoveryAttempts) {
      return {
        success: false,
        reason: 'max_attempts_exceeded'
      };
    }

    this._recoveryAttempts++;

    const lastCheckpoint = this.getLastCheckpoint();
    if (!lastCheckpoint) {
      return {
        success: false,
        reason: 'no_checkpoint_available'
      };
    }

    try {
      const restored = await this._checkpoint.restoreFromCheckpoint(lastCheckpoint);
      
      if (restored) {
        return {
          success: true,
          checkpointId: lastCheckpoint,
          attempt: this._recoveryAttempts
        };
      }

      return {
        success: false,
        reason: 'restore_failed'
      };
    } catch (err) {
      return {
        success: false,
        reason: 'restore_error',
        error: (err as Error).message
      };
    }
  }

  private getLastCheckpoint(): string | null {
    const checkpoints = Array.from(this._checkpoint.getCheckpoints().keys());
    return checkpoints.length > 0 ? checkpoints[checkpoints.length - 1] : null;
  }

  reset(): void {
    this._recoveryAttempts = 0;
  }
}

interface RecoveryResult {
  success: boolean;
  reason?: string;
  checkpointId?: string;
  attempt?: number;
  error?: string;
}
```

```typescript
class ErrorLogger {
  private _logPath: string;
  private _buffer: ErrorLogEntry[] = [];
  private _flushInterval = 5000;
  private _maxBufferSize = 100;

  constructor(logPath: string = 'logs/errors.jsonl') {
    this._logPath = logPath;
    this.startFlushTimer();
  }

  log(error: HeartbeatError, context: ErrorContext): void {
    const entry: ErrorLogEntry = {
      timestamp: Date.now(),
      type: error.type,
      code: error.code,
      message: error.message,
      stack: error.stack,
      recoverable: error.recoverable,
      context: {
        taskId: context.taskId,
        daemonState: context.daemonState,
        healthMode: context.healthMode,
        recoveryAttempts: context.recoveryAttempts
      }
    };

    this._buffer.push(entry);

    if (this._buffer.length >= this._maxBufferSize) {
      this.flush();
    }
  }

  private startFlushTimer(): void {
    setInterval(() => this.flush(), this._flushInterval);
  }

  async flush(): Promise<void> {
    if (this._buffer.length === 0) return;

    const entries = [...this._buffer];
    this._buffer = [];

    const lines = entries.map(e => JSON.stringify(e)).join('\n') + '\n';
    
    await fs.appendFile(this._logPath, lines);
  }
}

interface ErrorLogEntry {
  timestamp: number;
  type: HeartbeatErrorType;
  code: string;
  message: string;
  stack?: string;
  recoverable: boolean;
  context: {
    taskId: string | null;
    daemonState: DaemonState;
    healthMode: HealthMode;
    recoveryAttempts: number;
  };
}
```

```typescript
class DiagnosticCollector {
  async collect(): Promise<DiagnosticInfo> {
    return {
      timestamp: Date.now(),
      system: await this.collectSystemInfo(),
      process: this.collectProcessInfo(),
      memory: this.collectMemoryInfo(),
      network: await this.collectNetworkInfo(),
      daemon: this.collectDaemonInfo()
    };
  }

  private async collectSystemInfo(): Promise<SystemDiagnostics> {
    return {
      platform: process.platform,
      arch: process.arch,
      nodeVersion: process.version,
      hostname: os.hostname(),
      uptime: os.uptime(),
      loadAverage: os.loadavg()
    };
  }

  private collectProcessInfo(): ProcessDiagnostics {
    return {
      pid: process.pid,
      ppid: process.ppid,
      uptime: process.uptime(),
      cpuUsage: process.cpuUsage(),
      memoryUsage: process.memoryUsage()
    };
  }

  private collectMemoryInfo(): MemoryDiagnostics {
    const usage = process.memoryUsage();
    return {
      heapTotal: usage.heapTotal,
      heapUsed: usage.heapUsed,
      external: usage.external,
      rss: usage.rss,
      arrayBuffers: usage.arrayBuffers
    };
  }

  private async collectNetworkInfo(): Promise<NetworkDiagnostics> {

    return {
      interfaces: Object.keys(interfaces),

      healthMode: this._daemon.getCurrentHealth(),
      uptime: this._daemon.getUptime(),
      timerStatus: this._daemon.getTimerStatus()
    };
  }

      return true;
    } catch {
      return false;
    }
  }
}
```
