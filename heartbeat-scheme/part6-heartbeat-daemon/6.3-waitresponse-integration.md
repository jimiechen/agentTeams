# 6.3 与 waitResponse 的集成点

## 6.3.1 概述

`waitResponse` 是 mvp-runner 中等待 AI 响应完成的核心函数。本节定义 HeartbeatDaemon 与 `waitResponse` 的集成方案，实现心跳监控与响应等待的协同工作。

## 6.3.2 现有 waitResponse 分析

### 6.3.2.1 当前实现

```typescript
export async function waitResponse(cdp: CDPClient, opts?: WaitResponseOptions): Promise<string> {
  const timeoutMs = opts?.timeoutMs ?? 300000;
  const pollMs = opts?.pollMs ?? 2000;
  const taskName = opts?.taskName ?? 'PMCLI';

  const startTime = Date.now();
  const snapshots: SystemSnapshot[] = [];
  let checkCount = 0;

  while (Date.now() - startTime < timeoutMs) {
    checkCount++;
    const snap = await captureSnapshot(cdp);
    snapshots.push(snap);

    if (snapshots.length > 100) {
      snapshots.shift();
    }

    if (isTaskCompleted(snap)) {
      return parseResult(cdp);
    }

    if (isModelStalled(snap)) {
      await recoverModelStalled(cdp, policy);
    }

    await sleep(pollMs);
  }

  throw new ResponseTimeoutError(taskName, elapsed);
}
```

### 6.3.2.2 集成痛点

| 问题 | 描述 | 影响 |
|------|------|------|
| 重复检测 | waitResponse 和 HeartbeatDaemon 都在做状态检测 | 资源浪费 |
| 状态不一致 | 两个系统的状态判断可能不同步 | 恢复动作冲突 |
| 心跳日志分散 | waitResponse 的日志与心跳日志分离 | 难以追踪 |
| 恢复策略分散 | 恢复逻辑在两处实现 | 维护困难 |

## 6.3.5 改造后的 waitResponse

### 6.3.5.1 主函数实现

```typescript
export async function waitResponse(
  cdp: CDPClient,
  opts?: WaitResponseOptions
): Promise<string> {
  const timeoutMs = opts?.timeoutMs ?? 300000;
  const taskName = opts?.taskName ?? 'PMCLI';
  const heartbeatEnabled = opts?.heartbeat?.enabled ?? true;

  const taskId = generateTaskId();
  const startTime = Date.now();

  const context: HeartbeatContext = {
    taskId,
    startTime,
    options: opts || {},
    abortSignal: opts?.abortSignal
  };

  const daemon = opts?.heartbeat?.daemon || new HeartbeatDaemon(
    opts?.heartbeat?.config
  );

  const eventEmitter = new EventEmitter();

  try {
    if (heartbeatEnabled) {
      await daemon.start(taskId, cdp.page);
      setupDaemonEventHandlers(daemon, eventEmitter, context);
    }

    const result = await waitForCompletion(
      cdp,
      daemon,
      context,
      timeoutMs,
      eventEmitter
    );

    eventEmitter.emit('wait:completed', {
      taskId,
      timestamp: Date.now(),
      result
    });

    return result;

  } catch (error) {
    if (error instanceof ResponseTimeoutError) {
      eventEmitter.emit('wait:timeout', {
        taskId,
        timestamp: Date.now(),
        error
      });
    }

    throw error;

  } finally {
    if (heartbeatEnabled) {
      const summary = await daemon.stop();
      await writeHeartbeatSummary(taskId, summary);
    }
  }
}
```

### 6.3.5.2 等待完成逻辑

```typescript
async function waitForCompletion(
  cdp: CDPClient,
  daemon: HeartbeatDaemon,
  context: HeartbeatContext,
  timeoutMs: number,
  eventEmitter: EventEmitter
): Promise<string> {
  return new Promise((resolve, reject) => {
    const timeoutId = setTimeout(() => {
      reject(new ResponseTimeoutError(
        context.options.taskName || 'PMCLI',
        Date.now() - context.startTime
      ));
    }, timeoutMs);

    const abortHandler = () => {
      clearTimeout(timeoutId);
      reject(new Error('Aborted by user'));
    };
    context.abortSignal?.addEventListener('abort', abortHandler);

    const completionHandler = async (data: { completed: boolean; result?: string }) => {
      if (data.completed) {
        clearTimeout(timeoutId);
        context.abortSignal?.removeEventListener('abort', abortHandler);
        
        try {
          const result = data.result || await parseLastAIResponse(cdp);
          resolve(result);
        } catch (error) {
          reject(error);
        }
      }
    };

    daemon.on('task:completed', completionHandler);
    daemon.on('task:interrupted', (data) => {
      clearTimeout(timeoutId);
      reject(new TaskInterruptedError(data.reason));
    });
  });
}
```

### 6.3.5.3 事件处理设置

```typescript
function setupDaemonEventHandlers(
  daemon: HeartbeatDaemon,
  eventEmitter: EventEmitter,
  context: HeartbeatContext
): void {
  daemon.on('health:changed', (data) => {
    eventEmitter.emit('heartbeat:health_changed', {
      taskId: context.taskId,
      timestamp: Date.now(),
      healthMode: data.newMode,
      previousMode: data.previousMode
    });

    if (data.newMode === 'model-queuing') {
      handleQueuingMode(daemon, context);
    }
  });

  daemon.on('recovery:started', (data) => {
    eventEmitter.emit('heartbeat:recovery_started', {
      taskId: context.taskId,
      timestamp: Date.now(),
      recoveryType: data.type
    });
  });

  daemon.on('recovery:completed', (data) => {
    eventEmitter.emit('heartbeat:recovery_completed', {
      taskId: context.taskId,
      timestamp: Date.now(),
      recoveryType: data.type,
      success: data.success
    });
  });

  daemon.on('recovery:failed', (data) => {
    eventEmitter.emit('heartbeat:recovery_failed', {
      taskId: context.taskId,
      timestamp: Date.now(),
      recoveryType: data.type,
      error: data.error
    });
  });
}
```

## 6.3.6 状态同步机制

### 6.3.6.1 完成状态检测

```typescript
class CompletionDetector {
  private _daemon: HeartbeatDaemon;
  private _completionCallbacks: Set<(result: CompletionResult) => void> = new Set();

  constructor(daemon: HeartbeatDaemon) {
    this._daemon = daemon;
    this.setupHealthListener();
  }

  private setupHealthListener(): void {
    this._daemon.on('health:changed', (data) => {
      if (data.newMode === 'completed') {
        this.notifyCompletion({
          completed: true,
          reason: 'task_completed'
        });
      } else if (data.newMode === 'task-interrupted') {
        this.notifyCompletion({
          completed: false,
          reason: 'task_interrupted'
        });
      }
    });
  }

  onCompletion(callback: (result: CompletionResult) => void): void {
    this._completionCallbacks.add(callback);
  }

  offCompletion(callback: (result: CompletionResult) => void): void {
    this._completionCallbacks.delete(callback);
  }

  private notifyCompletion(result: CompletionResult): void {
    for (const callback of this._completionCallbacks) {
      try {
        callback(result);
      } catch (error) {
        console.error('Error in completion callback:', error);
      }
    }
  }
}

interface CompletionResult {
  completed: boolean;
  reason: 'task_completed' | 'task_interrupted' | 'timeout' | 'error';
  result?: string;
}
```

```typescript
  context: HeartbeatContext
```

## 6.3.7 超时与心跳协调

### 6.3.7.1 动态超时调整

```typescript
class TimeoutManager {
  private _baseTimeout: number;
  private _currentTimeout: number;
  private _daemon: HeartbeatDaemon;
  private _extensions: number = 0;
  private _maxExtensions: number = 3;

  constructor(baseTimeout: number, daemon: HeartbeatDaemon) {
    this._baseTimeout = baseTimeout;
    this._currentTimeout = baseTimeout;
    this._daemon = daemon;

    this.setupHealthListener();
  }

  private setupHealthListener(): void {
    this._daemon.on('health:changed', (data) => {
      if (data.newMode === 'model-queuing') {
        this.extendTimeout();
      }
    });
  }

  private extendTimeout(): void {
    if (this._extensions >= this._maxExtensions) {
      return;
    }

    const extension = Math.min(
      this._baseTimeout * 0.5,
      60000
    );

    this._currentTimeout += extension;
    this._extensions++;

    this._daemon.emit('timeout:extended', {
      newTimeout: this._currentTimeout,
      extensions: this._extensions
    });
  }

  get currentTimeout(): number {
    return this._currentTimeout;
  }

  reset(): void {
    this._currentTimeout = this._baseTimeout;
    this._extensions = 0;
  }
}
```

### 6.3.7.2 超时预警

```typescript
class TimeoutWarningManager {
  private _warningThresholds = [0.5, 0.75, 0.9];
  private _triggeredWarnings: Set<number> = new Set();
  private _daemon: HeartbeatDaemon;

  constructor(daemon: HeartbeatDaemon) {
    this._daemon = daemon;
  }

  check(elapsed: number, timeout: number): void {
    const ratio = elapsed / timeout;

    for (const threshold of this._warningThresholds) {
      if (ratio >= threshold && !this._triggeredWarnings.has(threshold)) {
        this._triggeredWarnings.add(threshold);
        this._daemon.emit('timeout:warning', {
          threshold,
          elapsed,
          remaining: timeout - elapsed
        });
      }
    }
  }

  reset(): void {
    this._triggeredWarnings.clear();
  }
}
```

### 6.3.8.1 统一日志格式

```typescript
interface HeartbeatLogEntry {
  timestamp: number;
  taskId: string;
  source: 'waitResponse' | 'heartbeat' | 'recovery';
  level: 'debug' | 'info' | 'warn' | 'error';
  event: string;
  data: Record<string, any>;
}

class HeartbeatLogger {
  private _taskId: string;
  private _logger: WorkspaceLogger;
  private _entries: HeartbeatLogEntry[] = [];

  constructor(taskId: string, logger?: WorkspaceLogger) {
    this._taskId = taskId;
    this._logger = logger || console;
  }

  log(
    source: HeartbeatLogEntry['source'],
    level: HeartbeatLogEntry['level'],
    event: string,
    data: Record<string, any> = {}
  ): void {
    const entry: HeartbeatLogEntry = {
      timestamp: Date.now(),
      taskId: this._taskId,
      source,
      level,
      event,
      data
    };

    this._entries.push(entry);

    const message = `[${source}] ${event}`;
    switch (level) {
      case 'debug':
        this._logger.debug?.(message, data);
        break;
      case 'info':
        this._logger.info?.(message, data);
        break;
      case 'warn':
        this._logger.warn?.(message, data);
        break;
      case 'error':
        this._logger.error?.(message, data);
        break;
    }
  }

  getEntries(): HeartbeatLogEntry[] {
    return [...this._entries];
  }

  flush(): HeartbeatLogEntry[] {
    const entries = this.getEntries();
    this._entries = [];
    return entries;
  }
}
```

```typescript
export async function waitResponse(
  cdp: CDPClient,
  opts?: WaitResponseOptions
): Promise<string> {
  const heartbeatEnabled = opts?.heartbeat?.enabled ?? true;

  if (!heartbeatEnabled) {
    return waitResponseLegacy(cdp, opts);
  }

  return waitResponseWithHeartbeat(cdp, opts);
}

async function waitResponseLegacy(
  cdp: CDPClient,
  opts?: WaitResponseOptions
): Promise<string> {
  const timeoutMs = opts?.timeoutMs ?? 300000;
  const pollMs = opts?.pollMs ?? 2000;
  const taskName = opts?.taskName ?? 'PMCLI';

  const startTime = Date.now();

  while (Date.now() - startTime < timeoutMs) {
    const snap = await captureSnapshot(cdp);

    if (isTaskCompleted(snap)) {
      return parseLastAIResponse(cdp);
    }

    if (isModelStalled(snap)) {
      await recoverModelStalled(cdp, opts?.recoveryPolicy);
    }

    await sleep(pollMs);
  }

  throw new ResponseTimeoutError(taskName, Date.now() - startTime);
}
```
