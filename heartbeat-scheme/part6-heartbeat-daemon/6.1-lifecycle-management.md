# 6.1 HeartbeatDaemon 的生命周期管理

## 6.1.1 概述

HeartbeatDaemon 是三层心跳架构的核心调度器，负责协调 Layer 1/2/3 的采集任务、状态机管理和恢复动作执行。本节定义 Daemon 的生命周期状态机及其管理策略。

## 6.1.2 Daemon 状态机定义

### 状态枚举

```typescript
type DaemonState = 'stopped' | 'running' | 'paused';
```

### 状态说明

| 状态 | 含义 | Timer 状态 | 可接受事件 |
|------|------|-----------|-----------|
| `stopped` | Daemon 已停止，所有 Timer 已清除 | 全部停止 | `START` |
| `running` | Daemon 正常运行，三层 Timer 活跃 | 全部运行 | `PAUSE`, `STOP` |
| `paused` | Daemon 暂停，Timer 保留但未触发 | 全部暂停 | `RESUME`, `STOP` |

## 6.1.3 状态转移图

```
                    START
    ┌─────────────────────────────────┐
    │                                 │
    ▼                                 │
┌─────────┐  PAUSE  ┌─────────┐       │
│ stopped │ ──────► │ running │       │
└─────────┘         └────┬────┘       │
    ▲                    │            │
    │                    │ PAUSE      │
    │                    ▼            │
    │              ┌─────────┐        │
    │   STOP       │ paused  │        │
    │ ┌────────────┴─────────┴──────┐ │
    │ │            RESUME           │ │
    │ └─────────────────────────────┘ │
    │                                 │
    └─────────────────────────────────┘
                    STOP
```

## 6.1.4 状态转移矩阵

| 当前状态 | 事件 | 目标状态 | 动作 |
|---------|------|---------|------|
| `stopped` | `START` | `running` | 初始化资源，启动三层 Timer |
| `running` | `PAUSE` | `paused` | 暂停所有 Timer，保留状态 |
| `running` | `STOP` | `stopped` | 停止所有 Timer，清理资源 |
| `paused` | `RESUME` | `running` | 恢复所有 Timer |
| `paused` | `STOP` | `stopped` | 清除暂停的 Timer，清理资源 |

## 6.1.5 HeartbeatDaemon 类定义

```typescript
interface HeartbeatDaemonConfig {
  layer1Interval: number;
  layer2Interval: number;
  layer3Interval: number;

interface HeartbeatDaemonState {

class HeartbeatDaemon {
  private _state: DaemonState = 'stopped';
  private _daemonState: HeartbeatDaemonState;
  private _config: HeartbeatDaemonConfig;
  
  private _layer1Timer: ReturnType<typeof setInterval> | null = null;
  private _layer2Timer: ReturnType<typeof setInterval> | null = null;
  private _layer3Timer: ReturnType<typeof setInterval> | null = null;
  
  private _layer1Collector: Layer1Collector;
  private _layer2Collector: Layer2Collector;
  private _layer3Collector: Layer3Collector;
  
  private _stateMachine: HealthStateMachine;
  private _recoveryManager: RecoveryManager;
  private _metricsCollector: MetricsCollector;
  private _persistenceManager: PersistenceManager;
  
  private _pausedTimerStates: {
    layer1Remaining: number;
    layer2Remaining: number;
    layer3Remaining: number;
  } | null = null;

  constructor(config: Partial<HeartbeatDaemonConfig> = {}) {
    this._config = {
      layer1Interval: 5000,
      layer2Interval: 15000,
      layer3Interval: 60000,
      maxRecoveryRetries: 3,
      metricsEnabled: true,
      persistenceEnabled: true,
      persistencePath: './heartbeat.jsonl',
      ...config
    };

    this._daemonState = {
      state: 'stopped',
      currentTaskId: null,
      currentHealthMode: 'healthy',
      startedAt: null,
      pausedAt: null,
      totalPausedDuration: 0,
      recoveryAttempts: 0,
      lastCollectTime: null
    };

    this._layer1Collector = new Layer1Collector();
    this._layer2Collector = new Layer2Collector();
    this._layer3Collector = new Layer3Collector();
    this._stateMachine = new HealthStateMachine();
    this._recoveryManager = new RecoveryManager();
    this._metricsCollector = new MetricsCollector();
    this._persistenceManager = new PersistenceManager(this._config.persistencePath);
  }

    return this._state;
  }

  get daemonState(): Readonly<HeartbeatDaemonState> {
    return { ...this._daemonState };
  }
}
```

## 6.1.6 START 事件处理

```typescript
class HeartbeatDaemon {
  async start(taskId: string, page: Page): Promise<void> {
    if (this._state !== 'stopped') {
      throw new Error(`Cannot start daemon in state: ${this._state}`);
    }

    this._daemonState.currentTaskId = taskId;
    this._daemonState.startedAt = Date.now();
    this._daemonState.state = 'running';

    await this.initializeCollectors(page);
    this.startAllTimers(page);

    this._state = 'running';
    
    this.emit('daemon:started', {
      taskId,
      timestamp: this._daemonState.startedAt
    });

    this._persistenceManager.append({
      type: 'daemon_started',
      taskId,
      timestamp: this._daemonState.startedAt,
      config: this._config
    });
  }

  private async initializeCollectors(page: Page): Promise<void> {
    await Promise.all([
      this._layer1Collector.initialize(page),
      this._layer2Collector.initialize(page),
      this._layer3Collector.initialize(page)
    ]);
  }

  private startAllTimers(page: Page): void {
    this._layer1Timer = setInterval(
      () => this.collectLayer1(page),
      this._config.layer1Interval
    );

    this._layer2Timer = setInterval(
      () => this.collectLayer2(page),
      this._config.layer2Interval
    );

    this._layer3Timer = setInterval(
      () => this.collectLayer3(page),
      this._config.layer3Interval
    );
  }
}
```

## 6.1.7 PAUSE 事件处理

```typescript
class HeartbeatDaemon {
  pause(): void {
    if (this._state !== 'running') {
      throw new Error(`Cannot pause daemon in state: ${this._state}`);
    }

    this._pausedTimerStates = this.captureTimerStates();
    this.stopAllTimers();

    this._daemonState.pausedAt = Date.now();
    this._daemonState.state = 'paused';
    this._state = 'paused';

    this.emit('daemon:paused', {
      taskId: this._daemonState.currentTaskId,
      timestamp: this._daemonState.pausedAt
    });

    this._persistenceManager.append({
      type: 'daemon_paused',
      taskId: this._daemonState.currentTaskId,
      timestamp: this._daemonState.pausedAt
    });
  }

  private captureTimerStates(): {
    layer1Remaining: number;
    layer2Remaining: number;
    layer3Remaining: number;
  } {
    const now = Date.now();
    const startedAt = this._daemonState.startedAt || now;
    const totalPaused = this._daemonState.totalPausedDuration;

    const effectiveElapsed = now - startedAt - totalPaused;

    return {
      layer1Remaining: this._config.layer1Interval - (effectiveElapsed % this._config.layer1Interval),
      layer2Remaining: this._config.layer2Interval - (effectiveElapsed % this._config.layer2Interval),
      layer3Remaining: this._config.layer3Interval - (effectiveElapsed % this._config.layer3Interval)
    };
  }

  private stopAllTimers(): void {
    if (this._layer1Timer) {
      clearInterval(this._layer1Timer);
      this._layer1Timer = null;
    }
    if (this._layer2Timer) {
      clearInterval(this._layer2Timer);
      this._layer2Timer = null;
    }
    if (this._layer3Timer) {
      clearInterval(this._layer3Timer);
      this._layer3Timer = null;
    }
  }
}
```

## 6.1.8 RESUME 事件处理

```typescript
class HeartbeatDaemon {
  async resume(page: Page): Promise<void> {
    if (this._state !== 'paused') {
      throw new Error(`Cannot resume daemon in state: ${this._state}`);
    }

    const pausedDuration = Date.now() - (this._daemonState.pausedAt || 0);
    this._daemonState.totalPausedDuration += pausedDuration;
    this._daemonState.pausedAt = null;
    this._daemonState.state = 'running';

    await this.restoreTimersWithRemaining(page);

    this._state = 'running';

    this.emit('daemon:resumed', {
      taskId: this._daemonState.currentTaskId,
      timestamp: Date.now(),
      pausedDuration
    });

    this._persistenceManager.append({
      type: 'daemon_resumed',
      taskId: this._daemonState.currentTaskId,
      timestamp: Date.now(),
      pausedDuration
    });
  }

  private async restoreTimersWithRemaining(page: Page): Promise<void> {
    if (!this._pausedTimerStates) {
      this.startAllTimers(page);
      return;
    }

    const { layer1Remaining, layer2Remaining, layer3Remaining } = this._pausedTimerStates;

    setTimeout(() => {
      this.collectLayer1(page);
      this._layer1Timer = setInterval(
        () => this.collectLayer1(page),
        this._config.layer1Interval
      );
    }, layer1Remaining);

    setTimeout(() => {
      this.collectLayer2(page);
      this._layer2Timer = setInterval(
        () => this.collectLayer2(page),
        this._config.layer2Interval
      );
    }, layer2Remaining);

    setTimeout(() => {
      this.collectLayer3(page);
      this._layer3Timer = setInterval(
        () => this.collectLayer3(page),
        this._config.layer3Interval
      );
    }, layer3Remaining);

    this._pausedTimerStates = null;
  }
}
```

## 6.1.9 STOP 事件处理

```typescript
class HeartbeatDaemon {
  async stop(): Promise<DaemonSummary> {
    if (this._state === 'stopped') {
      throw new Error('Daemon is already stopped');
    }

    this.stopAllTimers();
    this._pausedTimerStates = null;

    const summary = await this.generateSummary();

    await this.cleanup();

    this._state = 'stopped';
    this._daemonState.state = 'stopped';

    this.emit('daemon:stopped', {
      taskId: this._daemonState.currentTaskId,
      timestamp: Date.now(),
      summary
    });

    this._persistenceManager.append({
      type: 'daemon_stopped',
      taskId: this._daemonState.currentTaskId,
      timestamp: Date.now(),
      summary
    });

    return summary;
  }

  private async generateSummary(): Promise<DaemonSummary> {
    const metrics = this._metricsCollector.getMetrics();
    const stateHistory = this._stateMachine.getHistory();

    return {
      taskId: this._daemonState.currentTaskId,
      startedAt: this._daemonState.startedAt,
      stoppedAt: Date.now(),
      totalDuration: Date.now() - (this._daemonState.startedAt || 0),
      totalPausedDuration: this._daemonState.totalPausedDuration,
      effectiveDuration: Date.now() - (this._daemonState.startedAt || 0) - this._daemonState.totalPausedDuration,
      finalHealthMode: this._daemonState.currentHealthMode,
      recoveryAttempts: this._daemonState.recoveryAttempts,
      metrics,
      stateTransitions: stateHistory.length
    };
  }

  private async cleanup(): Promise<void> {
    await this._layer1Collector.cleanup();
    await this._layer2Collector.cleanup();
    await this._layer3Collector.cleanup();
    
    this._stateMachine.reset();
    this._recoveryManager.reset();
    
    if (this._config.metricsEnabled) {
      await this._metricsCollector.flush();
    }
  }
}

interface DaemonSummary {
  taskId: string | null;
  startedAt: number | null;
  stoppedAt: number;
  totalDuration: number;
  totalPausedDuration: number;
  effectiveDuration: number;
  finalHealthMode: HealthMode;
  recoveryAttempts: number;
  metrics: LayerMetrics;
  stateTransitions: number;
}
```

## 6.1.10 状态查询接口

```typescript
class HeartbeatDaemon {
  isRunning(): boolean {
    return this._state === 'running';
  }

  isPaused(): boolean {
    return this._state === 'paused';
  }

  isStopped(): boolean {
    return this._state === 'stopped';
  }

  getUptime(): number {
    if (!this._daemonState.startedAt) return 0;
    return Date.now() - this._daemonState.startedAt - this._daemonState.totalPausedDuration;
  }

  getEffectiveUptime(): number {
    return this.getUptime();
  }

  getCurrentHealth(): HealthMode {
    return this._daemonState.currentHealthMode;
  }

  getTimerStatus(): TimerStatus {
    return {
      layer1: {
        active: this._layer1Timer !== null,
        interval: this._config.layer1Interval
      },
      layer2: {
        active: this._layer2Timer !== null,
        interval: this._config.layer2Interval
      },
      layer3: {
        active: this._layer3Timer !== null,
        interval: this._config.layer3Interval
      }
    };
  }
}

interface TimerStatus {
  layer1: { active: boolean; interval: number };
  layer2: { active: boolean; interval: number };
  layer3: { active: boolean; interval: number };
}
```

## 6.1.11 事件系统

```typescript
type DaemonEvent = 
  | 'daemon:started'
  | 'daemon:paused'
  | 'daemon:resumed'
  | 'daemon:stopped'
  | 'daemon:error'
  | 'health:changed'
  | 'recovery:started'
  | 'recovery:completed'
  | 'recovery:failed';

interface DaemonEventData {
  [key: string]: any;
}

class HeartbeatDaemon extends EventEmitter {
  private _eventHandlers: Map<DaemonEvent, Set<EventHandler>> = new Map();

  on(event: DaemonEvent, handler: EventHandler): void {
    if (!this._eventHandlers.has(event)) {
      this._eventHandlers.set(event, new Set());
    }
    this._eventHandlers.get(event)!.add(handler);
  }

  off(event: DaemonEvent, handler: EventHandler): void {
    this._eventHandlers.get(event)?.delete(handler);
  }

  protected emit(event: DaemonEvent, data: DaemonEventData): void {
    const handlers = this._eventHandlers.get(event);
    if (handlers) {
      for (const handler of handlers) {
        try {
          handler(data);
        } catch (error) {
          console.error(`Error in event handler for ${event}:`, error);
        }
      }
    }
  }
}

type EventHandler = (data: DaemonEventData) => void | Promise<void>;
```

## 6.1.12 状态持久化

```typescript
interface DaemonPersistedState {
  version: string;
  taskId: string;
  state: DaemonState;
  startedAt: number;
  pausedAt: number | null;
  totalPausedDuration: number;
  healthMode: HealthMode;
  layer1State: Layer1State;
  layer2State: Layer2State;
  recoveryAttempts: number;
}

class HeartbeatDaemon {
  async persistState(): Promise<void> {
    const state: DaemonPersistedState = {
      version: '1.0.0',
      taskId: this._daemonState.currentTaskId || '',
      state: this._state,
      startedAt: this._daemonState.startedAt || 0,
      pausedAt: this._daemonState.pausedAt,
      totalPausedDuration: this._daemonState.totalPausedDuration,
      healthMode: this._daemonState.currentHealthMode,
      layer1State: this._layer1Collector.getState(),
      layer2State: this._layer2Collector.getState(),
      recoveryAttempts: this._daemonState.recoveryAttempts
    };

    await this._persistenceManager.saveState(state);
  }

  async restoreState(state: DaemonPersistedState, page: Page): Promise<void> {
    this._daemonState.currentTaskId = state.taskId;
    this._daemonState.startedAt = state.startedAt;
    this._daemonState.pausedAt = state.pausedAt;
    this._daemonState.totalPausedDuration = state.totalPausedDuration;
    this._daemonState.currentHealthMode = state.healthMode;
    this._daemonState.recoveryAttempts = state.recoveryAttempts;

    this._layer1Collector.restoreState(state.layer1State);
    this._layer2Collector.restoreState(state.layer2State);

    if (state.state === 'running') {
      await this.start(state.taskId, page);
    } else if (state.state === 'paused') {
      await this.start(state.taskId, page);
      this.pause();
    }
  }
}
```

## 6.1.14 测试用例

```typescript
describe('HeartbeatDaemon Lifecycle', () => {
  it('应正确启动 Daemon', async () => {
    const daemon = new HeartbeatDaemon();
    const page = createMockPage();
    
    await daemon.start('task-123', page);
    
    expect(daemon.state).toBe('running');
    expect(daemon.isRunning()).toBe(true);
    expect(daemon.getTimerStatus().layer1.active).toBe(true);
  });

  it('应正确暂停 Daemon', async () => {
    const daemon = new HeartbeatDaemon();
    const page = createMockPage();
    
    await daemon.start('task-123', page);
    daemon.pause();
    
    expect(daemon.state).toBe('paused');
    expect(daemon.isPaused()).toBe(true);
    expect(daemon.getTimerStatus().layer1.active).toBe(false);
  });

  it('应正确恢复 Daemon', async () => {
    const daemon = new HeartbeatDaemon();
    const page = createMockPage();
    
    await daemon.start('task-123', page);
    daemon.pause();
    await daemon.resume(page);
    
    expect(daemon.state).toBe('running');
    expect(daemon.isRunning()).toBe(true);
  });

  it('应正确停止 Daemon', async () => {
    const daemon = new HeartbeatDaemon();
    const page = createMockPage();
    
    await daemon.start('task-123', page);
    const summary = await daemon.stop();
    
    expect(daemon.state).toBe('stopped');
    expect(daemon.isStopped()).toBe(true);
    expect(summary.taskId).toBe('task-123');
  });

  it('应正确计算有效运行时间', async () => {
    const daemon = new HeartbeatDaemon();
    const page = createMockPage();
    
    await daemon.start('task-123', page);
    await sleep(1000);
    daemon.pause();
    await sleep(500);
    await daemon.resume(page);
    await sleep(500);
    
    const uptime = daemon.getUptime();
    expect(uptime).toBeGreaterThanOrEqual(1500);
    expect(uptime).toBeLessThan(2000);
  });

  it('应拒绝非法状态转换', async () => {
    const daemon = new HeartbeatDaemon();
    
    await expect(daemon.pause()).rejects.toThrow('Cannot pause daemon in state: stopped');
    await expect(daemon.resume(createMockPage())).rejects.toThrow('Cannot resume daemon in state: stopped');
  });

  it('应正确发送生命周期事件', async () => {
    const daemon = new HeartbeatDaemon();
    const page = createMockPage();
    const events: string[] = [];
    
    daemon.on('daemon:started', () => events.push('started'));
    daemon.on('daemon:paused', () => events.push('paused'));
    daemon.on('daemon:resumed', () => events.push('resumed'));
    daemon.on('daemon:stopped', () => events.push('stopped'));
    
    await daemon.start('task-123', page);
    daemon.pause();
    await daemon.resume(page);
    await daemon.stop();
    
    expect(events).toEqual(['started', 'paused', 'resumed', 'stopped']);
  });
});
```
