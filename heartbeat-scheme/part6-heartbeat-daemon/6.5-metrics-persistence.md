# 6.5 metrics 持久化

## 6.5.1 概述

HeartbeatDaemon 在运行过程中产生大量监控数据，包括三层采集结果、健康状态变化、恢复动作记录等。本节定义 metrics 的持久化策略和 `heartbeat.jsonl` 的写入规范。

## 6.5.2 持久化数据类型

### 6.5.2.1 数据分类

| 类型 | 来源 | 写入频率 | 保留策略 |
|------|------|---------|---------|
| 心跳事件 | Layer 1/2/3 采集 | 每次采集 | 完整保留 |
| 状态变更 | HealthStateMachine | 每次变更 | 完整保留 |
| 恢复记录 | RecoveryManager | 每次恢复 | 完整保留 |
| 错误日志 | ErrorHandler | 每次错误 | 完整保留 |
| 摘要报告 | Daemon 停止时 | 任务结束 | 完整保留 |

## 6.5.4 PersistenceManager 实现

### 6.5.4.1 类定义

```typescript
class PersistenceManager {
  private _filePath: string;
  private _fileHandle: fs.FileHandle | null = null;
  private _writeBuffer: HeartbeatJsonlRecord[] = [];
  private _flushInterval = 1000;
  private _maxBufferSize = 100;
  private _sequenceNumber = 0;
  private _metadata: PersistenceMetadata;
  private _flushTimer: ReturnType<typeof setInterval> | null = null;

  constructor(filePath: string = 'heartbeat.jsonl') {
    this._filePath = filePath;
    this._metadata = {
      version: '1.0.0',
      createdAt: Date.now(),
      taskId: null
    };
  }

  async initialize(taskId: string): Promise<void> {
    this._metadata.taskId = taskId;
    
    const dir = path.dirname(this._filePath);
    await fs.mkdir(dir, { recursive: true });
    
    this._fileHandle = await fs.open(this._filePath, 'a');
    
    this.startFlushTimer();
  }

  private startFlushTimer(): void {
    this._flushTimer = setInterval(() => {
      this.flush();
    }, this._flushInterval);
  }
}

interface PersistenceMetadata {
  version: string;
  createdAt: number;
  taskId: string | null;
}
```

### 6.5.4.2 写入方法

```typescript
class PersistenceManager {
  append(event: PersistenceEvent): void {
    const record: HeartbeatJsonlRecord = {
      seq: ++this._sequenceNumber,
      ts: event.timestamp || Date.now(),
      taskId: event.taskId,
      type: event.type,
      layer: this.extractLayer(event.type),
      payload: event.data,
      meta: {
        version: this._metadata.version,
        daemonState: event.data.daemonState,
        healthMode: event.data.healthMode
      }
    };

    this._writeBuffer.push(record);

    if (this._writeBuffer.length >= this._maxBufferSize) {
      this.flush();
    }
  }

  appendLayerResult(
    layer: 1 | 2 | 3,
    result: LayerResult,
    taskId: string
  ): void {
    const eventType = `layer${layer}_collected` as PersistenceEventType;

    this.append({
      type: eventType,
      timestamp: result.payload.timestamp || Date.now(),
      taskId,
      data: {
        ...result.payload,
        cost: result.cost,
        mode: result.mode
      }
    });
  }

  appendHealthChange(
    previousMode: HealthMode,
    newMode: HealthMode,
    taskId: string,
    reason?: string
  ): void {
    this.append({
      type: 'health_changed',
      timestamp: Date.now(),
      taskId,
      data: {
        previousMode,
        newMode,
        reason
      }
    });
  }

  appendRecovery(
    type: RecoveryType,
    status: 'started' | 'completed' | 'failed',
    taskId: string,
    details?: Record<string, any>
  ): void {
    const eventType = `recovery_${status}` as PersistenceEventType;

    this.append({
      type: eventType,
      timestamp: Date.now(),
      taskId,
      data: {
        recoveryType: type,
        ...details
      }
    });
  }

  private extractLayer(eventType: string): 1 | 2 | 3 | undefined {
    if (eventType.startsWith('layer1')) return 1;
    if (eventType.startsWith('layer2')) return 2;
    if (eventType.startsWith('layer3')) return 3;
    return undefined;
  }
}
```

### 6.5.4.3 刷新与关闭

```typescript
class PersistenceManager {
  async flush(): Promise<void> {
    if (this._writeBuffer.length === 0 || !this._fileHandle) return;

    const records = [...this._writeBuffer];
    this._writeBuffer = [];

    const lines = records
      .map(r => JSON.stringify(r))
      .join('\n') + '\n';

    try {
      await this._fileHandle.write(lines);
    } catch (error) {
      console.error('Failed to write heartbeat records:', error);
      this._writeBuffer.unshift(...records);
    }
  }

  async close(): Promise<void> {
    if (this._flushTimer) {
      clearInterval(this._flushTimer);
      this._flushTimer = null;
    }

    await this.flush();

    if (this._fileHandle) {
      await this._fileHandle.close();
      this._fileHandle = null;
    }
  }
}
```

### 6.5.5.2 记录方法

```typescript
class MetricsCollector {
  recordLayer1Result(result: Layer1Result): void {
    const m = this._layer1Metrics;
    m.collectCount++;

    if (result.cost >= 0) {
      m.successCount++;
      m.modeDistribution[result.mode]++;
      m.avgCost = this.updateAverage(m.avgCost, result.cost, m.successCount);
      m.maxCost = Math.max(m.maxCost, result.cost);
    } else {
      m.failureCount++;
    }
  }

  recordLayer2Result(result: Layer2Result): void {
    const m = this._layer2Metrics;
    m.collectCount++;

    if (result.cost >= 0) {
      m.successCount++;
      m.modeDistribution[result.mode]++;
      m.avgCost = this.updateAverage(m.avgCost, result.cost, m.successCount);
      m.maxCost = Math.max(m.maxCost, result.cost);

      if (result.payload.charsDelta !== undefined) {
        m.avgCharsDelta = this.updateAverage(
          m.avgCharsDelta,
          Math.abs(result.payload.charsDelta),
          m.successCount
        );
      }

      if (result.mode === 'model-stalled') {
        m.stalledDetections++;
      }

      if (result.mode === 'model-queuing' && result.payload.queuePosition) {
        m.queueDetections++;
        m.maxQueuePosition = Math.max(m.maxQueuePosition, result.payload.queuePosition);
      }
    } else {
      m.failureCount++;
    }
  }

  recordLayer3Result(result: Layer3Result): void {
    const m = this._layer3Metrics;
    m.collectCount++;

    if (result.cost >= 0) {
      m.successCount++;
      m.avgCost = this.updateAverage(m.avgCost, result.cost, m.successCount);
      m.maxCost = Math.max(m.maxCost, result.cost);

      m.avgHealthScore = this.updateAverage(m.avgHealthScore, result.healthScore, m.successCount);
      m.minHealthScore = Math.min(m.minHealthScore, result.healthScore);

      const level = this.getHealthLevel(result.healthScore);
      m.healthLevelDistribution[level]++;

      for (const factor of result.riskFactors) {
        m.riskFactorCounts[factor.type] = (m.riskFactorCounts[factor.type] || 0) + 1;
      }
    } else {
      m.failureCount++;
    }
  }

  recordStateTransition(from: HealthMode, to: HealthMode): void {
    this._daemonMetrics.stateTransitions++;
  }

  recordRecoveryAttempt(success: boolean): void {
    this._daemonMetrics.recoveryAttempts++;
    if (success) {
      this._daemonMetrics.recoverySuccesses++;
    } else {
      this._daemonMetrics.recoveryFailures++;
    }
  }

  recordError(): void {
    this._daemonMetrics.errorCount++;
  }

  addPausedDuration(duration: number): void {
    this._daemonMetrics.totalPausedDuration += duration;
  }

  private updateAverage(current: number, value: number, n: number): number {
    return current * (n - 1) / n + value / n;
  }

  private getHealthLevel(score: number): HealthLevel {
    if (score >= 80) return 'healthy';
    if (score >= 60) return 'degraded';
    if (score >= 40) return 'unhealthy';
    return 'critical';
  }
}
```

## 6.5.6 文件轮转策略

### 6.5.6.1 轮转管理器

```typescript
class FileRotationManager {
  private _basePath: string;
  private _maxFileSize: number;
  private _maxFiles: number;
  private _currentFile: string;
  private _currentSize: number = 0;

  constructor(
    basePath: string,
    maxFileSize: number = 10 * 1024 * 1024,
    maxFiles: number = 10
  ) {
    this._basePath = basePath;
    this._maxFileSize = maxFileSize;
    this._maxFiles = maxFiles;
    this._currentFile = this.getCurrentFilePath();
  }

  async checkRotation(bytesWritten: number): Promise<boolean> {
    this._currentSize += bytesWritten;

    if (this._currentSize >= this._maxFileSize) {
      await this.rotate();
      return true;
    }

    return false;
  }

  private async rotate(): Promise<void> {
    await this.archiveCurrentFile();
    this._currentFile = this.getCurrentFilePath();
    this._currentSize = 0;
    await this.pruneOldFiles();
  }

  private async archiveCurrentFile(): Promise<void> {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const archivedPath = `${this._basePath}.${timestamp}`;
    
    try {
      await fs.rename(this._currentFile, archivedPath);
    } catch (error) {
      console.error('Failed to archive file:', error);
    }
  }

  private async pruneOldFiles(): Promise<void> {
    const files = await this.getArchiveFiles();

    if (files.length > this._maxFiles) {
      const toDelete = files.slice(0, files.length - this._maxFiles);
      
      for (const file of toDelete) {
        await fs.unlink(file).catch(() => {});
      }
    }
  }

  private async getArchiveFiles(): Promise<string[]> {
    const dir = path.dirname(this._basePath);
    const base = path.basename(this._basePath);

    const files = await fs.readdir(dir);
    return files
      .filter(f => f.startsWith(base) && f !== base)
      .map(f => path.join(dir, f))
      .sort();
  }

  private getCurrentFilePath(): string {
    return this._basePath;
  }

  getCurrentFile(): string {
    return this._currentFile;
  }
}
```

### 6.5.7.1 日志读取器

```typescript
class HeartbeatLogReader {
  private _filePath: string;

  constructor(filePath: string) {
    this._filePath = filePath;
  }

  async *readAll(): AsyncGenerator<HeartbeatJsonlRecord> {
    const stream = fs.createReadStream(this._filePath, { encoding: 'utf-8' });
    const rl = readline.createInterface({ input: stream });

    for await (const line of rl) {
      if (line.trim()) {
        try {
          yield JSON.parse(line);
        } catch (error) {
          console.warn('Failed to parse line:', line);
        }
      }
    }
  }

  async readByType(type: PersistenceEventType): Promise<HeartbeatJsonlRecord[]> {
    const results: HeartbeatJsonlRecord[] = [];

    for await (const record of this.readAll()) {
      if (record.type === type) {
        results.push(record);
      }
    }

    return results;
  }

  async readByLayer(layer: 1 | 2 | 3): Promise<HeartbeatJsonlRecord[]> {
    const results: HeartbeatJsonlRecord[] = [];

    for await (const record of this.readAll()) {
      if (record.layer === layer) {
        results.push(record);
      }
    }

    return results;
  }

  async readByTimeRange(start: number, end: number): Promise<HeartbeatJsonlRecord[]> {
    const results: HeartbeatJsonlRecord[] = [];

    for await (const record of this.readAll()) {
      if (record.ts >= start && record.ts <= end) {
        results.push(record);
      }
    }

    return results;
  }

  async getSummary(): Promise<LogSummary> {
    let totalRecords = 0;
    let startTime = Infinity;
    let endTime = 0;
    const typeCounts: Record<string, number> = {};

    for await (const record of this.readAll()) {
      totalRecords++;
      startTime = Math.min(startTime, record.ts);
      endTime = Math.max(endTime, record.ts);
      typeCounts[record.type] = (typeCounts[record.type] || 0) + 1;
    }

    return {
      totalRecords,
      startTime: startTime === Infinity ? null : startTime,
      endTime: endTime === 0 ? null : endTime,
      typeCounts
    };
  }
}

interface LogSummary {
  totalRecords: number;
  startTime: number | null;
  endTime: number | null;
  typeCounts: Record<string, number>;
}
```

### 6.5.7.2 统计分析

```typescript
class HeartbeatAnalyzer {
  private _reader: HeartbeatLogReader;

  constructor(filePath: string) {
    this._reader = new HeartbeatLogReader(filePath);
  }

  async analyzeHealthTransitions(): Promise<HealthTransitionAnalysis> {
    const transitions: HealthTransition[] = [];
    let previousMode: HealthMode | null = null;

    for await (const record of this._reader.readByType('health_changed')) {
      const { previousMode: from, newMode: to, reason } = record.payload;
      
      transitions.push({
        timestamp: record.ts,
        from,
        to,
        reason
      });

      previousMode = to;
    }

    return {
      totalTransitions: transitions.length,
      transitions,
      modeDurations: this.calculateModeDurations(transitions)
    };
  }

  private calculateModeDurations(transitions: HealthTransition[]): Record<HealthMode, number> {
    const durations: Record<HealthMode, number> = {} as any;
    let lastTransition: HealthTransition | null = null;

    for (const t of transitions) {
      if (lastTransition) {
        const duration = t.timestamp - lastTransition.timestamp;
        durations[lastTransition.to] = (durations[lastTransition.to] || 0) + duration;
      }
      lastTransition = t;
    }

    return durations;
  }

  async analyzeRecoveryPerformance(): Promise<RecoveryAnalysis> {
    const started = await this._reader.readByType('recovery_started');
    const completed = await this._reader.readByType('recovery_completed');
    const failed = await this._reader.readByType('recovery_failed');

    const successCount = completed.length;
    const failureCount = failed.length;
    const totalDuration = this.calculateTotalRecoveryDuration(started, completed);

    return {
      totalAttempts: started.length,
      successCount,
      failureCount,
      successRate: started.length > 0 ? successCount / started.length : 0,
      avgRecoveryDuration: completed.length > 0 ? totalDuration / completed.length : 0
    };
  }

  private calculateTotalRecoveryDuration(
    started: HeartbeatJsonlRecord[],
    completed: HeartbeatJsonlRecord[]
  ): number {
    let total = 0;

    for (const complete of completed) {
      const start = started.find(s => 
        s.payload.recoveryType === complete.payload.recoveryType &&
        s.ts < complete.ts
      );
      
      if (start) {
        total += complete.ts - start.ts;
      }
    }

    return total;
  }

  async analyzeLayerPerformance(layer: 1 | 2 | 3): Promise<LayerPerformanceAnalysis> {
    const records = await this._reader.readByLayer(layer);

    const costs = records.map(r => r.payload.cost || 0);
    const avgCost = costs.reduce((a, b) => a + b, 0) / costs.length;
    const maxCost = Math.max(...costs);

    const modeCounts: Record<string, number> = {};
    for (const r of records) {
      const mode = r.payload.mode;
      if (mode) {
        modeCounts[mode] = (modeCounts[mode] || 0) + 1;
      }
    }

    return {
      collectCount: records.length,
      avgCost,
      maxCost,
      modeDistribution: modeCounts
    };
  }
}

interface HealthTransitionAnalysis {
  totalTransitions: number;
  transitions: HealthTransition[];
  modeDurations: Record<HealthMode, number>;
}

interface HealthTransition {
  timestamp: number;
  from: HealthMode;
  to: HealthMode;
  reason?: string;
}

interface RecoveryAnalysis {
  totalAttempts: number;
  successCount: number;
  failureCount: number;
  successRate: number;
  avgRecoveryDuration: number;
}

interface LayerPerformanceAnalysis {
  collectCount: number;
  avgCost: number;
  maxCost: number;
  modeDistribution: Record<string, number>;
}
```

```typescript
    };
  }

  private summarizeLayer2(m: Layer2Metrics): LayerSummary {
    return {
      collectCount: m.collectCount,
      successRate: m.collectCount > 0 ? m.successCount / m.collectCount : 0,
      avgCost: m.avgCost,
      maxCost: m.maxCost,
      topModes: this.getTopModes(m.modeDistribution, 3),
      extra: {
        avgCharsDelta: m.avgCharsDelta,
        stalledDetections: m.stalledDetections,
        queueDetections: m.queueDetections,
        maxQueuePosition: m.maxQueuePosition
      }
    };
  }

  private summarizeLayer3(m: Layer3Metrics): LayerSummary {
    return {
      collectCount: m.collectCount,
      successRate: m.collectCount > 0 ? m.successCount / m.collectCount : 0,
      avgCost: m.avgCost,
      maxCost: m.maxCost,
      extra: {
        avgHealthScore: m.avgHealthScore,
        minHealthScore: m.minHealthScore
      }
    };
  }

  private getTopModes(distribution: Record<string, number>, n: number): string[] {
    return Object.entries(distribution)
      .sort((a, b) => b[1] - a[1])
      .slice(0, n)
      .map(([mode]) => mode);
  }
}
```

```typescript
function formatDuration(ms: number): string {
  const seconds = Math.floor(ms / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);

  if (hours > 0) {
    return `${hours}h ${minutes % 60}m ${seconds % 60}s`;
  }
  if (minutes > 0) {
    return `${minutes}m ${seconds % 60}s`;
  }
  return `${seconds}s`;
}
```
