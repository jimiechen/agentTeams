# 5.4 防抖与幂等性保护

## 5.4.1 概述

在心跳系统中，恢复动作可能被多次触发。为确保系统稳定性和操作安全性，必须实现：

- **防抖（Debounce）**：防止短时间内重复触发相同操作
- **幂等性（Idempotency）**：确保同一操作多次执行产生相同结果

## 5.4.2 防抖机制

### 5.4.2.1 防抖原理

```
┌─────────────────────────────────────────────────────────────────────────┐
│                         防抖机制原理                                      │
└─────────────────────────────────────────────────────────────────────────┘

触发序列:
T=0ms    ────┐
T=100ms  ────┤
T=200ms  ────┤    触发请求
T=300ms  ────┤
T=400ms  ────┘
              │
              ▼
         ┌────────────┐
         │ 等待窗口   │  500ms
         │ (debounce) │
         └─────┬──────┘
               │
               ▼
T=900ms  ───────────────► 执行操作（仅执行一次）
```

### 5.4.2.3 防抖器实现

```typescript
class Debouncer<T extends (...args: any[]) => any> {
  private timer: NodeJS.Timeout | null = null;
  private lastCallTime: number = 0;
  private lastArgs: any[] = [];
  private config: DebounceConfig;
  private fn: T;
  
  constructor(fn: T, config: DebounceConfig = DEFAULT_DEBOUNCE_CONFIG) {
    this.fn = fn;
    this.config = config;
  }

  call(...args: Parameters<T>): void {
    const now = Date.now();
    this.lastArgs = args;
    
    if (this.config.leading && !this.timer) {
      this.execute();
    }
    
    if (this.timer) {
      clearTimeout(this.timer);
    }
    
    const remainingWait = this.getRemainingWait(now);
    
    this.timer = setTimeout(() => {
      if (this.config.trailing) {
        this.execute();
      }
      this.timer = null;
    }, remainingWait);
  }

  cancel(): void {
    if (this.timer) {
      clearTimeout(this.timer);
      this.timer = null;
    }
    this.lastArgs = [];
  }

  flush(): void {
    if (this.timer) {
      this.execute();
      this.cancel();
    }
  }

  private execute(): void {
    this.lastCallTime = Date.now();
    this.fn(...this.lastArgs);
  }

  private getRemainingWait(now: number): number {
    const timeSinceLastCall = now - this.lastCallTime;
    const maxWait = this.config.maxWaitMs - timeSinceLastCall;
    
    return Math.min(this.config.waitMs, Math.max(0, maxWait));
  }
}
```

### 5.4.2.4 防抖管理器

```typescript
class DebounceManager {
  private debouncers: Map<string, Debouncer<any>> = new Map();
  
  getDebouncer<T extends (...args: any[]) => any>(
    key: string,
    fn: T,
    config?: DebounceConfig
  ): Debouncer<T> {
    let debouncer = this.debouncers.get(key);
    
    if (!debouncer) {
      debouncer = new Debouncer(fn, config);
      this.debouncers.set(key, debouncer);
    }
    
    return debouncer;
  }
  
  cancel(key: string): void {
    const debouncer = this.debouncers.get(key);
    if (debouncer) {
      debouncer.cancel();
    }
  }
  
  cancelAll(): void {
    for (const debouncer of this.debouncers.values()) {
      debouncer.cancel();
    }
  }
  
  flush(key: string): void {
    const debouncer = this.debouncers.get(key);
    if (debouncer) {
      debouncer.flush();
    }
  }
  
  remove(key: string): void {
    this.cancel(key);
    this.debouncers.delete(key);
  }
}
```

## 5.4.3 幂等性保护

### 5.4.3.1 幂等性原理

```
┌─────────────────────────────────────────────────────────────────────────┐
│                         幂等性保护原理                                    │
└─────────────────────────────────────────────────────────────────────────┘

场景一：正常执行
┌──────────┐     执行     ┌──────────┐
│ 请求 A   │ ──────────► │ 执行成功  │
└──────────┘              └──────────┘

场景二：重复请求（幂等保护）
┌──────────┐     执行     ┌──────────┐
│ 请求 A   │ ──────────► │ 执行成功  │
└──────────┘              └──────────┘
┌──────────┐     检查     ┌──────────┐
│ 请求 A'  │ ──────────► │ 已执行    │ ──► 跳过
└──────────┘              └──────────┘
     (相同 ID)
```

### 5.4.3.2 幂等键设计

```typescript
interface IdempotencyKey {
  workspaceId: string;
  actionType: string;
  targetId: string;
  hash: string;
}

class IdempotencyKeyGenerator {
  generate(
    workspaceId: string,
    actionType: string,
    params: Record<string, any>
  ): IdempotencyKey {
    const targetId = this.extractTargetId(actionType, params);
    const hash = this.generateHash(params);
    
    return {
      workspaceId,
      actionType,
      targetId,
      hash
    };
  }
  
  toString(key: IdempotencyKey): string {
    return `${key.workspaceId}:${key.actionType}:${key.targetId}:${key.hash}`;
  }
  
  private extractTargetId(actionType: string, params: Record<string, any>): string {
    switch (actionType) {
      case 'click-button':
        return params.buttonType || 'unknown';
      case 'stop-retry':
        return params.taskId || 'unknown';
      default:
        return 'default';
    }
  }
  
  private generateHash(params: Record<string, any>): string {
    const content = JSON.stringify(params, Object.keys(params).sort());
    let hash = 0;
    
    for (let i = 0; i < content.length; i++) {
      const char = content.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    
    return Math.abs(hash).toString(16);
  }
}
```

### 5.4.3.3 幂等性存储

```typescript
interface IdempotencyRecord {
  key: string;
  status: 'pending' | 'completed' | 'failed';
  result?: any;
  createdAt: string;
  completedAt?: string;
  ttl: number;
}

class IdempotencyStore {
  private records: Map<string, IdempotencyRecord> = new Map();
  private readonly DEFAULT_TTL = 60000;
  
  async check(key: string): Promise<{
    exists: boolean;
    status?: 'pending' | 'completed' | 'failed';
    result?: any;
  }> {
    const record = this.records.get(key);
    
    if (!record) {
      return { exists: false };
    }
    
    if (this.isExpired(record)) {
      this.records.delete(key);
      return { exists: false };
    }
    
    return {
      exists: true,
      status: record.status,
      result: record.result
    };
  }
  
  async start(key: string, ttl: number = this.DEFAULT_TTL): Promise<void> {
    const record: IdempotencyRecord = {
      key,
      status: 'pending',
      createdAt: new Date().toISOString(),
      ttl
    };
    
    this.records.set(key, record);
  }
  
  async complete(key: string, result?: any): Promise<void> {
    const record = this.records.get(key);
    
    if (record) {
      record.status = 'completed';
      record.result = result;
      record.completedAt = new Date().toISOString();
    }
  }
  
  async fail(key: string): Promise<void> {
    const record = this.records.get(key);
    
    if (record) {
      record.status = 'failed';
      record.completedAt = new Date().toISOString();
    }
  }
  
  private isExpired(record: IdempotencyRecord): boolean {
    const createdAt = new Date(record.createdAt).getTime();
    return Date.now() - createdAt > record.ttl;
  }
  
  cleanup(): void {
    for (const [key, record] of this.records.entries()) {
      if (this.isExpired(record)) {
        this.records.delete(key);
      }
    }
  }
}
```

### 5.4.3.4 幂等性执行器

```typescript
class IdempotentExecutor {
  private keyGenerator: IdempotencyKeyGenerator;
  private store: IdempotencyStore;
  
  constructor() {
    this.keyGenerator = new IdempotencyKeyGenerator();
    this.store = new IdempotencyStore();
  }

  async execute<T>(
    workspaceId: string,
    actionType: string,
    params: Record<string, any>,
    fn: () => Promise<T>,
    options: IdempotencyOptions = {}
  ): Promise<IdempotentResult<T>> {
    const key = this.keyGenerator.generate(workspaceId, actionType, params);
    const keyString = this.keyGenerator.toString(key);
    
    const existing = await this.store.check(keyString);
    
    if (existing.exists) {
      console.log(`[IdempotentExecutor] 操作已存在: ${keyString}`);
      
      if (existing.status === 'completed') {
        return {
          success: true,
          result: existing.result,
          cached: true,
          key: keyString
        };
      }
      
      if (existing.status === 'pending') {
        return {
          success: false,
          result: null,
          cached: false,
          key: keyString,
          error: '操作正在执行中'
        };
      }
    }
    
    await this.store.start(keyString, options.ttl);
    
    try {
      const result = await fn();
      await this.store.complete(keyString, result);
      
      return {
        success: true,
        result,
        cached: false,
        key: keyString
      };
    } catch (error) {
      await this.store.fail(keyString);
      
      return {
        success: false,
        result: null,
        cached: false,
        key: keyString,
        error: error instanceof Error ? error.message : '执行失败'
      };
    }
  }
}

interface IdempotencyOptions {
  ttl?: number;
  skipCache?: boolean;
}

interface IdempotentResult<T> {
  success: boolean;
  result: T | null;
  cached: boolean;
  key: string;
  error?: string;
}
```

## 5.4.4 恢复动作防抖与幂等性集成

### 5.4.4.1 集成架构

```typescript
class ProtectedRecoveryExecutor {
  private debounceManager: DebounceManager;
  private idempotentExecutor: IdempotentExecutor;
  private actionExecutors: Map<string, ActionExecutor>;
  
  constructor() {
    this.debounceManager = new DebounceManager();
    this.idempotentExecutor = new IdempotentExecutor();
    this.actionExecutors = new Map();
  }

  async execute(
    context: RecoveryContext,
    actionType: string,
    params: Record<string, any>
  ): Promise<RecoveryResult> {
    const debounceKey = `${context.workspaceId}:${actionType}`;
    const debounceConfig = DEBOUNCE_CONFIGS_BY_ACTION[actionType] || DEFAULT_DEBOUNCE_CONFIG;
    
    return new Promise((resolve) => {
      const debouncer = this.debounceManager.getDebouncer(
        debounceKey,
        async () => {
          const result = await this.executeWithIdempotency(
            context,
            actionType,
            params
          );
          resolve(result);
        },
        debounceConfig
      );
      
      debouncer.call();
    });
  }

  private async executeWithIdempotency(
    context: RecoveryContext,
    actionType: string,
    params: Record<string, any>
  ): Promise<RecoveryResult> {
    const result = await this.idempotentExecutor.execute(
      context.workspaceId,
      actionType,
      params,
      async () => {
        const executor = this.actionExecutors.get(actionType);
        
        if (!executor) {
          throw new Error(`未找到执行器: ${actionType}`);
        }
        
        return executor.execute(context);
      }
    );
    
    if (result.cached) {
      return {
        success: true,
        action: actionType,
        reason: '操作已执行（缓存结果）',
        durationMs: 0,
        nextState: 'verifying'
      };
    }
    
    if (!result.success) {
      return {
        success: false,
        action: actionType,
        reason: result.error || '执行失败',
        durationMs: 0,
        nextState: context.mode
      };
    }
    
    return result.result;
  }
}
```

## 5.4.5 并发控制

### 5.4.5.1 并发锁

```typescript
class ConcurrencyLock {
  private locks: Map<string, LockEntry> = new Map();
  
  async acquire(key: string, timeout: number = 30000): Promise<LockHandle> {
    const existing = this.locks.get(key);
    
    if (existing && !this.isExpired(existing)) {
      throw new Error(`锁已被占用: ${key}`);
    }
    
    const entry: LockEntry = {
      key,
      acquiredAt: Date.now(),
      timeout
    };
    
    this.locks.set(key, entry);
    
    return {
      key,
      release: () => this.release(key)
    };
  }
  
  async tryAcquire(key: string, timeout: number = 30000): Promise<LockHandle | null> {
    try {
      return await this.acquire(key, timeout);
    } catch {
      return null;
    }
  }
  
  release(key: string): void {
    this.locks.delete(key);
  }
  
  isLocked(key: string): boolean {
    const entry = this.locks.get(key);
    return entry ? !this.isExpired(entry) : false;
  }
  
  private isExpired(entry: LockEntry): boolean {
    return Date.now() - entry.acquiredAt > entry.timeout;
  }
}

interface LockEntry {
  key: string;
  acquiredAt: number;
  timeout: number;
}

interface LockHandle {
  key: string;
  release: () => void;
}
```

### 5.4.5.2 并发执行器

```typescript
class ConcurrentRecoveryExecutor {
  private lock: ConcurrencyLock;
  private maxConcurrent: number;
  private activeCount: number = 0;
  
  constructor(maxConcurrent: number = 3) {
    this.lock = new ConcurrencyLock();
    this.maxConcurrent = maxConcurrent;
  }

  async execute(
    context: RecoveryContext,
    actionType: string,
    fn: () => Promise<RecoveryResult>
  ): Promise<RecoveryResult> {
    if (this.activeCount >= this.maxConcurrent) {
      return {
        success: false,
        action: actionType,
        reason: '已达到最大并发数',
        durationMs: 0,
        nextState: context.mode
      };
    }
    
    const lockKey = `${context.workspaceId}:${actionType}`;
    const lockHandle = await this.lock.tryAcquire(lockKey);
    
    if (!lockHandle) {
      return {
        success: false,
        action: actionType,
        reason: '操作正在执行中',
        durationMs: 0,
        nextState: context.mode
      };
    }
    
    this.activeCount++;
    
    try {
      return await fn();
    } finally {
      this.activeCount--;
      lockHandle.release();
    }
  }
}
```

### 5.4.6.1 操作签名

```typescript
interface OperationSignature {
  workspaceId: string;
  actionType: string;
  buttonType?: string;
  taskId?: string;
  timestamp: number;
}

class OperationSignatureGenerator {
  generate(
    workspaceId: string,
    actionType: string,
    params: Record<string, any>
  ): OperationSignature {
    return {
      workspaceId,
      actionType,
      buttonType: params.buttonType,
      taskId: params.taskId,
      timestamp: Date.now()
    };
  }
  
  toKey(signature: OperationSignature): string {
    const parts = [
      signature.workspaceId,
      signature.actionType,
      signature.buttonType || '',
      signature.taskId || ''
    ];
    
    return parts.filter(Boolean).join(':');
  }
  
  isSimilar(a: OperationSignature, b: OperationSignature, windowMs: number): boolean {
    if (a.workspaceId !== b.workspaceId) return false;
    if (a.actionType !== b.actionType) return false;
    if (a.buttonType !== b.buttonType) return false;
    if (a.taskId !== b.taskId) return false;
    
    return Math.abs(a.timestamp - b.timestamp) <= windowMs;
  }
}
```

```typescript
class OperationDeduplicator {
  private recentOperations: OperationSignature[] = [];
  private readonly WINDOW_MS = 5000;
  private readonly MAX_OPERATIONS = 100;
  
  check(signature: OperationSignature): boolean {
    this.cleanup();
    
    const similar = this.recentOperations.find(op =>
      this.isSimilar(op, signature, this.WINDOW_MS)
    );
    
    if (similar) {
      console.log(`[Deduplicator] 检测到重复操作: ${signature.actionType}`);
      return false;
    }
    
    this.recentOperations.push(signature);
    return true;
  }
  
  private cleanup(): void {
    const cutoff = Date.now() - this.WINDOW_MS;
    this.recentOperations = this.recentOperations
      .filter(op => op.timestamp > cutoff)
      .slice(-this.MAX_OPERATIONS);
  }
  
  private isSimilar(a: OperationSignature, b: OperationSignature, windowMs: number): boolean {
    if (a.workspaceId !== b.workspaceId) return false;
    if (a.actionType !== b.actionType) return false;
    if (a.buttonType !== b.buttonType) return false;
    if (a.taskId !== b.taskId) return false;
    
    return Math.abs(a.timestamp - b.timestamp) <= windowMs;
  }
}
```

```typescript
describe('IdempotentExecutor', () => {
  it('应缓存已执行的操作', async () => {
    const executor = new IdempotentExecutor();
    let callCount = 0;
    
    const fn = async () => {
      callCount++;
      return { value: 42 };
    };
    
    const result1 = await executor.execute('ws-1', 'test-action', {}, fn);
    expect(result1.cached).toBe(false);
    expect(callCount).toBe(1);
    
    const result2 = await executor.execute('ws-1', 'test-action', {}, fn);
    expect(result2.cached).toBe(true);
    expect(callCount).toBe(1);
    expect(result2.result).toEqual({ value: 42 });
  });
  
  it('不同参数应分别执行', async () => {
    const executor = new IdempotentExecutor();
    let callCount = 0;
    
    const fn = async () => {
      callCount++;
      return { value: callCount };
    };
    
    const result1 = await executor.execute('ws-1', 'test', { id: 1 }, fn);
    const result2 = await executor.execute('ws-1', 'test', { id: 2 }, fn);
    
    expect(result1.cached).toBe(false);
    expect(result2.cached).toBe(false);
    expect(callCount).toBe(2);
  });
});
```
