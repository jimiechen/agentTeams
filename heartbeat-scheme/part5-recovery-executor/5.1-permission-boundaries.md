# 5.1 RecoveryExecutor 权限边界

## 5.1.1 权限边界概述

RecoveryExecutor 是心跳系统中执行恢复动作的核心组件。为确保安全可控，必须定义清晰的权限边界：

- **最小权限原则**：只允许执行预定义的白名单操作
- **操作审计**：所有恢复动作必须记录日志
- **人工确认**：高风险操作需要人工确认
- **速率限制**：防止恢复动作过于频繁

### 5.1.2.3 按钮查找器

```typescript
class ButtonFinder {
  private page: Page;
  
  constructor(page: Page) {
    this.page = page;
  }

  async findButton(buttonType: string): Promise<{
    found: boolean;
    selector: string | null;
    element: Locator | null;
  }> {
    const selectors = BUTTON_SELECTOR_MAP[buttonType] || [];
    
    for (const selector of selectors) {
      try {
        const element = this.page.locator(selector).first();
        const isVisible = await element.isVisible({ timeout: 1000 });
        
        if (isVisible) {
          return {
            found: true,
            selector,
            element
          };
        }
      } catch (error) {
        continue;
      }
    }
    
    return {
      found: false,
      selector: null,
      element: null
    };
  }

  async findAllButtons(): Promise<Map<string, Locator>> {
    const foundButtons = new Map<string, Locator>();
    
    for (const [buttonType, selectors] of Object.entries(BUTTON_SELECTOR_MAP)) {
      const result = await this.findButton(buttonType);
      if (result.found && result.element) {
        foundButtons.set(buttonType, result.element);
      }
    }
    
    return foundButtons;
  }
}
```

## 5.1.3 权限检查器

### 5.1.3.1 权限检查接口

```typescript
interface PermissionCheckResult {
  allowed: boolean;
  reason: string;
  requiresConfirmation: boolean;
  remainingQuota: number;
}

class PermissionChecker {
  private clickHistory: Map<string, number[]> = new Map();
  
  checkPermission(buttonType: string): PermissionCheckResult {
    const whitelistEntry = BUTTON_WHITELIST.find(
      entry => entry.buttonSelector.includes(buttonType)
    );
    
    if (!whitelistEntry) {
      return {
        allowed: false,
        reason: `按钮 "${buttonType}" 不在白名单中`,
        requiresConfirmation: true,
        remainingQuota: 0
      };
    }
    
    const clicksThisHour = this.getClicksThisHour(buttonType);
    const remainingQuota = whitelistEntry.maxClicksPerHour - clicksThisHour;
    
    if (remainingQuota <= 0) {
      return {
        allowed: false,
        reason: `已达到每小时点击上限 (${whitelistEntry.maxClicksPerHour})`,
        requiresConfirmation: true,
        remainingQuota: 0
      };
    }
    
    return {
      allowed: true,
      reason: '权限检查通过',
      requiresConfirmation: whitelistEntry.requiresConfirmation,
      remainingQuota
    };
  }
  
  recordClick(buttonType: string): void {
    const now = Date.now();
    
    if (!this.clickHistory.has(buttonType)) {
      this.clickHistory.set(buttonType, []);
    }
    
    this.clickHistory.get(buttonType)!.push(now);
    this.cleanupOldClicks(buttonType);
  }
  
  private getClicksThisHour(buttonType: string): number {
    const history = this.clickHistory.get(buttonType) || [];
    const oneHourAgo = Date.now() - 3600000;
    
    return history.filter(timestamp => timestamp > oneHourAgo).length;
  }
  
  private cleanupOldClicks(buttonType: string): void {
    const history = this.clickHistory.get(buttonType);
    if (!history) return;
    
    const oneHourAgo = Date.now() - 3600000;
    const filtered = history.filter(timestamp => timestamp > oneHourAgo);
    this.clickHistory.set(buttonType, filtered);
  }
}
```

### 5.1.3.2 操作审计日志

```typescript
interface AuditLog {
  timestamp: string;
  workspaceId: string;
  taskId: string;
  action: string;
  buttonType: string;
  selector: string;
  result: 'success' | 'failed' | 'skipped';
  reason: string;
  durationMs: number;
}

class AuditLogger {
  private logs: AuditLog[] = [];
  private readonly MAX_LOGS = 10000;
  
  log(entry: Omit<AuditLog, 'timestamp'>): void {
    const log: AuditLog = {
      ...entry,
      timestamp: new Date().toISOString()
    };
    
    this.logs.push(log);
    
    if (this.logs.length > this.MAX_LOGS) {
      this.logs = this.logs.slice(-this.MAX_LOGS);
    }
    
    this.writeToFile(log);
  }
  
  private writeToFile(log: AuditLog): void {
    const logLine = JSON.stringify(log) + '\n';
    const logPath = `logs/recovery-audit-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, logLine);
  }
  
  getLogs(filter?: {
    workspaceId?: string;
    action?: string;
    since?: Date;
  }): AuditLog[] {
    let filtered = [...this.logs];
    
    if (filter?.workspaceId) {
      filtered = filtered.filter(log => log.workspaceId === filter.workspaceId);
    }
    
    if (filter?.action) {
      filtered = filtered.filter(log => log.action === filter.action);
    }
    
    if (filter?.since) {
      filtered = filtered.filter(
        log => new Date(log.timestamp) >= filter.since!
      );
    }
    
    return filtered;
  }
}
```

## 5.1.4 风险等级与确认策略

### 5.1.4.1 风险等级定义

| 风险等级 | 说明 | 示例操作 | 确认要求 |
|---------|------|---------|---------|
| **low** | 可逆、无副作用 | 点击"后台运行"、"保留" | 无需确认 |
| **medium** | 可能中断当前任务 | 点击"停止"、"取消" | 无需确认（有速率限制） |
| **high** | 不可逆、有数据丢失风险 | 点击"删除" | 需要人工确认 |

### 5.1.4.2 确认策略实现

```typescript
interface ConfirmationStrategy {
  requireConfirmation(action: RecoveryAction): boolean;
  getConfirmationMessage(action: RecoveryAction): string;
  getConfirmationTimeout(): number;
}

class DefaultConfirmationStrategy implements ConfirmationStrategy {
  requireConfirmation(action: RecoveryAction): boolean {
    const highRiskActions: string[] = [
      'delete',
      'stop-and-delete',
      'force-cancel'
    ];
    
    return highRiskActions.includes(action.type);
  }
  
  getConfirmationMessage(action: RecoveryAction): string {
    const messages: Record<string, string> = {
      'delete': '即将执行删除操作，此操作不可恢复。是否继续？',
      'stop-and-delete': '即将停止任务并删除相关数据。是否继续？',
      'force-cancel': '即将强制取消当前任务。是否继续？'
    };
    
    return messages[action.type] || `确认执行 ${action.type} 操作？`;
  }
  
  getConfirmationTimeout(): number {
    return 30000;
  }
}
```

### 5.1.4.3 确认请求处理器

```typescript
class ConfirmationHandler {
  private pendingConfirmations: Map<string, {
    action: RecoveryAction;
    resolve: (confirmed: boolean) => void;
    timeout: NodeJS.Timeout;
  }> = new Map();

  async requestConfirmation(
    action: RecoveryAction,
    strategy: ConfirmationStrategy
  ): Promise<boolean> {
    return new Promise((resolve) => {
      const confirmationId = `${action.workspaceId}-${Date.now()}`;
      
      const timeout = setTimeout(() => {
        this.pendingConfirmations.delete(confirmationId);
        resolve(false);
      }, strategy.getConfirmationTimeout());
      
      this.pendingConfirmations.set(confirmationId, {
        action,
        resolve,
        timeout
      });
      
      this.notifyUser(confirmationId, action, strategy);
    });
  }

  handleUserResponse(confirmationId: string, confirmed: boolean): void {
    const pending = this.pendingConfirmations.get(confirmationId);
    
    if (pending) {
      clearTimeout(pending.timeout);
      pending.resolve(confirmed);
      this.pendingConfirmations.delete(confirmationId);
    }
  }

  private notifyUser(
    confirmationId: string,
    action: RecoveryAction,
    strategy: ConfirmationStrategy
  ): void {
    console.log(`[确认请求] ${confirmationId}`);
    console.log(`  操作: ${action.type}`);
    console.log(`  消息: ${strategy.getConfirmationMessage(action)}`);
  }
}
```

```typescript
    windowMs: 60000,

    windowMs: 300000,
```

### 5.1.5.2 速率限制器

```typescript
class RateLimiter {
  private operations: Map<string, number[]> = new Map();
  private lastOperationTime: Map<string, number> = new Map();
  
  async checkRateLimit(
    operationType: string,
    config: RateLimitConfig = RATE_LIMIT_CONFIGS[operationType] || RATE_LIMIT_CONFIGS['click']
  ): Promise<{
    allowed: boolean;
    waitTimeMs: number;
    reason: string;
  }> {
    const now = Date.now();
    const key = operationType;
    
    const lastTime = this.lastOperationTime.get(key) || 0;
    const timeSinceLastOp = now - lastTime;
    
    if (timeSinceLastOp < config.cooldownMs) {
      return {
        allowed: false,
        waitTimeMs: config.cooldownMs - timeSinceLastOp,
        reason: `冷却中，需等待 ${config.cooldownMs - timeSinceLastOp}ms`
      };
    }
    
    const history = this.operations.get(key) || [];
    const windowStart = now - config.windowMs;
    const recentOps = history.filter(t => t > windowStart);
    
    if (recentOps.length >= config.maxOperations) {
      const oldestOp = Math.min(...recentOps);
      const waitTime = oldestOp + config.windowMs - now;
      
      return {
        allowed: false,
        waitTimeMs: waitTime,
        reason: `已达到速率限制，需等待 ${waitTime}ms`
      };
    }
    
    return {
      allowed: true,
      waitTimeMs: 0,
      reason: '速率检查通过'
    };
  }
  
  recordOperation(operationType: string): void {
    const now = Date.now();
    
    if (!this.operations.has(operationType)) {
      this.operations.set(operationType, []);
    }
    
    this.operations.get(operationType)!.push(now);
    this.lastOperationTime.set(operationType, now);
    
    this.cleanupOldOperations(operationType);
  }
  
  private cleanupOldOperations(operationType: string): void {
    const history = this.operations.get(operationType);
    if (!history) return;
    
    const config = RATE_LIMIT_CONFIGS[operationType] || RATE_LIMIT_CONFIGS['click'];
    const windowStart = Date.now() - config.windowMs * 2;
    
    const filtered = history.filter(t => t > windowStart);
    this.operations.set(operationType, filtered);
  }
}
```

## 5.1.7 权限边界执行器

```typescript
class PermissionBoundaryExecutor {
  private config: PermissionBoundaryConfig;
  private permissionChecker: PermissionChecker;
  private rateLimiter: RateLimiter;
  private auditLogger: AuditLogger;
  private confirmationHandler: ConfirmationHandler;
  
  constructor(config: PermissionBoundaryConfig = DEFAULT_PERMISSION_CONFIG) {
    this.config = config;
    this.permissionChecker = new PermissionChecker();
    this.rateLimiter = new RateLimiter();
    this.auditLogger = new AuditLogger();
    this.confirmationHandler = new ConfirmationHandler();
  }
  
  async executeAction(
    action: RecoveryAction,
    executor: () => Promise<void>
  ): Promise<{
    success: boolean;
    reason: string;
  }> {
    if (!this.config.enabled) {
      await executor();
      return { success: true, reason: '权限边界已禁用' };
    }
    
    const permissionResult = this.permissionChecker.checkPermission(action.buttonType);
    
    if (!permissionResult.allowed) {
      this.auditLogger.log({
        workspaceId: action.workspaceId,
        taskId: action.taskId,
        action: action.type,
        buttonType: action.buttonType,
        selector: action.selector || '',
        result: 'skipped',
        reason: permissionResult.reason,
        durationMs: 0
      });
      
      return { success: false, reason: permissionResult.reason };
    }
    
    const rateLimitResult = await this.rateLimiter.checkRateLimit(action.type);
    
    if (!rateLimitResult.allowed) {
      return { success: false, reason: rateLimitResult.reason };
    }
    
    if (permissionResult.requiresConfirmation) {
      const confirmed = await this.confirmationHandler.requestConfirmation(
        action,
        this.config.confirmationStrategy
      );
      
      if (!confirmed) {
        return { success: false, reason: '用户未确认' };
      }
    }
    
    const startTime = Date.now();
    
    try {
      await executor();
      
      this.permissionChecker.recordClick(action.buttonType);
      this.rateLimiter.recordOperation(action.type);
      
      this.auditLogger.log({
        workspaceId: action.workspaceId,
        taskId: action.taskId,
        action: action.type,
        buttonType: action.buttonType,
        selector: action.selector || '',
        result: 'success',
        reason: '执行成功',
        durationMs: Date.now() - startTime
      });
      
      return { success: true, reason: '执行成功' };
    } catch (error) {
      this.auditLogger.log({
        workspaceId: action.workspaceId,
        taskId: action.taskId,
        action: action.type,
        buttonType: action.buttonType,
        selector: action.selector || '',
        result: 'failed',
        reason: error instanceof Error ? error.message : '未知错误',
        durationMs: Date.now() - startTime
      });
      
      return { success: false, reason: error instanceof Error ? error.message : '执行失败' };
    }
  }
}
```
