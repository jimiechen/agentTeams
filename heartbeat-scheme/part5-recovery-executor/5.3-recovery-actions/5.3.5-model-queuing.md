# 5.3.5 Model Queuing 恢复动作

## 5.3.5.1 问题定义

**model-queuing** 是指 AI 模型请求进入排队状态，显示"排在第 X 位"等提示。这实际上不是异常，而是服务容量限制的正常表现。

### 识别特征

| 特征 | 说明 |
|------|------|
| **触发条件** | 检测到"排在第 X 位"文本 |
| **检测层级** | Layer 2（内容心跳） |
| **时效要求** | 无需立即干预 |
| **自治可行性** | ❌ 只能等待 |

### 状态判定

```typescript
interface ModelQueuingSignal {
  queuePosition: number;
  estimatedWaitTime: number;
  queueMessage: string;
  hasCancelBtn: boolean;
}

function isModelQueuing(signal: ModelQueuingSignal): boolean {
  return signal.queuePosition > 0 || 
         signal.queueMessage.includes('排在') ||
         signal.queueMessage.includes('queue');
}
```

## 5.3.5.3 排队状态检测

```typescript
const QUEUING_PATTERNS = {
  position: [
    /排在第\s*(\d+)\s*位/,
    /queue position[:\s]*(\d+)/i,
    /position[:\s]*(\d+)\s*in\s*queue/i,
    /您前面还有\s*(\d+)\s*人/
  ],
  waitTime: [
    /预计等待[:\s]*(\d+)\s*分钟/,
    /estimated wait[:\s]*(\d+)\s*min/i,
    /等待时间约[:\s]*(\d+)\s*秒/
  ]
};

class QueuingDetector {
  private page: Page;
  
  constructor(page: Page) {
    this.page = page;
  }

  async detect(): Promise<ModelQueuingSignal> {
    const content = await this.getPageContent();
    
    const position = this.extractPosition(content);
    const waitTime = this.extractWaitTime(content);
    const hasCancelBtn = await this.hasCancelButton();
    
    return {
      queuePosition: position,
      estimatedWaitTime: waitTime,
      queueMessage: content,
      hasCancelBtn
    };
  }
  
  private async getPageContent(): Promise<string> {
    try {
      const statusElement = this.page.locator('[data-testid="queue-status"]').first();
      return await statusElement.textContent() || '';
    } catch {
      return '';
    }
  }
  
  private extractPosition(content: string): number {
    for (const pattern of QUEUING_PATTERNS.position) {
      const match = content.match(pattern);
      if (match) {
        return parseInt(match[1], 10);
      }
    }
    return 0;
  }
  
  private extractWaitTime(content: string): number {
    for (const pattern of QUEUING_PATTERNS.waitTime) {
      const match = content.match(pattern);
      if (match) {
        return parseInt(match[1], 10);
      }
    }
    return 0;
  }
  
  private async hasCancelButton(): Promise<boolean> {
    try {
      const cancelBtn = this.page.locator('button:has-text("取消")').first();
      return await cancelBtn.isVisible({ timeout: 1000 });
    } catch {
      return false;
    }
  }
}
```

## 5.3.5.4 排队状态处理器

```typescript
class ModelQueuingHandler {
  private detector: QueuingDetector;
  private stateTracker: QueuingStateTracker;
  private heartbeatAdjuster: HeartbeatAdjuster;
  
  constructor(page: Page) {
    this.detector = new QueuingDetector(page);
    this.stateTracker = new QueuingStateTracker();
    this.heartbeatAdjuster = new HeartbeatAdjuster();
  }

  async handle(context: RecoveryContext): Promise<QueuingResult> {
    console.log(`[ModelQueuing] 检测到排队状态: ${context.workspaceId}`);
    
    const signal = await this.detector.detect();
    
    if (signal.queuePosition === 0) {
      return {
        handled: true,
        stillQueuing: false,
        message: '未检测到排队状态'
      };
    }
    
    this.stateTracker.record(context.workspaceId, signal);
    
    this.heartbeatAdjuster.reduceFrequency(context.workspaceId, 'queuing');
    
    return {
      handled: true,
      stillQueuing: true,
      position: signal.queuePosition,
      estimatedWaitTime: signal.estimatedWaitTime,
      message: `排队中，位置: ${signal.queuePosition}`
    };
  }
  
  async checkProgress(context: RecoveryContext): Promise<QueuingProgress> {
    const signal = await this.detector.detect();
    const lastState = this.stateTracker.getLastState(context.workspaceId);
    
    if (signal.queuePosition === 0) {
      this.heartbeatAdjuster.restoreFrequency(context.workspaceId);
      this.stateTracker.clear(context.workspaceId);
      
      return {
        completed: true,
        previousPosition: lastState?.queuePosition || 0,
        waitTimeMs: lastState ? 
          Date.now() - new Date(lastState.timestamp).getTime() : 0
      };
    }
    
    return {
      completed: false,
      previousPosition: lastState?.queuePosition || 0,
      currentPosition: signal.queuePosition,
      positionDelta: (lastState?.queuePosition || 0) - signal.queuePosition
    };
  }
}

interface QueuingResult {
  handled: boolean;
  stillQueuing: boolean;
  position?: number;
  estimatedWaitTime?: number;
  message: string;
}

interface QueuingProgress {
  completed: boolean;
  previousPosition: number;
  currentPosition?: number;
  positionDelta?: number;
  waitTimeMs?: number;
}
```

## 5.3.5.5 排队状态追踪器

```typescript
interface QueuingState {
  workspaceId: string;
  queuePosition: number;
  estimatedWaitTime: number;
  timestamp: string;
  history: QueuingHistoryEntry[];
}

interface QueuingHistoryEntry {
  timestamp: string;
  position: number;
  waitTime: number;
}

class QueuingStateTracker {
  private states: Map<string, QueuingState> = new Map();
  
  record(workspaceId: string, signal: ModelQueuingSignal): void {
    const existing = this.states.get(workspaceId);
    const entry: QueuingHistoryEntry = {
      timestamp: new Date().toISOString(),
      position: signal.queuePosition,
      waitTime: signal.estimatedWaitTime
    };
    
    if (existing) {
      existing.history.push(entry);
      existing.queuePosition = signal.queuePosition;
      existing.estimatedWaitTime = signal.estimatedWaitTime;
      existing.timestamp = entry.timestamp;
    } else {
      this.states.set(workspaceId, {
        workspaceId,
        queuePosition: signal.queuePosition,
        estimatedWaitTime: signal.estimatedWaitTime,
        timestamp: entry.timestamp,
        history: [entry]
      });
    }
  }
  
  getLastState(workspaceId: string): QueuingState | null {
    return this.states.get(workspaceId) || null;
  }
  
  clear(workspaceId: string): void {
    const state = this.states.get(workspaceId);
    if (state) {
      this.writeToLog(state);
      this.states.delete(workspaceId);
    }
  }
  
  private writeToLog(state: QueuingState): void {
    const logPath = `logs/queuing-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, JSON.stringify(state) + '\n');
  }
  
  getStats(): {
    activeQueues: number;
    avgPosition: number;
    avgWaitTime: number;
  } {
    const states = Array.from(this.states.values());
    
    const avgPosition = states.length > 0 ?
      states.reduce((sum, s) => sum + s.queuePosition, 0) / states.length : 0;
    
    const avgWaitTime = states.length > 0 ?
      states.reduce((sum, s) => sum + s.estimatedWaitTime, 0) / states.length : 0;
    
    return {
      activeQueues: states.length,
      avgPosition,
      avgWaitTime
    };
  }
}
```

## 5.3.5.6 心跳频率调整器

```typescript
interface HeartbeatFrequency {
  layer1: number;
  layer2: number;
  layer3: number;
}

const FREQUENCY_PRESETS: Record<string, HeartbeatFrequency> = {
  normal: {
    layer1: 5000,
    layer2: 15000,
    layer3: 60000
  },
  queuing: {
    layer1: 10000,
    layer2: 30000,
    layer3: 120000
  },
  accelerated: {
    layer1: 2000,
    layer2: 5000,
    layer3: 30000
  }
};

class HeartbeatAdjuster {
  private currentFrequencies: Map<string, HeartbeatFrequency> = new Map();
  
  reduceFrequency(workspaceId: string, preset: string): void {
    const frequency = FREQUENCY_PRESETS[preset] || FREQUENCY_PRESETS.queuing;
    this.currentFrequencies.set(workspaceId, frequency);
    
    console.log(`[HeartbeatAdjuster] 降低心跳频率: ${workspaceId}`, frequency);
  }
  
  restoreFrequency(workspaceId: string): void {
    this.currentFrequencies.set(workspaceId, FREQUENCY_PRESETS.normal);
    
    console.log(`[HeartbeatAdjuster] 恢复正常心跳频率: ${workspaceId}`);
  }
  
  accelerateFrequency(workspaceId: string): void {
    this.currentFrequencies.set(workspaceId, FREQUENCY_PRESETS.accelerated);
    
    console.log(`[HeartbeatAdjuster] 加速心跳频率: ${workspaceId}`);
  }
  
  getFrequency(workspaceId: string): HeartbeatFrequency {
    return this.currentFrequencies.get(workspaceId) || FREQUENCY_PRESETS.normal;
  }
}
```

## 5.3.5.7 PMCLI/DEVCLI 策略差异

### PMCLI 策略

```typescript
const PMCLI_QUEUING_CONFIG = {
  reduceHeartbeat: true,
  heartbeatPreset: 'queuing',
  notifyOnQueue: false,
  notifyOnDequeue: false,
  maxWaitTime: 600000,
  cancelOnTimeout: false,
  logLevel: 'debug'
};
```

**PMCLI 特点**：
- 降低心跳频率
- 不发送排队通知
- 最大等待 10 分钟
- 不自动取消

### DEVCLI 策略

```typescript
const DEVCLI_QUEUING_CONFIG = {
  reduceHeartbeat: true,
  heartbeatPreset: 'queuing',
  notifyOnQueue: true,
  notifyOnDequeue: true,
  maxWaitTime: 300000,
  cancelOnTimeout: false,
  logLevel: 'info',
  preserveContext: true
};
```

**DEVCLI 特点**：
- 降低心跳频率
- 发送排队/出队通知
- 最大等待 5 分钟
- 保留上下文

## 5.3.5.9 日志与监控

```typescript
interface ModelQueuingLog {
  timestamp: string;
  workspaceId: string;
  taskId: string;
  queuePosition: number;
  estimatedWaitTime: number;
  actualWaitTime?: number;
  completed: boolean;
}

class ModelQueuingLogger {
  private logs: ModelQueuingLog[] = [];
  
  logEntry(entry: Omit<ModelQueuingLog, 'timestamp'>): void {
    const log: ModelQueuingLog = {
      ...entry,
      timestamp: new Date().toISOString()
    };
    
    this.logs.push(log);
    this.writeToFile(log);
  }
  
  logCompletion(workspaceId: string, actualWaitTime: number): void {
    const lastLog = this.logs
      .filter(l => l.workspaceId === workspaceId && !l.completed)
      .pop();
    
    if (lastLog) {
      lastLog.actualWaitTime = actualWaitTime;
      lastLog.completed = true;
      this.writeToFile(lastLog);
    }
  }
  
  private writeToFile(log: ModelQueuingLog): void {
    const logPath = `logs/model-queuing-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, JSON.stringify(log) + '\n');
  }
  
  getStats(): {
    totalQueues: number;
    avgWaitTime: number;
    avgPosition: number;
    completedRate: number;
  } {
    const total = this.logs.length;
    const completed = this.logs.filter(l => l.completed).length;
    
    const waitTimes = this.logs
      .filter(l => l.actualWaitTime !== undefined)
      .map(l => l.actualWaitTime!);
    
    const avgWaitTime = waitTimes.length > 0 ?
      waitTimes.reduce((a, b) => a + b, 0) / waitTimes.length : 0;
    
    const avgPosition = total > 0 ?
      this.logs.reduce((sum, l) => sum + l.queuePosition, 0) / total : 0;
    
    return {
      totalQueues: total,
      avgWaitTime,
      avgPosition,
      completedRate: total > 0 ? completed / total : 0
    };
  }
}
```
