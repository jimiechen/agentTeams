# 5.3.4 Task Interrupted 恢复动作

## 5.3.4.1 问题定义

**task-interrupted** 是指任务进入终态中断状态。这是一种不可自动恢复的异常，需要人工介入处理。

### 识别特征

| 特征 | 说明 |
|------|------|
| **触发条件** | 任务状态变为"中断" |
| **检测层级** | Layer 1（快速心跳） |
| **时效要求** | 立即识别 |
| **自治可行性** | ❌ 需要人工介入 |

### 状态判定

```typescript
interface TaskInterruptedSignal {
  taskStatus: 'interrupted' | 'cancelled' | 'error';
  errorMessage?: string;
  errorStack?: string;
  canResume: boolean;
  hasRecoveryOptions: boolean;
}

function isTaskInterrupted(signal: TaskInterruptedSignal): boolean {
  return signal.taskStatus === 'interrupted' ||
         signal.taskStatus === 'cancelled' ||
         signal.taskStatus === 'error';
}
```

## 5.3.4.3 中断类型分类

```typescript
enum InterruptType {
  USER_CANCELLED = 'user-cancelled',
  SYSTEM_ERROR = 'system-error',
  TIMEOUT = 'timeout',
  RESOURCE_EXHAUSTED = 'resource-exhausted',
  NETWORK_ERROR = 'network-error',
  UNKNOWN = 'unknown'
}

interface InterruptDetails {
  type: InterruptType;
  timestamp: string;
  taskId: string;
  workspaceId: string;
  errorMessage: string;
  errorStack?: string;
  context: InterruptContext;
  recoveryOptions: RecoveryOption[];
}

interface RecoveryOption {
  id: string;
  label: string;
  description: string;
  action: () => Promise<void>;
  riskLevel: 'low' | 'medium' | 'high';
}
```

## 5.3.4.4 中断处理器实现

```typescript
class TaskInterruptedHandler {
  private alertService: AlertService;
  private contextCollector: ContextCollector;
  private recoveryOptionProvider: RecoveryOptionProvider;
  
  constructor() {
    this.alertService = new AlertService();
    this.contextCollector = new ContextCollector();
    this.recoveryOptionProvider = new RecoveryOptionProvider();
  }

  async handle(
    page: Page,
    signal: TaskInterruptedSignal,
    context: RecoveryContext
  ): Promise<InterruptedResult> {
    console.log(`[TaskInterrupted] 检测到任务中断: ${context.workspaceId}`);
    
    const details = await this.collectDetails(page, signal, context);
    
    await this.sendAlert(details);
    
    if (signal.canResume) {
      const options = await this.recoveryOptionProvider.getOptions(details);
      return {
        handled: true,
        requiresManualIntervention: true,
        details,
        recoveryOptions: options
      };
    }
    
    return {
      handled: true,
      requiresManualIntervention: true,
      details,
      recoveryOptions: []
    };
  }

  private async collectDetails(
    page: Page,
    signal: TaskInterruptedSignal,
    context: RecoveryContext
  ): Promise<InterruptDetails> {
    const type = this.determineInterruptType(signal);
    const errorContext = await this.contextCollector.collect(page, context);
    
    return {
      type,
      timestamp: new Date().toISOString(),
      taskId: context.taskId,
      workspaceId: context.workspaceId,
      errorMessage: signal.errorMessage || '未知错误',
      errorStack: signal.errorStack,
      context: errorContext,
      recoveryOptions: []
    };
  }

  private determineInterruptType(signal: TaskInterruptedSignal): InterruptType {
    if (signal.taskStatus === 'cancelled') {
      return InterruptType.USER_CANCELLED;
    }
    
    if (signal.errorMessage) {
      if (signal.errorMessage.includes('timeout')) {
        return InterruptType.TIMEOUT;
      }
      if (signal.errorMessage.includes('network') || signal.errorMessage.includes('connection')) {
        return InterruptType.NETWORK_ERROR;
      }
      if (signal.errorMessage.includes('resource') || signal.errorMessage.includes('limit')) {
        return InterruptType.RESOURCE_EXHAUSTED;
      }
    }
    
    return InterruptType.UNKNOWN;
  }

  private async sendAlert(details: InterruptDetails): Promise<void> {
    await this.alertService.send({
      level: 'critical',
      title: `任务中断: ${details.workspaceId}`,
      message: details.errorMessage,
      details: {
        type: details.type,
        taskId: details.taskId,
        timestamp: details.timestamp
      }
    });
  }
}

interface InterruptedResult {
  handled: boolean;
  requiresManualIntervention: boolean;
  details: InterruptDetails;
  recoveryOptions: RecoveryOption[];
}
```

## 5.3.4.5 上下文收集器

```typescript
interface InterruptContext {
  url: string;
  conversationLength: number;
  lastUserMessage: string;
  lastAssistantMessage: string;
  codeBlocks: string[];
  filesModified: string[];
  terminalOutput: string;
  screenshot?: string;
}

class ContextCollector {
  async collect(page: Page, context: RecoveryContext): Promise<InterruptContext> {
    const url = page.url();
    
    const conversationLength = await this.getConversationLength(page);
    const lastUserMessage = await this.getLastMessage(page, 'user');
    const lastAssistantMessage = await this.getLastMessage(page, 'assistant');
    const codeBlocks = await this.getCodeBlocks(page);
    const filesModified = await this.getModifiedFiles(page);
    const terminalOutput = await this.getTerminalOutput(page);
    
    return {
      url,
      conversationLength,
      lastUserMessage,
      lastAssistantMessage,
      codeBlocks,
      filesModified,
      terminalOutput
    };
  }
  
  private async getConversationLength(page: Page): Promise<number> {
    try {
      const messages = await page.locator('[data-testid="message"]').count();
      return messages;
    } catch {
      return 0;
    }
  }
  
  private async getLastMessage(page: Page, role: 'user' | 'assistant'): Promise<string> {
    try {
      const selector = `[data-testid="message"][data-role="${role}"]`;
      const lastMessage = page.locator(selector).last();
      return await lastMessage.textContent() || '';
    } catch {
      return '';
    }
  }
  
  private async getCodeBlocks(page: Page): Promise<string[]> {
    try {
      const blocks = await page.locator('pre code').allTextContents();
      return blocks;
    } catch {
      return [];
    }
  }
  
  private async getModifiedFiles(page: Page): Promise<string[]> {
    try {
      const files = await page.locator('[data-testid="modified-file"]').allTextContents();
      return files;
    } catch {
      return [];
    }
  }
  
  private async getTerminalOutput(page: Page): Promise<string> {
    try {
      const terminal = page.locator('[data-testid="terminal-output"]').first();
      return await terminal.textContent() || '';
    } catch {
      return '';
    }
  }
}
```

## 5.3.4.6 恢复选项提供器

```typescript
class RecoveryOptionProvider {
  async getOptions(details: InterruptDetails): Promise<RecoveryOption[]> {
    const options: RecoveryOption[] = [];
    
    switch (details.type) {
      case InterruptType.TIMEOUT:
        options.push({
          id: 'retry-with-longer-timeout',
          label: '使用更长超时重试',
          description: '增加超时时间后重新执行任务',
          action: async () => { },
          riskLevel: 'low'
        });
        break;
        
      case InterruptType.NETWORK_ERROR:
        options.push({
          id: 'retry-after-network-check',
          label: '网络恢复后重试',
          description: '等待网络稳定后重新执行任务',
          action: async () => { },
          riskLevel: 'low'
        });
        break;
        
      case InterruptType.RESOURCE_EXHAUSTED:
        options.push({
          id: 'wait-and-retry',
          label: '等待资源释放后重试',
          description: '等待系统资源释放后重新执行',
          action: async () => { },
          riskLevel: 'low'
        });
        break;
        
      case InterruptType.USER_CANCELLED:
        options.push({
          id: 'resume-from-checkpoint',
          label: '从检查点恢复',
          description: '从上次保存的检查点继续执行',
          action: async () => { },
          riskLevel: 'medium'
        });
        break;
    }
    
    options.push({
      id: 'restart-task',
      label: '重新开始任务',
      description: '从头开始执行任务（可能丢失进度）',
      action: async () => { },
      riskLevel: 'high'
    });
    
    return options;
  }
}
```

## 5.3.4.7 PMCLI/DEVCLI 策略差异

### PMCLI 策略

```typescript
const PMCLI_INTERRUPTED_CONFIG = {
  alertLevel: 'warning',
  collectContext: true,
  saveCheckpoint: true,
  notifyChannels: ['email', 'dashboard'],
  autoRetryOnTransientErrors: false,
  maxAutoRetryDelay: 0
};
```

**PMCLI 特点**：
- 告警级别 warning
- 保存检查点
- 通知渠道：邮件 + 仪表板
- 不自动重试

### DEVCLI 策略

```typescript
const DEVCLI_INTERRUPTED_CONFIG = {
  alertLevel: 'critical',
  collectContext: true,
  saveCheckpoint: true,
  preserveCodeState: true,
  notifyChannels: ['email', 'dashboard', 'slack'],
  autoRetryOnTransientErrors: true,
  maxAutoRetryDelay: 60000,
  rollbackOnFailure: true
};
```

**DEVCLI 特点**：
- 告警级别 critical
- 保存代码状态
- 多渠道通知
- 对瞬态错误自动重试
- 失败时回滚

## 5.3.4.8 告警服务

```typescript
interface AlertPayload {
  level: 'info' | 'warning' | 'critical';
  title: string;
  message: string;
  details: Record<string, any>;
}

class AlertService {
  private readonly WEBHOOK_URL = process.env.ALERT_WEBHOOK_URL;
  
  async send(payload: AlertPayload): Promise<void> {
    console.log(`[AlertService] 发送 ${payload.level} 告警: ${payload.title}`);
    
    const alert = {
      ...payload,
      timestamp: new Date().toISOString(),
      source: 'heartbeat-recovery'
    };
    
    if (this.WEBHOOK_URL) {
      try {
        await fetch(this.WEBHOOK_URL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(alert)
        });
      } catch (error) {
        console.error('[AlertService] 发送告警失败:', error);
      }
    }
    
    this.writeToAlertLog(alert);
  }
  
  private writeToAlertLog(alert: any): void {
    const logPath = `logs/alerts-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, JSON.stringify(alert) + '\n');
  }
}
```

## 5.3.4.9 日志与监控

```typescript
interface TaskInterruptedLog {
  timestamp: string;
  workspaceId: string;
  taskId: string;
  interruptType: InterruptType;
  errorMessage: string;
  context: InterruptContext;
  recoveryOptionsProvided: number;
  userActionTaken: boolean;
}

class TaskInterruptedLogger {
  private logs: TaskInterruptedLog[] = [];
  
  log(entry: Omit<TaskInterruptedLog, 'timestamp'>): void {
    const log: TaskInterruptedLog = {
      ...entry,
      timestamp: new Date().toISOString()
    };
    
    this.logs.push(log);
    this.writeToFile(log);
  }
  
  private writeToFile(log: TaskInterruptedLog): void {
    const logPath = `logs/task-interrupted-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, JSON.stringify(log) + '\n');
  }
  
  getStats(): {
    total: number;
    byType: Record<InterruptType, number>;
    recoveryRate: number;
  } {
    const total = this.logs.length;
    const recovered = this.logs.filter(l => l.userActionTaken).length;
    
    const byType: Record<InterruptType, number> = {} as any;
    for (const log of this.logs) {
      byType[log.interruptType] = (byType[log.interruptType] || 0) + 1;
    }
    
    return {
      total,
      byType,
      recoveryRate: total > 0 ? recovered / total : 0
    };
  }
}
```

## 5.3.4.10 测试用例

```typescript
describe('TaskInterruptedHandler', () => {
  it('应正确识别用户取消类型', async () => {
    const handler = new TaskInterruptedHandler();
    
    const signal: TaskInterruptedSignal = {
      taskStatus: 'cancelled',
      canResume: true,
      hasRecoveryOptions: true
    };
    
    const type = handler['determineInterruptType'](signal);
    
    expect(type).toBe(InterruptType.USER_CANCELLED);
  });
  
  it('应正确识别超时类型', async () => {
    const handler = new TaskInterruptedHandler();
    
    const signal: TaskInterruptedSignal = {
      taskStatus: 'error',
      errorMessage: 'Request timeout after 30s',
      canResume: true,
      hasRecoveryOptions: true
    };
    
    const type = handler['determineInterruptType'](signal);
    
    expect(type).toBe(InterruptType.TIMEOUT);
  });
  
  it('应发送 critical 级别告警', async () => {
    const handler = new TaskInterruptedHandler();
    
    const details: InterruptDetails = {
      type: InterruptType.SYSTEM_ERROR,
      timestamp: new Date().toISOString(),
      taskId: 'task-1',
      workspaceId: 'ws-1',
      errorMessage: 'System error',
      context: {} as any,
      recoveryOptions: []
    };
    
    await handler['sendAlert'](details);
  });
  
  it('应为瞬态错误提供恢复选项', async () => {
    const provider = new RecoveryOptionProvider();
    
    const details: InterruptDetails = {
      type: InterruptType.NETWORK_ERROR,
      timestamp: new Date().toISOString(),
      taskId: 'task-1',
      workspaceId: 'ws-1',
      errorMessage: 'Network connection lost',
      context: {} as any,
      recoveryOptions: []
    };
    
    const options = await provider.getOptions(details);
    
    expect(options.length).toBeGreaterThan(0);
    expect(options.some(o => o.id === 'retry-after-network-check')).toBe(true);
  });
});
```
