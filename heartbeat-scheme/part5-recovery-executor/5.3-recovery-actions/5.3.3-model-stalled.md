# 5.3.3 Model Stalled 恢复动作

## 5.3.3.1 问题定义

**model-stalled** 是指 AI 模型推理过程中出现的停滞状态。这是一种隐式异常，需要通过时间窗口内的文本增长检测来判定。

### 识别特征

| 特征 | 说明 |
|------|------|
| **触发条件** | 连续多次检测到文本无增长 |
| **检测层级** | Layer 2（内容心跳） |
| **时效要求** | 30 秒内识别 |
| **自治可行性** | ✅ stop + retry |

### 判定条件

```typescript
interface ModelStalledSignal {
  consecutiveZeroDeltas: number;
  lastContentLength: number;
  currentContentLength: number;
  timeSinceLastGrowth: number;
  hasStopBtn: boolean;
}

function isModelStalled(signal: ModelStalledSignal, config: StalledConfig): boolean {
  return signal.consecutiveZeroDeltas >= config.minZeroDeltas &&
         signal.timeSinceLastGrowth >= config.minStallTime &&
         signal.hasStopBtn;
}
```

## 5.3.3.4 恢复执行器实现

```typescript
class ModelStalledRecoveryExecutor implements ActionExecutor {
  readonly type = 'model-stalled';
  readonly action = 'stop-retry';
  
  private page: Page;
  private buttonFinder: ButtonFinder;
  private permissionChecker: PermissionChecker;
  private retryCounter: RetryCounter;
  
  constructor(page: Page) {
    this.page = page;
    this.buttonFinder = new ButtonFinder(page);
    this.permissionChecker = new PermissionChecker();
    this.retryCounter = new RetryCounter();
  }

  async execute(context: RecoveryContext): Promise<RecoveryResult> {
    const startTime = Date.now();
    
    console.log(`[ModelStalled] 开始恢复工作区 ${context.workspaceId}`);
    
    const retryState = this.retryCounter.getState(context.workspaceId);
    
    if (retryState.currentAttempts >= retryState.maxAttempts) {
      return {
        success: false,
        action: this.action,
        reason: `已达到最大重试次数 (${retryState.maxAttempts})`,
        durationMs: Date.now() - startTime,
        nextState: 'interrupted'
      };
    }
    
    const stopPermission = this.permissionChecker.checkPermission('stop');
    if (!stopPermission.allowed) {
      return {
        success: false,
        action: 'stop',
        reason: stopPermission.reason,
        durationMs: Date.now() - startTime,
        nextState: 'model-stalled'
      };
    }
    
    const stopResult = await this.executeStop();
    
    if (!stopResult.success) {
      return stopResult;
    }
    
    await this.sleep(2000);
    
    const retryResult = await this.executeRetry(context);
    
    if (retryResult.success) {
      this.retryCounter.increment(context.workspaceId);
    }
    
    return retryResult;
  }

  private async executeStop(): Promise<RecoveryResult> {
    const startTime = Date.now();
    
    const stopBtn = await this.buttonFinder.findButton('stop');
    
    if (!stopBtn.found || !stopBtn.element) {
      return {
        success: false,
        action: 'stop',
        reason: '未找到"停止"按钮',
        durationMs: Date.now() - startTime,
        nextState: 'model-stalled'
      };
    }
    
    try {
      await stopBtn.element.click();
      this.permissionChecker.recordClick('stop');
      
      console.log('[ModelStalled] 已点击"停止"按钮');
      
      return {
        success: true,
        action: 'stop',
        reason: '成功点击"停止"按钮',
        durationMs: Date.now() - startTime,
        nextState: 'model-stalled'
      };
    } catch (error) {
      return {
        success: false,
        action: 'stop',
        reason: `点击失败: ${error instanceof Error ? error.message : '未知错误'}`,
        durationMs: Date.now() - startTime,
        nextState: 'model-stalled'
      };
    }
  }

  private async executeRetry(context: RecoveryContext): Promise<RecoveryResult> {
    const startTime = Date.now();
    
    const retryBtn = await this.buttonFinder.findButton('retry');
    
    if (!retryBtn.found || !retryBtn.element) {
      const regenerateBtn = await this.buttonFinder.findButton('regenerate');
      
      if (!regenerateBtn.found || !regenerateBtn.element) {
        return {
          success: false,
          action: 'retry',
          reason: '未找到"重试"或"重新生成"按钮',
          durationMs: Date.now() - startTime,
          nextState: 'model-stalled'
        };
      }
      
      return await this.clickButton(regenerateBtn.element, 'regenerate', startTime);
    }
    
    return await this.clickButton(retryBtn.element, 'retry', startTime);
  }

  private async clickButton(
    element: Locator,
    buttonType: string,
    startTime: number
  ): Promise<RecoveryResult> {
    try {
      await element.click();
      this.permissionChecker.recordClick(buttonType);
      
      console.log(`[ModelStalled] 已点击"${buttonType}"按钮`);
      
      return {
        success: true,
        action: `click-${buttonType}`,
        reason: `成功点击"${buttonType}"按钮`,
        durationMs: Date.now() - startTime,
        nextState: 'verifying'
      };
    } catch (error) {
      return {
        success: false,
        action: buttonType,
        reason: `点击失败: ${error instanceof Error ? error.message : '未知错误'}`,
        durationMs: Date.now() - startTime,
        nextState: 'model-stalled'
      };
    }
  }

  async canExecute(context: RecoveryContext): Promise<boolean> {
    const stopBtn = await this.buttonFinder.findButton('stop');
    return stopBtn.found;
  }
  
  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

## 5.3.3.5 重试计数器

```typescript
interface RetryState {
  workspaceId: string;
  currentAttempts: number;
  maxAttempts: number;
  lastAttemptTime: string;
  lastSuccessTime: string | null;
}

class RetryCounter {
  private states: Map<string, RetryState> = new Map();
  private readonly DEFAULT_MAX_ATTEMPTS = 3;
  private readonly RESET_AFTER_MS = 300000;
  
  getState(workspaceId: string): RetryState {
    const existing = this.states.get(workspaceId);
    
    if (existing) {
      if (this.shouldReset(existing)) {
        return this.resetState(workspaceId);
      }
      return existing;
    }
    
    return this.createState(workspaceId);
  }
  
  increment(workspaceId: string): void {
    const state = this.getState(workspaceId);
    state.currentAttempts++;
    state.lastAttemptTime = new Date().toISOString();
    this.states.set(workspaceId, state);
  }
  
  reset(workspaceId: string): void {
    this.resetState(workspaceId);
  }
  
  markSuccess(workspaceId: string): void {
    const state = this.getState(workspaceId);
    state.lastSuccessTime = new Date().toISOString();
    state.currentAttempts = 0;
    this.states.set(workspaceId, state);
  }
  
  private shouldReset(state: RetryState): boolean {
    if (state.lastSuccessTime) {
      const timeSinceSuccess = Date.now() - new Date(state.lastSuccessTime).getTime();
      return timeSinceSuccess > this.RESET_AFTER_MS;
    }
    return false;
  }
  
  private createState(workspaceId: string): RetryState {
    const state: RetryState = {
      workspaceId,
      currentAttempts: 0,
      maxAttempts: this.DEFAULT_MAX_ATTEMPTS,
      lastAttemptTime: new Date().toISOString(),
      lastSuccessTime: null
    };
    this.states.set(workspaceId, state);
    return state;
  }
  
  private resetState(workspaceId: string): RetryState {
    const state = this.createState(workspaceId);
    this.states.set(workspaceId, state);
    return state;
  }
}
```

## 5.3.3.7 验证逻辑

```typescript
class ModelStalledVerifier {
  private readonly VERIFY_WINDOW_MS = 60000;
  private readonly CHECK_INTERVAL_MS = 5000;
  private readonly SUCCESS_THRESHOLD = 10;
  
  private lastCharCount: number = 0;
  
  async verify(page: Page): Promise<boolean> {
    const startTime = Date.now();
    this.lastCharCount = await this.getCharCount(page);
    
    while (Date.now() - startTime < this.VERIFY_WINDOW_MS) {
      await this.sleep(this.CHECK_INTERVAL_MS);
      
      const currentCharCount = await this.getCharCount(page);
      const charsDelta = currentCharCount - this.lastCharCount;
      
      if (charsDelta >= this.SUCCESS_THRESHOLD) {
        console.log(`[ModelStalled] 检测到文本增长 ${charsDelta} 字符，恢复成功`);
        return true;
      }
      
      this.lastCharCount = currentCharCount;
    }
    
    console.log('[ModelStalled] 验证窗口超时，文本无增长');
    return false;
  }
  
  private async getCharCount(page: Page): Promise<number> {
    try {
      const contentElement = page.locator('[data-testid="response-content"]').first();
      const content = await contentElement.textContent({ timeout: 5000 });
      return content?.length || 0;
    } catch {
      return 0;
    }
  }
  
  private sleep(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

## 5.3.3.9 日志与监控

```typescript
interface ModelStalledLog {
  timestamp: string;
  workspaceId: string;
  taskId: string;
  action: 'stop-retry';
  attemptNumber: number;
  result: 'success' | 'failed' | 'exhausted';
  reason: string;
  durationMs: number;
  contentSnapshot: {
    beforeLength: number;
    afterLength: number;
    charsDelta: number;
  };
}

class ModelStalledLogger {
  private logs: ModelStalledLog[] = [];
  
  log(entry: Omit<ModelStalledLog, 'timestamp'>): void {
    const log: ModelStalledLog = {
      ...entry,
      timestamp: new Date().toISOString()
    };
    
    this.logs.push(log);
    this.writeToFile(log);
    
    if (entry.result === 'exhausted') {
      this.sendCriticalAlert(log);
    }
  }
  
  private writeToFile(log: ModelStalledLog): void {
    const logPath = `logs/model-stalled-${new Date().toISOString().split('T')[0]}.jsonl`;
    fs.appendFileSync(logPath, JSON.stringify(log) + '\n');
  }
  
  private sendCriticalAlert(log: ModelStalledLog): void {
    console.error(`[ModelStalled 严重告警] 工作区 ${log.workspaceId} 重试次数已耗尽`);
  }
  
  getStats(): {
    total: number;
    success: number;
    failed: number;
    exhausted: number;
    avgAttempts: number;
  } {
    const total = this.logs.length;
    const success = this.logs.filter(l => l.result === 'success').length;
    const failed = this.logs.filter(l => l.result === 'failed').length;
    const exhausted = this.logs.filter(l => l.result === 'exhausted').length;
    
    const totalAttempts = this.logs.reduce((sum, l) => sum + l.attemptNumber, 0);
    const avgAttempts = total > 0 ? totalAttempts / total : 0;
    
    return { total, success, failed, exhausted, avgAttempts };
  }
}
```
